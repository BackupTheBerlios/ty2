210:   public static Weapon createWeapon(int[][] list,int level) {
211:     // add a few bonus/negative quality levels
212:     int improve=RPG.r(2)*(RPG.d(3,3)-6);
213:     level-=RPG.sign(improve);
214: 
215:     do {level=level+RPG.d(2)-RPG.d(2);}
216:     while (RPG.d(2)==1);
217:     level=RPG.middle(0,level,list.length-1);
218:       
219:     Weapon w=createSpecificWeapon(list[level][RPG.r(list[level].length)]);
220: 
221:     // curse a few!
222:     if (RPG.d(100)<=level) {
223:       w.flags|=ITEM_CURSED;
224:       improve-=2; 
225:     }
226: 
227:     w.quality=RPG.middle(1,w.quality+improve,12);
228:     return w;
229:   }

235:     }

240:   // hit target with weapon
241:   public int use(Thing wielder, Thing target, int action) {
242:     switch (action) {
243:       case USE_HIT:
244:         int dam=super.use(wielder,target,action);
245:         
246:         Map map=target.getMap();
247: 
248:         boolean ishero=(target==Game.hero);
249:         
250:         // special weapon effects if 
251:         if (map!=null) switch (type) {
252:           case KRAZYKNIFE: case POISONDAGGER:
253:             if (dam>0) {
254:               target.addThing(new PoisonAttribute(2,RPG.DT_POISON,500*RPG.d(2,6),5));
255:               if (ishero) Game.message("You feel that you have been poisoned!");
256:             }
257:             break;
258:             
259:           case FLAMESWORD: case SHAMANSTAFF:
260:             Fire.createFire(map,target.x,target.y,2);
261:             break;
262:             
263:           case VAPOURSWORD:
264:             VAPOURSPELL.castAtLocation(null,map,target.x,target.y);
265:             break;  
266:             
267:           case ICESWORD:
268:             ICESPELL.castAtLocation(null,map,target.x,target.y);
269:         }
270:         
271:         return dam; 
272: 
273:       default:
274:           return super.use(wielder,target,action);     
275:       }
276:   }

280:   public int getStat(int s) {
281:     switch(s) {
282:       case RPG.ST_ATTACKCOST: return  attackcosts[type];
283:       case RPG.ST_ASKMULTIPLIER: return    skillmuls[type]*qualitymuls[quality]/100;
284:       case RPG.ST_ASTMULTIPLIER: return    strengthmuls[type]*qualitymuls[quality]/100;
285:       case RPG.ST_DSKMULTIPLIER: return    defencemuls[type]*qualitymuls[quality]/100;
286:       case RPG.ST_ASKBONUS: return    skillbonuses[type];
287:       case RPG.ST_ASTBONUS: return    strengthbonuses[type];
288:       case RPG.ST_DSKBONUS: return    defencebonuses[type];
289:       
290:       case RPG.ST_ITEMVALUE: return RPG.niceNumber(values[type])*Lib.qualityvalues[quality]/100;
291:       
292:       default: return super.getStat(s);
293:     }    
294:   }
295:   
296:   public int getModifier(int s) {
297:     switch(s) {
298:       case RPG.ST_DSKMULTIPLIER: 
299:         return defencemuls[type]*qualitymuls[quality]/100;
300:       case RPG.ST_DSKBONUS: 
301:         return defencebonuses[type];
302:       default: {
303:         // check for special bonus
304:         if (s==stats[type]) return bonuses[type]+super.getModifier(s); 
305:         
306:         return super.getModifier(s);
307:       }
308:     }  
309:   }

311:   // return toughness for the weapon
312:   // increases with higher quality
313:   public int getToughness() {
314:     if (quality>=13) return 0;
315:     return (toughnesses[type]*quality*quality)/10; 
316:   }
317:   
318:   // damage routine causes weapon to deteriorate
319:   public int damage(int dam, int damtype) {
320:     int tuff=toughnesses[type];
321:     if (tuff<=0) return 0;
322:     
323:     switch (damtype) {
324:       case RPG.DT_ACID:
325:         dam=dam*10; break;
326:     } 
327:     
328:     int det=RPG.po(dam*(13-quality),5*tuff*(quality+4));
329:     if ((quality<13)&&(det>0)) {
330:       quality-=det;
331:       if (place==Game.hero) {
332:         if (quality>0) {
333:           Game.message("Your "+getName()+" is damaged!"); 
334:         } else {
335:           die(); 
336:         }
337:       }
338:     }
339:     return dam; 
340:   }
