01: //
02: // AI class to guard a particulat region
03: //
04: package rl;
05: 
06: public class GuardAI extends BaseAI {
07:   int x1; 
08:   int y1; 
09:   int x2;
10:   int y2;
11:   Thing bound1;
12:   Thing bound2;
13:   
14:   public GuardAI(int p1, int q1, int p2, int q2) {
15:     super(); 
16:     setArea(p1,q1,p2,q2);
17:   }
18:   
19:   public GuardAI(Thing b1, Thing b2) {
20:     super();
21:     bound1=b1;
22:     bound2=b2; 
23:   }
24:   
25:   // set area to guard, reversing co-ordinates if necessary
26:   public void setArea(int p1, int q1, int p2, int q2) {
27:     if (p1<p2) {x1=p1; x2=p2;} else {x2=p1; x1=p2;};
28:     if (q1<q2) {y1=q1; y2=q2;} else {y2=q1; y1=q2;};
29:   }
30:   
31:   public boolean doAction(Mobile m) {
32:     if ((x1==0)&&(bound1!=null)) {
33:       setArea(bound1.getMapX(),bound1.getMapY(),bound2.getMapX(),bound2.getMapY());
34:     }
35:     
36:     int tx=(x1+x2)/2; 
37:     int ty=(y1+y2)/2;
38:     
39:     Map map=m.getMap();
40:     
41:     if (map==null) return false;
42:     
43:     Thing f=map.findNearestFoe(m);
44:     if ((f!=null)&&RPG.dist(m.x,m.y,f.x,f.y)<=8) {
45:       NastyCritterAI.instance.doAttack(m,f.x,f.y);
46:     } else if ((m.x<x1)||(m.x>x2)||(m.y<y1)||(m.y>y2)) {
47:       // outside area, need to get back in! 
48:       NastyCritterAI.instance.doAttack(m,tx,ty);
49:     } else {
50:       boolean changedirection=(RPG.d(6)==1)||((m.xdirection==0)&&(m.ydirection==0));
51:       if (changedirection) m.xdirection=RPG.r(3)-1;
52:       if (changedirection) m.ydirection=RPG.r(3)-1;
53:       NastyCritterAI.instance.doAttack(m,m.x+m.xdirection,m.y+m.ydirection);
54:     }
55:     return true;
56:   }
57: }
