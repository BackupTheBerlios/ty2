01: package rl;
02: 
03: public class Web extends Special {
04:     protected int image;
05:     protected int density;
06:     
07:     private static final Description DESC_WEB = 
08:         new Describer("web","A large spider web with sticky strands.");
09:     
10:   // new web of given density
11:   //   10 = weak
12:   //   25 = normal
13:   //   50 = strong
14:     public Web(int webdensity) {
15:       image=48;
16:     density=webdensity;
17:     }
18:     
19:     public int damage(int dam, int damtype) {
20:     switch (damtype) {
21:       case RPG.DT_NORMAL: case RPG.DT_IMPACT: case RPG.DT_FIRE:
22:         if (dam>=RPG.r(4)) remove();
23:         return dam; 
24:       default:
25:         return 0;
26:     } 
27:   }
28:   
29:   public int getImage() {return image;}
30:     
31:     public int getZ() {return Z_ITEM;}
32:     
33:     public void trigger(Map map, int tx, int ty) {
34:     if (map==null) return;
35:     Being b=(Being)map.getObject(tx,ty,Being.class);
36:     if (b!=null) {
37:       if ((b instanceof Creature)&&(((Creature)b).creaturetype==Creature.SPIDER)) return;
38:       remove();
39:       b.addAttribute(new TemporaryAttribute("Webbed",100*density,new int[] {RPG.ST_MOVECOST,density,RPG.ST_ATTACKCOST,density},"You untangle yourself"));
40:     }
41:   }
42: 
43:   public Description getDescription() {return DESC_WEB;}
44: }
