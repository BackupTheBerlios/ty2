163:   private void buildArea(int x, int y, int t) {
164:     if (t==-1) t=RPG.pick(OUTER);
165:     if (t==-2) t=RPG.pick(INNER);
166:     
167:     switch (t) {
168:       
169:       // wooded area
170:       case 0: {
171:         for (int i=0; i<20; i++) {
172:           addThing(GameScenery.create(3),x,y,x+15,y+15);
173:         } 
174:         addThing(new Person(Person.RANGER),x,y,x+15,y+15);
175:         
176:         break;
177:       }
178:       
179:       // hut
180:       case 3: {
181:         int hw=RPG.r(3)+3;  
182:         int hh=RPG.r(3)+3;  
183:         int hx=x+1+RPG.r(14-hw);
184:         int hy=y+1+RPG.r(14-hh);
185:         int dx=RPG.r(2)*2-1; 
186:         int dy=RPG.r(2)*2-1;
187:         if (RPG.d(2)==1) dx=0; else dy=0;
188:         addRoom(hx,hy,hx+hw,hy+hh,dx,dy);
189:         break;
190:       }
191:       
192:       // block
193:       case 4: {
194:         addRoom(x+8-RPG.d(2,3),y+8-RPG.d(2,3),x+8,y+8,0,-1); 
195:         addRoom(x+8-RPG.d(2,3),y+8,x+8,y+8+RPG.d(2,3),0,1); 
196:         addRoom(x+8,y+8-RPG.d(2,3),x+8+RPG.d(2,3),y+8,0,-1); 
197:         addRoom(x+8,y+8,x+8+RPG.d(2,3),y+8+RPG.d(2,3),0,1); 
198:         break;
199:       }
200:       
201:       // hall
202:       case 5: {
203:         int left=RPG.d(2,3);
204:         int right=RPG.d(2,3);
205:         fillArea(x+8-left,y+3,x+8+right,y+12,Tile.STONEFLOOR);
206:         fillBorder(x+8-left,y+3,x+8+right,y+12,Tile.STONEWALL);
207:         
208:         if (left<=4) {
209:           fillRoom(x+1,y+3+RPG.r(2),x+8-left,y+8+RPG.r(3),Tile.STONEWALL,Tile.STONEFLOOR); 
210:           setTile(x+8-left,y+5+RPG.r(3),Tile.STONEFLOOR);
211:         }
212:         
213:         // main door
214:         setTile(x+8,y+12,Tile.STONEFLOOR);
215:         addThing(new Door("ornate door",getLevel()),x+8,y+12);
216:         
217:         addThing(new Person(Person.PRIEST),x,y,x+15,y+15);
218:     
219:         // random rotate
220:         rotateArea(x,y,16,RPG.r(4));
221:         break;
222:       }
223: 
224:       // central crossroads with shops
225:       case 6: {
226:         addShop(x,y,x+RPG.d(3,2),y+RPG.d(3,2),1,0,4);
227:         addShop(x+15-RPG.d(3,2),y,x+15,y+RPG.d(3,2),0,1,0);
228:         addShop(x,y+15-RPG.d(3,2),x+RPG.d(3,2),y+15,0,-1,2);
229:         addShop(x+15-RPG.d(3,2),y+15-RPG.d(3,2),x+15,y+15,-1,0,3);
230:         break;
231:       }
232:       
233:       // pond
234:       case 7: {
235:         setTile(x+8,y+8,Tile.SEA);
236:         fractalize(x+4,y+4,x+11,y+11,4);
237:         for (int i=0; i<10; i++) {
238:           addThing(GameScenery.create(GameScenery.BUSH),x,y,x+15,y+15); 
239:         } 
240:         break;
241:       }
242:       
243:       // river
244:       case 8: case 9: case 10: case 11: {
245:         // N/S river
246:         makeRandomPath(x+8,y,x+8,y+15,x,y,x+15,y+15,Tile.SEA,false);
247:         spreadTiles(x,y,x+15,y+15,Tile.SEA,floortile);
248:         if ((t&1)==1) {
249:           // add the bridge
250:           fillArea(x,y+7,x+15,y+9,floortile);
251:         }
252:         
253:         // rotate for E/W river
254:         if ((t&2)>0) rotateArea(x,y,16,1);
255:         break;
256:       }
257:       
258:       // NS and EW streets
259:       case 12: case 13: {
260:         int p1,p2;
261:         p1=8-RPG.d(4);
262:         p2=8+RPG.d(4);
263:         addFeature(x,y+RPG.r(2),x+p1,y+5+RPG.r(3),0,1,0);
264:         addFeature(x+p1,y+RPG.r(2),x+p2,y+5+RPG.r(3),0,1,0);
265:         addFeature(x+p2,y+RPG.r(2),x+15,y+5+RPG.r(3),0,1,0);
266: 
267:         p1=8-RPG.d(4);
268:         p2=8+RPG.d(4);
269:         addFeature(x,y+9+RPG.r(3),x+p1,y+15-RPG.r(2),0,-1,0);
270:         addFeature(x+p1,y+9+RPG.r(3),x+p2,y+15-RPG.r(2),0,-1,0);
271:         addFeature(x+p2,y+9+RPG.r(3),x+15,y+15-RPG.r(2),0,-1,0);
272:         
273:         // rotate for N/S street
274:         if (t==12) rotateArea(x,y,16,1);
275:         break;
276:       } 
277: 
278:       // orchard
279:       case 14: {
280:         for (int i=0; i<20; i++) {
281:           int px=x+1+RPG.r(14);
282:           int py=y+1+RPG.r(14);
283:           if (!isBlocked(px,py)) addThing(GameScenery.create(7),px,py);
284:         } 
285:         for (int i=0; i<20; i++) {
286:           int px=x+1+RPG.r(14);
287:           int py=y+1+RPG.r(14);
288:           if (!isBlocked(px,py)) addThing(Food.createFood(1+4*RPG.r(2)),px,py);
289:         } 
290:         break;
291:       }
292:       
293:       // smithy
294:       case 15: {
295:         addShop(x+5,y+5,x+10,y+10,0,-1,5);
296:         rotateArea(x,y,16,RPG.r(4));
297:         break; 
298:       }
299:       
300:       // river bends
301:       case 16: case 17: case 18: case 19: {
302:         makeRandomPath(x+8,y,x,y+8,x,y,x+15,y+15,Tile.SEA,false);
303:         spreadTiles(x,y,x+15,y+15,Tile.SEA,floortile);
304:         
305:         // rotate clockwise to correct orientation
306:         // 16 = N/W position etc..
307:         rotateArea(x,y,16,t-16);
308:         break;
309:       }
310:       
311:       // river forks
312:       case 20: case 21: case 22: case 23: {
313:         makeRandomPath(x+8,y,x,y+8,x,y,x+15,y+15,Tile.SEA,false);
314:         makeRandomPath(x+8,y,x+15,y+8,x,y,x+15,y+15,Tile.SEA,false);
315:         spreadTiles(x,y,x+15,y+15,Tile.SEA,floortile);
316:         
317:         // rotate clockwise to correct orientation
318:         // 20 = N/W/E position etc..
319:         rotateArea(x,y,16,t-20);
320:         break;
321:       }
322:       
323:     }
324:   }

327:   public void addFeature(int x1, int y1, int x2, int y2,int dx, int dy, int t) {
328:     // randomize feature
329:     if (t==0) t=RPG.d(10);
330:     
331:     switch (t) {
332:       case 1:
333:         addShop(x1,y1,x2,y2,dx,dy,0); 
334:         break;
335:       
336:       default: 
337:         addRoom(x1,y1,x2,y2,dx,dy); 
338:         break; 
339:     }

360:

367:   // note that areas are likely to be rotated

370:   public void addShop(int x1, int y1, int x2, int y2,int dx, int dy, int t) {
371:     int doorx,doory;
372:     doorx=(dx==0)?RPG.rspread(x1+1,x2-1): ((dx==-1)?x1:x2);
373:     doory=(dy==0)?RPG.rspread(y1+1,y2-1): ((dy==-1)?y1:y2);
374:     fillArea(x1,y1,x2,y2,Tile.FLOOR);
375:     fillBorder(x1,y1,x2,y2,Tile.CAVEWALL);
376:     setTile(doorx,doory,Tile.FLOOR);
377:     
378:     Door door=new Door("shop door",getLevel());
379:     
380:     if (t==0) t=RPG.d(8);
381:     
382:     switch(t) {
383:       // magic type shop
384:       case 1: {
385:         for (int x=x1+1; x<x2; x++) for (int y=y1+1; y<y2; y++) {
386:           Item it=(Item)Lib.createMagicItem(Game.hero.getStat(RPG.ST_LEVEL));
387:           if (RPG.d(2)==1) it.identify(1000);
388:           it.flags|=Item.ITEM_OWNED;  // shop owned
389:           addThing(it,x,y); 
390:         }
391:         Being wiz=new Person(Person.WIZARD);
392: 
393:         Thing b1=new Secret(Secret.MARKER); addThing(b1,x1,y1);
394:         Thing b2=new Secret(Secret.MARKER); addThing(b2,x2-1,y2-1);
395:         wiz.setAI(new GuardAI(b1,b2));
396:         addThing(wiz,(x1+x2)/2,(y1+y2)/2);
397:         addThing(GameScenery.create("magic shop sign"),doorx+dx+dy*(RPG.r(2)*2-1),doory+dy+dx*(RPG.r(2)*2-1)); 
398:         break;
399:       }
400:       
401:       // weapon store
402:       case 2: {
403:         for (int x=x1+1; x<x2; x++) for (int y=y1+1; y<y2; y++) {
404:           Item it=null;
405:           switch (RPG.d(6)) {
406:             case 1: it=StandardWeapon.createWeapon(4); break; 
407:             case 2: it=StandardArmour.createArmour(0); break;
408:             case 3: it=StandardArmour.createShield(0); break; 
409:             case 4: it=Missile.createMissile(0); break; 
410:             case 5: it=RangedWeapon.createRangedWeapon(0); break; 
411:             default: continue;
412:           };  
413:           it.flags|=Item.ITEM_OWNED;  // shop owned
414:           addThing(it,x,y); 
415:         }
416:         Being guard=new Person(Person.GUARD);
417: 
418:         Thing b1=new Secret(Secret.MARKER); addThing(b1,x1+1,y1+1);
419:         Thing b2=new Secret(Secret.MARKER); addThing(b2,x2-1,y2-1);
420:         guard.setAI(new GuardAI(b1,b2));
421:         addThing(guard,(x1+x2)/2,(y1+y2)/2); 
422:         
423:         addThing(GameScenery.create("armoury sign"),doorx+dx+dy*(RPG.r(2)*2-1),doory+dy+dx*(RPG.r(2)*2-1)); 
424:         break; 
425:       }
426:       
427:       // food store
428:       case 3: {
429:         for (int x=x1+1; x<x2; x++) for (int y=y1+1; y<y2; y++) {
430:           Item it= Food.createFood(0);
431:           it.flags|=Item.ITEM_OWNED;  // shop owned
432:           addThing(it,x,y); 
433:         }
434:         Being shopkeeper=new Person(Person.SHOPPIE);
435: 
436:         Thing b1=new Secret(Secret.MARKER); addThing(b1,x1+1,y1+1);
437:         Thing b2=new Secret(Secret.MARKER); addThing(b2,x2-1,y2-1);
438:         shopkeeper.setAI(new GuardAI(b1,b2));
439:         addThing(shopkeeper,(x1+x2)/2,(y1+y2)/2);

443:       }
444:

448:         Being smith=new Person(Person.BLACKSMITH);
449:         Thing b1=new Secret(Secret.MARKER); addThing(b1,x1+1,y1+1);
450:         Thing b2=new Secret(Secret.MARKER); addThing(b2,x2-1,y2-1);

486:

491:           it.flags|=Item.ITEM_OWNED;  // shop owned
492:           addThing(it,x,y); 
493:         }
494:         Being shopkeeper=new Person(Person.SHOPPIE);
495: 
496:         Thing b1=new Secret(Secret.MARKER); addThing(b1,x1+1,y1+1);

509:   public void addBuilding() {
510:     int w=RPG.d(2,4)+3;
511:     int h=RPG.d(2,4)+3;
512:     int x1=RPG.r(width-w); 
513:     int y1=RPG.r(height-h);
514:     int x2=x1+w;
515:     int y2=y1+h;
516:     int dx=RPG.r(3)-1;
