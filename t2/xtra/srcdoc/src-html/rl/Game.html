060:   public static String getLine(String prompt, String result) {
061:     messagepanel.addMessage(prompt+result);
062:     while(true) {
063:         KeyEvent k=getInput();
064:         char ch=k.getKeyChar();
065:         if (k.getKeyCode()==k.VK_ENTER) break;
066:         if (k.getKeyCode()!=k.VK_BACK_SPACE) {
067:             // add the character to the input string if typed
068:             // i.e. don't include SHIFT, ALT etc.
069:           if (Character.isWhitespace(ch)||Character.isLetterOrDigit(ch)) {
070:               result = result+ch;
071:           }
072:         } else {
073:           try {
074:             result=result.substring(0,result.length()-1);
075:           } catch (Exception e) {}
076:         }
077:         messagepanel.setText(prompt+result);
078:     } 
079:     Game.message("");
080:     return result;
081:   }

144:   // Waits for user to select one of specified keys
145:   // returns: key pressed (from list)
146:   //      or: 'q' if ESC is pressed
147:   //      or: 'e' if ENTER is pressed
148:   public static char getOption(String s) {
149:     while (true) {
150:       KeyEvent k=getInput();
151:       char c=Character.toLowerCase(k.getKeyChar());
152:       if (s.indexOf(c)>=0) return c;
153:       if (k.getKeyCode()==k.VK_ESCAPE) return 'q';
154:       if (k.getKeyCode()==k.VK_ENTER) return 'e';
155:     }  
156:   }

158:   // Choose a direction, given as a Pont offset
159:   public static Point getDirection() {
160:     while (true) {
161:       KeyEvent e=Game.getInput();
162:   
163:       char k=Character.toLowerCase(e.getKeyChar());
164:       int i=e.getKeyCode();
165:       
166:       // handle key conversions
167:       if (i==e.VK_UP) k='8';   
168:       if (i==e.VK_DOWN) k='2'; 
169:       if (i==e.VK_LEFT) k='4';   
170:       if (i==e.VK_RIGHT) k='6';         
171:       if (i==e.VK_HOME) k='7';   
172:       if (i==e.VK_END) k='1';   
173:       if (i==e.VK_PAGE_UP) k='9';   
174:       if (i==e.VK_PAGE_DOWN) k='3';   
175:       if (i==e.VK_ESCAPE) k ='q'; 
176: 
177:       int dx=-2; int dy=0;
178:       if (k=='8') { dx=0; dy=-1; }
179:       if (k=='2') { dx=0; dy=1; }
180:       if (k=='4') { dx=-1; dy=0; }
181:       if (k=='6') { dx=1; dy=0; }
182:       if (k=='7') { dx=-1; dy=-1; }
183:       if (k=='9') { dx=1; dy=-1; }
184:       if (k=='1') { dx=-1; dy=1; }
185:       if (k=='3') { dx=1; dy=1; }
186:       
187:       if ((k=='5')||(k=='.')) { dx=0; dy=0;}
188:       if (k=='q') {return null;}
189:            
190:       if (dx!=-2) return new Point(dx,dy);
191:     }
192:   }

221:   // save game to local ZIP file
222:   public static boolean save() {
223:     if (QuestApp.isapplet) {
224:       Game.message("You cannot save in the applet version of Tyrant");
225:       Game.message("This is due to browser security restrictions");
226:       Game.message("Run the downloaded application version instead");
227:       return false; 
228:     }
229:     
230:     try {
231:       String filename=getLine("Enter a filename: ","tyrant.sav");
232:       FileOutputStream f=new FileOutputStream(filename);
233:       ZipOutputStream z=new ZipOutputStream(f);
234:       
235:       z.putNextEntry(new ZipEntry("data"));
236:       
237:       if (!save(new ObjectOutputStream(z))) {
238:         throw new Error("Save game failed"); 
239:       } else {
240:         Game.message("Game saved - "+filename);
241:       }
242:       z.closeEntry();
243:       z.close();
244:     } catch (Exception e) {
245:       System.out.println(e);
246:       return false;
247:     }
248:     return true;
249:   }

251:   // restore game from local zip file
252:   public static boolean restore() {
253:     if (QuestApp.isapplet) {
254:       Game.message("You cannot load in the applet version of Tyrant");
255:       Game.message("This is due to browser security restrictions");
256:       Game.message("Run the downloaded application version instead");
257:       return false; 
258:     }
259:     
260:     try {
261:       String filename=getLine("Enter save game filename: ","tyrant.sav");
262:       FileInputStream f=new FileInputStream(filename);
263:       ZipInputStream z=new ZipInputStream(f);
264:       z.getNextEntry();
265:       if (!restore(new ObjectInputStream(z))) {
266:         Game.message("Load game failed - "+filename); 
267:       }
268:       z.closeEntry();
269:       z.close();
270:     } catch (Exception e) {
271:       System.out.println(e);
272:       return false;
273:     }
274:     return true;
275:     
276:   }
