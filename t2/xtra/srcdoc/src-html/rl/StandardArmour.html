252:   // create armour from given source matrix
253:   public static StandardArmour createArmour(int[][] source, int level) {
254:     while (RPG.d(2)==1) {level=level+RPG.d(2)-RPG.d(2);}
255:     int improve=RPG.r(2)*(RPG.d(3,3)-6);
256:     level-=RPG.sign(improve);
257: 
258:     level=RPG.middle(0,level,source.length-1);
259:     int type=source[level][RPG.r(source[level].length)];
260: 
261:     StandardArmour w=new StandardArmour(type);
262:     
263:     // curse a few!
264:     if (RPG.d(100)<=level) {
265:       w.flags|=ITEM_CURSED;
266:       improve-=2; 
267:     }
268: 
269:     w.quality=RPG.middle(1,w.quality+improve,12);
270:     return w;
271:   }
