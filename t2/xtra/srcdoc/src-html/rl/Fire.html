01: package rl;
02: 
03: public class Fire extends Special implements Active {
04:   protected int strength;
05:   protected int decay;
06:     
07:     public static final Description DESC_FIRE =
08:         new Describer("fire","A blazing fire");
09:   public static final Description DESC_FIRE1 =
10:     new Describer("small fire","A small fire");
11:   public static final Description DESC_FIRE2 =
12:     new Describer("fire","A hot fire");
13:   public static final Description DESC_FIRE3 =
14:     new Describer("blazing fire","A blazing fire");
15:   public static final Description DESC_FIRE4 =
16:     new Describer("roaring fire","A large roaring fire");
17:   public static final Description DESC_FIRE5 =
18:     new Describer("inferno","An inferno of hellish flames");
19:     
20:   public Fire(int firestrength) {
21:       strength=firestrength;
22:     decay=2;
23:     }
24:   
25:   public Fire(int firestrength, int dec) {
26:     strength=firestrength;
27:     decay=dec;
28:   }
29: 
30:   // create fire at specified map location
31:   // add to existing fire if present
32:   public static void createFire(Map m, int tx, int ty, int s) {
33:     if (m==null) return;
34:     Fire f=(Fire)m.getObject(tx,ty,Fire.class);
35:     if (f==null) {
36:       m.addThing(new Fire(s),tx,ty);
37:     } else {
38:       f.strength+=s; 
39:     }
40:   }
41:   
42:   // feared unless creature resists fire
43:   public boolean isWarning(Thing t) {
44:     int arm=t.getStat(RPG.ST_ARMFIRE);
45:     int d=(strength*strength)/(arm+strength);
46:     return d >= RPG.power(t.getStat(RPG.ST_RESISTFIRE)); 
47:   }
48:   
49:   public void fireDamage(Thing t) {
50:       int dam=t.damageAll(RPG.r(strength+1),RPG.DT_FIRE);
51:       if (t==Game.hero) {
52:         if (dam>0) {
53:             Game.message("The fire burns you!");
54:         } else {
55:             Game.message("You are unharmed by the flames");
56:         }
57:       }    
58:     }
59:     
60:     public int damage(int dam, int damtype) {
61:       if (damtype==RPG.DT_ICE) {
62:         strength=strength-dam;
63:         return dam;
64:       }    
65:       return 0;
66:     }
67:     
68:     public Description getDescription() {
69:     if (strength<=2) return DESC_FIRE1;
70:     if (strength<=5) return DESC_FIRE2;
71:     if (strength<=15) return DESC_FIRE3;
72:     if (strength<=40) return DESC_FIRE4;
73:     return DESC_FIRE5;
74:   }
75:     
76:   public int getZ() {return Z_OVERHEAD;}
77:     
78:   public void action(int time) {
79:         if (place instanceof Map) {
80:       Map m = (Map) place;
81:       Thing[] t=m.getThings(x,y);
82:       for (int i=0; i<t.length; i++) {
83:         fireDamage(t[i]);    
84:           }
85:     } 
86:     int dec=RPG.po(time*decay/1000.0);
87:     for (int i=0; i<dec; i++) strength=(3*strength)/4;
88:     if (strength<=0) remove();    
89:     }
90:     
91:     public int getImage() {return 180+RPG.r(4);}
92: }
