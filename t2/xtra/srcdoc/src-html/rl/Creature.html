0312:     if (level<0) level=0;
0313:     for (int tries=0; tries<100; tries++) {
0314:       int thelevel=level;
0315:       
0316:       // Adjust level randomly. Not too much.
0317:       while (RPG.d(3)==1) thelevel+=RPG.r(3)-1; 
0318:       
0319:       // get random creature type from source
0320:       int i=source[RPG.r(source.length)];
0321:       // if zero, pick from whole creature array
0322:       if (i==0) i=RPG.d(types.length-1);  
0323:         
0324:       // brains check
0325:       if ((brains>=0)&&(brains!=intelligent[i])) continue;
0326:       
0327:       // rarity check
0328:       if (RPG.d(rarity[i])!=1) continue;
0329:       
0330:       if (levels[i]==thelevel) {
0331:         return new Creature(types[i],thelevel);
0332:       } else if (levels[i]<thelevel) {
0333:         switch (RPG.d(8)) {
0334:           case 1: case 2: case 3:
0335:             // create levelled-up creature now and again
0336:             return new Creature(types[i],thelevel);
0337:           case 4:
0338:             // occasionally just an easy critter
0339:             return new Creature(types[i],levels[i]);
0340:         } 
0341:       }
0342:     }
0343:     return Creature.createCreature(Creature.spawn,level);
0344:   }
