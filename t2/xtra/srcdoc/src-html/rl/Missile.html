139:   public void throwAt(Being thrower, Map m, int tx, int ty) {
140:     Thing t = m.getMobile(tx,ty);
141:     
142:     thrower.aps-=(throwcosts[type]*10)/(10+thrower.getStat(RPG.ST_THROWING));
143:     
144:     if (type==FIREFLASK) {
145:       remove(1);
146:       new Spell(Spell.BLAZE).castAtLocation(thrower,m,tx,ty);
147:       return;
148:     }
149:     
150:     // weapon skill calculation
151:     int wsk=thrower.getStat(RPG.ST_SK);
152:     wsk=(wsk*(4+thrower.getStat(RPG.ST_THROWING)))/4;
153:     
154:     // range
155:     int range=ranges[type];
156:     
157:     // improvisation modifiers
158:     if (missiletypes[type]!=RPG.MISSILE_THROWN) {
159:       // we have improvised, so reduce hit chance
160:       wsk/=2; 
161:       range/=2;
162:     }
163: 
164:     if (t!=null) {
165:       // calculate distance and offset
166:       int dx=tx-thrower.x;
167:       int dy=ty-thrower.y;
168:       int dist=(int)Math.sqrt(dx*dx+dy*dy);
169: 
170:       Game.mappanel.doShot(thrower.x,thrower.y,tx,ty,100,0.5);
171: 
172:       int rsk=(rskmuls[type]*wsk)/100 + rskbonuses[type];
173:       if (RPG.test(rsk,shotDifficulty(thrower,t,dist))) {
174:         //have scored a hit!
175:         hit(thrower,t,RPG.MISSILE_THROWN);
176:         
177:         // drop item if missile survives
178:         if (RPG.d(100)<=survivals[type]) moveTo(m,tx,ty);
179:         
180:         return;
181:       } else {
182:         Game.message(getTheName()+" misses"); 
183:       }
184:       moveTo(m,tx,ty);
185:       return;
186:     }
187: 
188:     // make boomerang come back!
189:     if ((thrower==Game.hero)&&(type==BOOMERANG)&&RPG.test(wsk,10)) {
190:       Game.message(getTheName()+" returns!");
191:       if (RPG.test(wsk,10)) {
192:         Game.message("You catch "+getTheName());
193:         thrower.addThing(this);
194:         return; 
195:       } else {
196:         Game.message(getTheName()+" lands near your feet");
197:         tx=thrower.x; 
198:         ty=thrower.y;
199:         moveTo(m,tx,ty);
200:         displace();
201:         return;
202:       }
203:     }
204:     
205:     // item lands
206:     moveTo(m,tx,ty);
207:   }

209:   // thrown item hits
210:   public void hit(Being thrower, Thing target, int firetype) {
211:     // weapon skill calculation
212:     int rst=(rstmuls[type]*(thrower.getStat(RPG.ST_ST)))/100 + rstbonuses[type];
213:     
214:     if (firetype==RPG.MISSILE_THROWN) {
215:       rst=(rst*(6+thrower.getStat(RPG.ST_THROWING)))/6;
216:     }
217:     
218:     if (missiletypes[type]!=firetype) {
219:       // we have improvised, so attack does minimal damage
220:       rst/=5; 
221:     }
222:     
223:     if (target.isVisible()) Game.message(getTheName()+" hits "+target.getTheName());
224:     hit(target,rst);
225:   }

227:   // calculate hit effect
228:   public void hit(Thing target, int rst) {
229: 
230:     switch (type) {
231:       case FLAMEARROW:
232:         target.damage(rst,RPG.DT_FIRE); break;
233:       
234:       case POISONARROW: {
235:         int d=target.damage(rst,RPG.DT_NORMAL); 
236:         if (d>0) {
237:           target.addThing(new PoisonAttribute(2,RPG.DT_POISON,500*RPG.d(2,9),3));
238:         }
239:         break;
240:       }
241:       
242:       case TANGLER:
243:         target.damage(rst,RPG.DT_NORMAL);
244:         if (RPG.po(20,target.getStat(RPG.ST_ST))>1) {
245:           target.addAttribute(new TemporaryAttribute("Tangled",5000/target.getStat(RPG.ST_ST),new int[] {RPG.ST_IMMOBILIZED,1},"You untangle yourself"));
246:         }
247:         break;
248:       
249:       default:
250:         target.damage(rst,RPG.DT_NORMAL); break;
251:     }
252:   }
