001: //  Defines Themes for different dungeons
002: //
003: //  Themes can be responsible for any or all of the following:
004: //   - Tile Types
005: //   - Random decorations
006: //   - Random objects
007: //   - Room decorations
008: //   - Creature generation
009: //
010: //
011: //  usage:
012: //       map.setTheme(new Theme("woods"));
013: //
014: 
015: package rl;
016: 
017: import java.io.*;
018: 
019: public class Theme implements Serializable {
020:   public static final int HARDFLOOR=1;
021:   public static final int HARDWALL=2;
022:   public static final int SOFTFLOOR=3;
023:   public static final int SOFTWALL=4;
024:   public static final int STREAM=5;
025:   public static final int RIVER=6;
026: 
027:   public static final String[] names   = {"woods",          "standard",     "village",       "town",           "darktown",       "city",           "goblins",       "hills",         "plains",       "forest",       "deepforest",     "caves",         "dungeon",    "labyrinthe",      "sewers",        "swamp"};
028:   public static final int[] hardwalls  = {Tile.WALL       , Tile.WALL,      Tile.WALL,       Tile.WALL,        Tile.WALL,        Tile.STONEWALL,   Tile.CAVEWALL,    Tile.CAVEWALL,  Tile.CAVEWALL,  Tile.CAVEWALL,  Tile.TREE,        Tile.CAVEWALL,   Tile.WALL,    Tile.STONEWALL,    Tile.CAVEWALL,   Tile.CAVEWALL};
029:   public static final int[] hardfloors = {Tile.FLOOR      , Tile.FLOOR,     Tile.FLOOR,      Tile.FLOOR,       Tile.FORESTFLOOR, Tile.STONEFLOOR,  Tile.CAVEFLOOR,   Tile.CAVEFLOOR, Tile.PLAINS,    Tile.CAVEFLOOR, Tile.FORESTFLOOR, Tile.CAVEFLOOR,  Tile.FLOOR,   Tile.STONEFLOOR,   Tile.GUNK,       Tile.GUNK};
030:   public static final int[] softwalls  = {Tile.TREE       , Tile.WALL,      Tile.CAVEWALL,   Tile.CAVEWALL,    Tile.CAVEWALL,    Tile.CAVEWALL,    Tile.CAVEWALL,    Tile.CAVEWALL,  Tile.CAVEWALL,  Tile.CAVEWALL,  Tile.CAVEWALL,    Tile.CAVEWALL,   Tile.WALL,    Tile.STONEWALL,    Tile.CAVEWALL,   Tile.CAVEWALL};
031:   public static final int[] softfloors = {Tile.FORESTFLOOR, Tile.FLOOR,     Tile.GRASS,      Tile.FORESTFLOOR, Tile.FORESTFLOOR, Tile.STONEFLOOR,  Tile.CAVEFLOOR,   Tile.CAVEFLOOR, Tile.PLAINS,    Tile.CAVEFLOOR, Tile.CAVEFLOOR,   Tile.CAVEFLOOR,  Tile.FLOOR,   Tile.STONEFLOOR,   Tile.CAVEFLOOR,  Tile.CAVEFLOOR};
032:   public static final int[] streams    = {Tile.STREAM     , Tile.STREAM,    Tile.STREAM,     Tile.STREAM,      Tile.STREAM,      Tile.STREAM,      Tile.STREAM,      Tile.STREAM,    Tile.STREAM,    Tile.STREAM,    Tile.STREAM,      Tile.STREAM,     Tile.STREAM,  Tile.STREAM,       Tile.GUNK,       Tile.GUNK};
033:   public static final int[] rivers     = {Tile.RIVER      , Tile.RIVER,     Tile.RIVER,      Tile.RIVER,       Tile.RIVER,       Tile.RIVER,       Tile.RIVER,       Tile.RIVER,     Tile.RIVER,     Tile.RIVER,     Tile.RIVER,       Tile.RIVER,      Tile.RIVER,   Tile.RIVER,        Tile.GUNK,       Tile.GUNK};
034:   
035:   // keep a Theme pointer for the Map's theme stack
036:   // this is so maps can use a theme temporarily like this:
037:   //
038:   // ** using the original theme **
039:   // map.newTheme(newtheme);
040:   // ** using the new theme **
041:   // map.oldTheme();
042:   // ** using the original theme again **
043:   //
044:   public Theme next;
045:   
046:   public int[] inhabitants;
047:   
048:   private int type;
049:     
050:   public int floor;
051:   public int wall;
052:   public int hardfloor;
053:   public int hardwall;
054:   public int softfloor;
055:   public int softwall;
056:   public int stream;
057:   public int river;
058:   
059:   // get the tile code for the particular theme wall type
060:   public int getTile(int t) {
061:     switch (t) {
062:       case HARDFLOOR: return hardfloors[type];
063:       case HARDWALL: return hardwalls[type];
064:       case SOFTFLOOR: return softfloors[type];
065:       case SOFTWALL: return softwalls[type];
066:       case STREAM: return streams[type];
067:       case RIVER: return rivers[type];
068:     } 
069:     return Tile.WALL;
070:   }
071:   
072:   public Theme(String s) {
073:     type=Text.index(s,names); 
074:     if (type<0) throw new Error("Theme not found: "+s);
075:     
076:     // set up default wall types
077:     hardwall=getTile(HARDWALL);
078:     hardfloor=getTile(HARDFLOOR);
079:     softwall=getTile(SOFTWALL);
080:     softfloor=getTile(SOFTFLOOR);
081:     stream=getTile(STREAM);
082:     river=getTile(RIVER);
083:     
084:     floor=hardfloor;
085:     wall=hardwall;
086:   }
087:   
088:   // Theme generation functions
089:   
090:   public Being createCreature(int level) {
091:     if (inhabitants!=null) {
092:       return new Creature(inhabitants[RPG.r(inhabitants.length)]); 
093:     }
094:     
095:     return Creature.createCreature(level); 
096:   }
097:   
098:   public void createDoor(Map map, int tx, int ty, int level) {
099:     map.setTile(tx,ty,map.floortile);
100:     map.addThing(Door.createDoor(level),tx,ty); 
101:   }
102: }
