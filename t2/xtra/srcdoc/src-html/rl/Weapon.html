001: package rl;
002: 
003: public class Weapon extends Item {
004:   // use to hit target
005:   public int use(Thing wielder, Thing target, int action) {
006:     switch (action) {
007:       // Weapon used to attack
008:       // return >0 if hit is successful
009:       case USE_ATTACK: {
010:         int attack=getASK(wielder,target);
011:         int defence=target.getStat(RPG.ST_DSK);
012: 
013:         int result=RPG.test(attack,defence)?1:0;
014:         
015:         if ((result==0)&&(target.isVisible())) {
016:           boolean ishero=(wielder==Game.hero);
017:           Game.message(Text.capitalise(wielder.getTheName()+
018:                      (ishero?" miss ":" misses ")+
019:                      target.getTheName()+
020:                      ".  ")+
021:                        Text.capitalise((target==Game.hero?(target.getTheName()+" are"):"it is")+
022:                        " now at "+
023:                        RPG.percentile(target.getStat(RPG.ST_HPS), target.getStat(RPG.ST_HPSMAX))+
024:                        "% health."));
025:         }  
026: 
027:         return result;
028:       }
029:       
030:       // Weapon has hit the given target
031:       case USE_HIT: {
032:         Map map=target.getMap();
033: 
034:         int ast=getAST(wielder,target);
035: 
036:         ast=RPG.r(ast+1);
037:         int dam=target.damageLocation(ast,RPG.DT_NORMAL);
038:         
039:         // report the attack
040:         if (target.isVisible()) {
041:           boolean heroattack = (wielder==Game.hero);
042:           //Game.message(Text.capitalise(wielder.getTheName()+ (heroattack?" hit ":" hits ")+target.getTheName()+" for "+Lib.damageDescription(dam,target.getStat(RPG.ST_HPSMAX))));
043:           Game.message(Text.capitalise(wielder.getTheName()+
044:                        (heroattack?" hit ":" hits ")+
045:                        target.getTheName()+
046:                        " for "+
047:                        Lib.damageDescription(dam,target.getStat(RPG.ST_HPSMAX))+
048:                      ".  ")+
049:                        Text.capitalise((target==Game.hero?(target.getTheName()+" are"):"it is")+
050:                        " now at "+
051:                        RPG.percentile(target.getStat(RPG.ST_HPS), target.getStat(RPG.ST_HPSMAX))+
052:                        "% health."));
053:         }
054: 
055:         // slight wear and tear to weapon 
056:         // lower hit damage = more armour = higher damage to weapon
057:         //   note: (ast-dam) is the number of hps deflected by armour
058:         damage((ast-dam)/3,RPG.DT_NORMAL);
059:         
060:         return dam;
061:       }
062:     }
063:     return 0;
064:   }
065:   
066:   public int getUse() {return USE_HIT;}
067:   
068:   // Get weapon attack skill
069:   public int getASK(Thing wielder, Thing target) {
070:     int ASK=wielder.getStat(RPG.ST_SK);
071:     ASK=(ASK*getStat(RPG.ST_ASKMULTIPLIER))/100;
072:     ASK=ASK+getStat(RPG.ST_ASKBONUS);
073:     return ASK;
074:   }
075:   
076:   // Get weapon attack strength
077:   public int getAST(Thing wielder, Thing target) {
078:     int AST=wielder.getStat(RPG.ST_ST);
079:     AST=(AST*getStat(RPG.ST_ASTMULTIPLIER))/100;
080:     AST=AST+getStat(RPG.ST_ASTBONUS);
081:     return AST;
082:   }
083:   
084:   // get weapon defence skill
085:   public int getDSK(Thing wielder, Thing attacker) {
086:     int DSK=wielder.getStat(RPG.ST_SK);
087:     DSK=(DSK*getStat(RPG.ST_DSKMULTIPLIER))/100;
088:     DSK=DSK+getStat(RPG.ST_DSKBONUS);
089:     return DSK;
090:   }
091:   
092:   public static Weapon createWeapon(int level) {
093:     return CommonWeapon.createWeapon(level);
094:     //switch (RPG.d(20)) {
095:     //     case 1: return StandardWeapon.createWeapon(StandardWeapon.specialsource,n);
096:     //    default: return StandardWeapon.createWeapon(StandardWeapon.source,n);
097:     //}
098:   }
099:           
100:   public int wieldType() {return RPG.WT_MAINHAND;}
101: }
