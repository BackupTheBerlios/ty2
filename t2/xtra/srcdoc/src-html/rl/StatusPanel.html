008: public class StatusPanel extends TPanel {
009:   public static final int boxborder=1;
010:   public static final Color healthcolor=new Color(180,80,0);
011:   public static final Color powercolor=new Color(0,128,60);
012: 
013:   public StatusPanel() {
014:     super();
015:     setBackground(QuestApp.panelcolor); 
016:   }
017:   
018:   public Dimension getPreferredSize() {
019:     return new Dimension(208,272); 
020:   }
021:   
022:   public void paint(Graphics g) {
023:     super.paint(g);
024:     
025:     FontMetrics met=g.getFontMetrics(g.getFont());
026:     int charwidth=met.charWidth(' ');
027:     int charheight=met.getMaxAscent()+met.getMaxDescent();
028: 
029:     Hero h=Game.hero;
030:     Map m=h.getMap();
031: 
032:     Rectangle bounds=getBounds();
033:     int width=bounds.width;
034:     int height=bounds.height;
035:     
036:     // old code to do panel borders
037:     //g.setColor(QuestApp.panelhighlight);
038:     //g.fillRect(0,0,width,2);
039:     //g.fillRect(0,0,2,height);
040:     //g.setColor(QuestApp.panelshadow);
041:     //g.fillRect(1,height-2,width-1,2);
042:     //g.fillRect(width-2,1,2,height-1);
043:     
044:     
045:     // draw energy bars
046:     int hp=h.getStat(RPG.ST_HPS);
047:     int hpm=h.getStat(RPG.ST_HPSMAX);
048:     float hel=((float)hp)/hpm;
049:     
050:     if (hel<0) hel=0;
051:     if (hel>1) hel=1;
052:     paintLabel(g,"Health: "+hp+"/"+hpm,10,18);
053:     paintBar(g,10+16*charwidth,10,width-20-16*charwidth,16,healthcolor,Color.black,hel);
054: 
055:     int mp=h.getStat(RPG.ST_MPS);
056:     int mpm=h.getStat(RPG.ST_MPSMAX);
057:     float pow=((float)mp)/mpm;
058: 
059:     paintLabel(g,"Power:  "+mp+"/"+mpm,10,44);
060:     if (pow<0) pow=0;
061:     if (pow>1) pow=1;
062:     paintBar(g,10+16*charwidth,36,width-20-16*charwidth,16,powercolor,Color.black,pow);
063:   
064:     paintLabel(g,(m==null)?"Six Feet Under":m.getLevelName(),10,70);
065:   
066:     // list current attributes
067:     { 
068:       Thing[] atts = h.inv.getContents(Attribute.class);
069:       StringList sl = new StringList();
070:       for (int i=0; i<atts.length; i++) {
071:         sl.add(((Attribute)atts[i]).getAttributeString()); 
072:       }
073:       
074:       // other effects to note
075:       int hunger=h.getHunger();
076:       if (hunger>0) sl.add("Hungry");
077:       
078:       Thing w=h.getWielded(RPG.WT_MAINHAND);
079:       if (w==null) sl.add("Unarmed");
080:       
081:       int followers=h.inv.getContents(Being.class).length;
082:       if (followers>0) sl.add(Integer.toString(followers)+((followers>1)?" companions":" companion"));
083:       
084:       sl=sl.compress();
085:       for (int i=0; i<sl.getCount(); i++) {
086:         paintLabel(g,sl.getString(i),20,70+(i+1)*charheight);
087:       }
088:     }
089:   }
090:   
091:   public static void paintBar(Graphics g, int x, int y, int w, int h, Color f, Color b, float amount) {
092:     if (amount>1) amount=1;
093:     g.setColor(f);
094:     g.fillRect(x,y,(int)(w*amount),h);
095:     g.setColor(b);
096:     g.fillRect(x+(int)(w*amount),y,(int)(w*(1-amount)),h);
097:     paintBox(g,x,y,w,h,false);
098:   } 
099:   
100:   public static int paintLabel(Graphics g, String s, int x, int y) {
101:     g.setColor(QuestApp.textcolor);
102:     
103:     FontMetrics met=g.getFontMetrics(g.getFont());
104:     int charwidth=met.charWidth(' ');
105:     int charheight=met.getMaxAscent()+met.getMaxDescent();
106: 
107:     g.drawString(s,x,y+met.getMaxAscent()-charheight/2);
108:     
109:     return charwidth*s.length();
110:   } 
111: 
112:   // paint a boxed area, raised or lowered
113:   public static void paintBox(Graphics g, int x,int y,int w,int h, boolean raised) {
114:     if (raised) g.setColor(QuestApp.panelhighlight);
115:       else g.setColor(QuestApp.panelshadow);
116:       
117:     g.fillRect(x,y,w,boxborder);
118:     g.fillRect(x,y,boxborder,h);
119: 
120:     if (!raised) g.setColor(QuestApp.panelhighlight);
121:       else g.setColor(QuestApp.panelshadow);
122: 
123:     g.fillRect(x+1,y+h-boxborder,w-1,boxborder);
124:     g.fillRect(x+w-boxborder,y+1,boxborder,h-1);
125:   }
126: }
