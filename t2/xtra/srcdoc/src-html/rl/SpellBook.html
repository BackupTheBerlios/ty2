001: //
002: // When read, the SpellBook confers the ability to cast a particular spell
003: // on the reader if they are sufficiently skilled
004: //
005: 
006: package rl;
007: 
008: public class SpellBook extends Item implements Description {
009:     private static final Description DESC_BOOK=
010:         new Describer("book","A slightly tattered book");
011:     private static final Description DESC_WORNBOOK=
012:         new Describer("book","A worn book");
013:     private static final Description DESC_DUSTYTOME=
014:         new Describer("tome","A dusty tome");
015:     private static final Description DESC_HEAVYTOME=
016:         new Describer("tome","A heavy tome");
017:     private static final Description DESC_WORNSPELLBOOK=
018:         new Describer("spell book","A worn spell book");
019:      
020:     protected Spell spell;
021:     protected Description desc;
022:     
023:     public int getUse() {return USE_NORMAL|USE_READ;}
024:     
025:   public SpellBook() {
026:     this(1000); 
027:   }
028:   
029:   public SpellBook(int level) {
030:     switch (RPG.d(6)) {
031:       case 1:
032:         desc=DESC_BOOK;    image=287;
033:         switch (RPG.d(6)) {
034:           case 1: spell=new Spell(Spell.HEAL); break;
035:           case 2: spell=new Spell(Spell.LIGHTHEAL); break;
036:           case 3: spell=new Spell(Spell.STRONGHEAL); break;
037:           case 4: spell=new Spell(Spell.ACIDBLAST); break;
038:           default: spell=new Spell(Spell.randomSpell(level)); break;
039:           }
040:           break;
041:         case 2:
042:         desc=DESC_WORNBOOK;    image=284;
043:           switch (RPG.d(5)) {
044:               case 1: spell=new Spell(Spell.HEAL); break;
045:               case 2: spell=new Spell(Spell.FIREBALL); break;
046:               case 3: spell=new Spell(Spell.POISON); break;
047:          default: spell=new Spell(Spell.randomSpell(level)); break;
048:           }
049:         break;            
050:         case 3:
051:         desc=DESC_DUSTYTOME; image=286;    
052:         switch (RPG.d(5)) {
053:           case 1: spell=new Spell(Spell.TELEPORT); break;
054:           case 2: spell=new Spell(Spell.RECHARGE); break;
055:           case 3: spell=new Spell(Spell.REPAIR); break;
056:           default: spell=new Spell(Spell.randomSpell(level)); break;
057:           }
058:         break;            
059:         case 4:
060:             desc=DESC_WORNSPELLBOOK; image=285;    
061:           switch (RPG.d(8)) {
062:               case 1: spell=new Spell(Spell.FORCEBOLT); break;
063:               case 2: spell=new Spell(Spell.POISONBOLT); break;
064:               case 3: spell=new Spell(Spell.FLAME); break;
065:               case 4: spell=new Spell(Spell.THUNDERBOLT); break;
066:               case 5: spell=new Spell(Spell.MISSILE); break;
067:          default: spell=new Spell(Spell.randomSpell(level)); break;
068:           }
069:         break;            
070:       case 5:
071:         desc=DESC_WORNSPELLBOOK; image=285;  
072:         switch (RPG.d(4)) {
073:           case 1: spell=new Spell(Spell.BLAST); break;
074:           case 2: spell=new Spell(Spell.BLAZE); break;
075:           case 3: spell=new Spell(Spell.FLAMECLOUD); break;
076:           default: spell=new Spell(Spell.randomSpell(level)); break;
077:         }
078:         break;        
079:       case 6:
080:         desc=DESC_HEAVYTOME; image=285+RPG.r(2);  
081:         switch (RPG.d(4)) {
082:           case 1: spell=new Spell(Spell.HASTE); break;
083:           case 2: spell=new Spell(Spell.SLOW); break;
084:           case 3: spell=new Spell(Spell.ARMOUR); break;
085:           default: spell=new Spell(Spell.randomSpell(level)); break;
086:         }
087:         break;        
088:       }
089:     }
090:     
091:     public int damage(int dam, int damtype) {
092:         switch(damtype) {
093:           case RPG.DT_FIRE: case RPG.DT_DISPELL: case RPG.DT_ACID:
094:               dam=dam*5;
095:           case RPG.DT_NORMAL: case RPG.DT_PIERCING: case RPG.DT_SPECIAL:
096:               break;
097:       case RPG.DT_ICE: case RPG.DT_IMPACT: case RPG.DT_UNARMED:
098:           dam=dam/2;
099:           break;
100:           default:    
101:               dam=0;
102:         }
103: 
104:       if (!RPG.test(dam,13)) return 0;
105:     Map map=getMap();
106:     int tx=getMapX();
107:     int ty=getMapY();
108:     remove();
109:     if (map!=null) {
110:            spell.castAtLocation(null,map,tx,ty);    
111:       }
112:       return dam;
113:     }
114:     
115:   public int getWeight() {return 1000;}
116:     
117:     public Spell getSpell() {return spell;}
118:     
119:     public void use(Being user) {
120:       use(user, USE_READ);
121:     }
122:     
123:     public void use(Being user, int usetype) {
124:       if (usetype!=USE_READ) return;
125:       
126:       Hero h=Game.hero;
127:       // see if book can be read
128:       if (h.getStat(RPG.ST_LITERACY)>0) {
129:           if (spell.canLearn(h)) {
130:             Spell s;
131:             s=(Spell)(getSpell().clone());
132:           
133:         Art added=h.arts.addArt(s);
134:         if (added==s) {
135:           Game.message("Your mind is filled with mystic energies!");
136:           Game.message("You learn the "+s.getName()+" spell");    
137:           flags=flags|ITEM_IDENTIFIED;
138:             } else {
139:           if (!isIdentified()) Game.message("You recognise this as a spell book of "+spell.getName());
140:           else Game.message("You already know "+spell.getName());
141:           flags=flags|ITEM_IDENTIFIED;
142:         }
143:         if (RPG.d(2)==1) {
144:           Game.message("The spellbook vanishes in a puff of smoke.");
145:           remove();
146:         }
147:       } else {
148:           Game.message("The runes are too complex for you to decipher");
149:         }
150:       } else {    
151:         Game.message ("You can't read!");
152:       }
153:     }
154: 
155:   public int getStat(int s) {
156:     switch (s) {
157:       case RPG.ST_ITEMVALUE: return isIdentified()?(750*spell.level*spell.level):1500;
158:       default: return super.getStat(s); 
159:     } 
160:   }
161: 
162:   public Description getDescription() {return this;}
163:   
164:   // Description interface 
165:   public String getName(int number, int article)    {
166:     if (isIdentified()) return Describer.getArticleName((number>1)?"spell books of ":"spell book of "+spell.getName(),number,article);    
167:     else return desc.getName(number,article);
168:   }
169:   
170:      public String getPronoun(int number, int acase) {
171:        return "it";    
172:      }
173:      
174:   public String getDescriptionText() {
175:     if (!isIdentified()) return desc.getDescriptionText();
176:     else return desc.getDescriptionText()+" ("+Text.capitalise(getName(1,ARTICLE_INDEFINITE))+")";
177:   }
178: }
