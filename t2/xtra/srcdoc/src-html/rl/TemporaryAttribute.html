12:     modifiers[0]=attribute;
13:     modifiers[1]=modifier;
14:   }
15: 
16:   public TemporaryAttribute(String name, int duration, int attribute, int modifier) {
17:     this(duration,attribute,modifier);
18:     attributestring=name;

21:   public TemporaryAttribute(String name, int duration, int[] mods, String end) {
22:     timeleft=duration;
23:     attributestring=name;
24:     modifiers=mods;
25:     endmessage=end;
26:   }
27:   
28:   // called for each time period
29:   public void action(int time) {
30:     if (time>=timeleft) time=timeleft;
31:     
32:     doEffect(time);
33:     
34:     timeleft-=time;
35:     if (timeleft<=0) {
36:       if ((place==Game.hero)&&(endmessage!=null)) Game.message(endmessage);
37:       remove();
38:     }
39:   }
40: 
41:   // placeholder for effects
42:   public void doEffect(int time) {
43:     
44:   }    
45:   
46:   // return stat modifier
47:   //   method: loop through modifiers array
48:   public int getModifier(int s) {
49:     int length=modifiers.length;
50:     for (int i=0; i<length ; i=i+2) {
51:       if (modifiers[i]==s) return modifiers[i+1];
52:     } 
53:     return 0;
54:   }
55:   
56:   // return attribute description
57:   public String getAttributeString() {
58:     if (QuestApp.debug) {
59:       int s=modifiers[1];
60:       return attributestring+" {"+ ((s>0) ? "+"+s : ""+s) +"}";
61:     }
62:       
63:     return attributestring;
64:   }
