080:   public void talk() {
081:     Hero h=Game.hero;
082:     
083:     switch(creaturetype) {
084:       // wizard identifies stuff
085:       case WIZARD: {
086:         Game.message("\"You want Identification, Training or Advice?\" (i/t/a)");
087:         char c=Game.getOption("ita");
088:         Game.messagepanel.setText("");
089:         if (c=='i') {
090:           if (h.getMoney()>=500) { 
091:             Thing o=Game.selectItem("Select item to sell:",h.inv.getContents());
092:             if (o instanceof Item) {
093:               ((Item)o).identify(10);
094:               Game.message(getTheName()+" inspects "+o.getTheName()+" carefully.");
095:               Game.message("\""+Text.capitalise(o.getFullName())+", I believe.\"");
096:               h.addMoney(-500);
097:             } else {
098:               Game.message("\"Hmmm.... come back later eh?\""); 
099:             }
100:           } else {
101:             Game.message("\"My talents cost 5 gold!\"");
102:             Game.message("\"Get yerself a job!\""); 
103:           }
104:         } else if (c=='t') {
105:           Game.message("\"Hmmm.... go practice first.\""); 
106:           Game.message("\"Then I'll give you a lesson!\""); 
107:         } else {
108:           Game.message("\""+chat()+"\""); 
109:         }
110:         break;
111:       }
112:       
113:       case BLACKSMITH: {
114:         Game.message("\"Yo Ho.\"");
115:         Game.message("\"You be needin' Repairs, Tools or Lessons?\" (r/t/l)");
116:         
117:         char c=Game.getOption("rtla");
118:         
119:         Game.messagepanel.setText("");
120:         if (c=='r') {
121: 
122:           Game.messagepanel.setText("");
123:           Thing o=Game.selectItem("Select item to sell:",h.inv.getContents());
124:           if (o instanceof Item) {
125:             int q=o.getQuality();
126:             if ((q<=0)||(q>5)) {
127:               Game.message("\"Can't do much with that, I'm afraid.\"");
128:               break;
129:             }
130: 
131:             int val=o.getStat(RPG.ST_ITEMVALUE)/2;  
132:             val=RPG.niceNumber(val);
133:             int funds=h.getMoney();
134: 
135:             Game.message("\"It'll cost "+Coin.valueString(val)+" to repair your "+o.getName()+((val<=funds)?".\" (y/n)":".\""));
136:             if ((val<=funds)&&(Game.getOption("yn")=='y')) {
137:               h.addMoney(-val);
138:               ((Item)o).quality++;           
139:               Game.message("\"There you are.... good as new!\"");
140:             } else {
141:               Game.message("\"Well, come back when you need it doing!\"");
142:               h.addThing(o);
143:             }
144:             break;
145:           }
146: 
147:         } else if (c=='t') {
148:           Game.message("Hmmm.... no stock right now. Come back in a day or so, eh?");

152:         break;
153:       }

156:       case RANGER: {
157:         if (RPG.d(3)==1) {
158:           Game.message("\"Do you need to buy any supplies?\" (y/n)"); 
159:           if (Game.getOption("yn")=='y') {
160:             Game.messagepanel.setText("");
161:             Thing o=Game.selectItem("What would you like to buy?",RANGERSTUFF);
162:             if (o instanceof Item) {
163:               o=(Thing)o.clone();
164:               int cost=o.getStat(RPG.ST_ITEMVALUE)*o.getNumber();
165:               Game.message(o.getTheName()+" will cost "+Coin.valueString(cost));
166:               int funds=h.getMoney();
167:               if (cost<=funds) {
168:                 Game.message("Buy? (y/n)");
169:                 if (Game.getOption("yn")=='y') {
170:                   Game.messagepanel.setText("");
171:                   h.addThing(o);
172:                   h.addMoney(-cost);
173:                 } 
174:               }
175:             } else {
176:               Game.message("See you later!"); 
177:             }
178:           }
179:         } else {
180:           Game.message("\""+chat()+"\""); 
181:         }
182:         break;
183:       }

187:         Game.message("\"Wanna sell something?\" (y/n)");
188:         if (Game.getOption("yn")=='y') {
189:           Game.messagepanel.setText("");
190:           Thing o=Game.selectItem("Select item to sell:",h.inv.getContents());
191:           if (o instanceof Item) {
192:             if ((o instanceof Stack)&&(((Stack)o).number>1)) {
193:               Game.message("Sell how many? (Enter=All)");
194:               char c=Game.getOption("0123456789");
195:               if ((c>='0')&&(c<='9')) {
196:                 o=o.remove((int)c-(int)'0');
197:               }
198:             }
199:             o.remove(); 
200:             int val=o.getStat(RPG.ST_ITEMVALUE)*o.getNumber();  
201:             int funds=getMoney();
202:             val=val/3;
203:             if (val>funds) val=funds;
204:             val=RPG.niceNumber(val);
205:             if (val<=0) {
206:               Game.message("\"Sorry, can't give you anything for that.\""); 
207:               h.addThing(o);
208:               break;
209:             }
210:             Game.message("\"I'll give you "+Coin.valueString(val)+" for your "+o.getName()+".\" (y/n)");
211:             if (Game.getOption("yn")=='y') {
212:               h.addMoney(val);
213:               addMoney(-val);
214:               o.remove();
215:               addThing(o);           
216:               Game.message("\"Nice doing business with yer!\"");
217:             } else {
218:               Game.message("\"Well, you won't get a better price!\"");
219:               h.addThing(o);
220:             }
221:             break;
222:           }
223:         } else {
224:           Game.message("\"Feel free to browse!\""); 
225:         }
226:         break; 
227:       }
228:       
229:       // priest gives you a fate point if you haven't got one
230:       case PRIEST: {
231:         h.setStat(RPG.ST_FATE,RPG.max(1,h.getStat(RPG.ST_FATE))); 
232:         Game.message("\""+chat()+"\"");
233:         break;
234:       }
235:       
236:       // standard chat response
237:       default: Game.message("\""+chat()+"\"");
238:     }
