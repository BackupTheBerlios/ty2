01: //
02: // AI for throwing and firing missile weapons
03: //
04: // Not used directly by any creatures, but can be called to provide
05: // targetting AI. e.g. NastyCritterAI will call ShooterAI when it decides
06: // to try a ranged attack
07: //
08: //
09: 
10: package rl;
11: 
12: public class ShooterAI extends BaseAI {
13:   public static final ShooterAI instance=new ShooterAI();
14:   
15:   public boolean doAction(Mobile m) {
16:     Hero h=Game.hero;
17:     Being b=(Being)m;
18:     Map map=b.getMap();
19: 
20:     m.aps-=10; // pause to think
21: 
22:     if (map==null) return false;
23: 
24:     if (b.isVisible()) {
25:       // find weapons
26:       RangedWeapon rw=(RangedWeapon)b.getWielded(RPG.WT_RANGEDWEAPON);
27:       Missile miss=(Missile)b.getWielded(RPG.WT_MISSILE);
28:     
29:       Thing p=map.findNearestFoe(m);
30:       if ((p!=null)&&m.canSee(p)) {
31:         if (rw==null) {
32:           if ((miss!=null)&&(miss.getStat(RPG.ST_MISSILETYPE)==0)) {
33:             ((Missile)miss.remove(1)).throwAt(b,map,p.x,p.y);
34:           } else {
35:             return false; 
36:           }
37:         } else if (rw.isValidAmmo(miss)) {
38:           rw.fireAt(b,(Missile)miss.remove(1),map,p.x,p.y);
39:         }  
40:         b.aps+=10; // make up for lost time
41:         return true;
42:       }
43:       return false;
44:     } else {
45:       return false; 
46:     }
47:   }
48: }
