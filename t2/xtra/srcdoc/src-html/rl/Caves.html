076:   // Make the caves. Pure Art
077:   public void makeCaves(int x1,int y1,int x2,int y2) {
078:     for (int buildloop=0;buildloop<200;buildloop++) {
079:       
080:       // first choose a direction

082:       int dy=0;

084:         case 1: dx=1; break;  //W

086:         case 3: dy=1; break;  //N
087:         case 4: dy=-1; break;  //S
088:       }  
089:       
090:       // now find a free extension point
091:       // Point p=findFreeSquare();
092:       // p=findEdge(p.x,p.y,dx,dy);
093:       Point p=findEdgeSquare(dx,dy,0);
094:       if (p==null) continue;
095:       // advance onto blank square
096:       p.x+=dx;
097:       p.y+=dy;
098:       
099:       // choose new feature to add
100:       switch (RPG.d(7)) {
101:         case 1: case 2: 
102:           makeThinPassage(p.x,p.y,dx,dy,false);
103:           break;
104:         
105:         case 3:
106:           makeThinPassage(p.x,p.y,dx,dy,true);
107:           break;
108:           
109:         case 4: case 5: case 6:
110:           makeOvalRoom(p.x,p.y,dx,dy);
111:           break;
112:           
113:         case 7:
114:           makeCrevice(p.x,p.y,dx,dy);
115:           break;
116:           
117:         /*
118:         case 8: { 
119:           int tx=p.x+dx*RPG.d(2,12)+(RPG.r(5)-2)*RPG.d(5);
120:           int ty=p.y+dy*RPG.d(2,12)+(RPG.r(5)-2)*RPG.d(5);

146:   // makes a thin passage
147:   // put a wall around first half of passage
148:   // this should prevent early branching
149:   public boolean makeThinPassage(int x, int y, int dx, int dy, boolean diagonals) {
150:     int len=RPG.d(2,12);
151:     int size=RPG.d(4);
152:     
153:     if (!isBlank(x+size*dy,y-size*dx,x-size*dy+len*dx,y+size*dx+len*dy)) return false;
154:     
155:     fillArea(x+size*dy,y-size*dx,x-size*dy+len*dx/2,y+size*dx+len*dy/2,walltile);   
156:     int p=0;
157:     for (int i=0; i<=len; i++) {
158:       setTile(x+i*dx+p*dy,y+i*dy-p*dx,floortile);
159:       if (RPG.d(2)==1) p=RPG.middle(-size,p+RPG.r(2)-RPG.r(2),size);
160:       if (!diagonals) setTile(x+i*dx+p*dy,y+i*dy-p*dx,floortile);
161:     }
162:     return true;
163:   }

165:   // nooks and crannies
166:   private boolean makeCrevice(int x,int y, int dx, int dy) {
167:     if (!isBlank(x+dy,y-dx,x-dy,y+dx)) return false;
168:     setTile(x+dy,y-dx,walltile); 
169:     setTile(x-dy,y+dx,walltile); 
170:     
171:     switch (RPG.d(10)) {
172:       case 1: // oval room continuation
173:         setTile(x,y,floortile);
174:         makeOvalRoom(x+dx,y+dy,dx,dy);
175:         break;
176:       
177:       case 2: // trap!
178:         setTile(x,y,floortile);
179:         addThing(new HeroTrap(getLevel()),x,y);
180:         if ((RPG.d(3)==1)&&(getTile(x+dx,y+dy)==0)) {
181:           setTile(x+dx,y+dy,floortile);
182:           if (RPG.d(3)==1) addThing(Lib.createItem(getLevel()),x+dx,y+dy); 
183:         }
184:         break;
185:         
186:       case 3: // recurse!
187:         setTile(x,y,floortile);  
188:         makeCrevice(x+dx+(RPG.r(3)-1)*dy,y+dy-(RPG.r(3)-1)*dx,dx,dy);
189:         break;
190:       
191:       case 4: // secret route
192:         makeSecretDoor(x,y);
193:         if (RPG.d(2)==1) {
194:           makeCrevice(x+dx,y+dy,dx,dy);
195:         } else {
196:           makeOvalRoom(x+dx,y+dy,dx,dy); 
197:         }
198:         break;
199:         
200:       default: setTile(x,y,floortile); 
201:     }
202:     
203:     return true;
204:   }

211:   // oval cave, pretty cool
212:   private void makeOvalRoom(int x,int y, int dx, int dy) {
213:     int w=RPG.d(7); 
214:     int h=RPG.d(7);
215:     int x1=x+(dx-1)*w;
216:     int y1=y+(dy-1)*h;
217:     int x2=x+(dx+1)*w;
218:     int y2=y+(dy+1)*h;
219:       
220:     if (!isBlank(x1,y1,x2,y2)) return;
221:     
222:     int cx=(x1+x2)/2;
223:     int cy=(y1+y2)/2;
224:     
225:     for (int lx=x1; lx<=(x1+w*2); lx++) for (int ly=y1; ly<(y1+h*2); ly++) {
226:       if ( (((lx-cx)*(lx-cx)*100)/(w*w) + ((ly-cy)*(ly-cy)*100)/(h*h)) < 100 )
227:            setTile(lx,ly,floortile);
228:     }    
229: 
230:     fillArea(cx,cy,x,y,floortile); 
231:   }

233:   // make a twisty path
234:   private boolean makePath(int x1,int y1, int x2, int y2) {
235:     int dx; int dy;
236:     while ((x1!=x2)||(y1!=y2)) {
237:       if (getTile(x1,y1)==0) {
238:         setTile(x1,y1,floortile);
239:       } else {
240:         // bail out if truly blocked
241:         if (isBlocked(x1,y1)) return false; 
242:       } 
243:       if (RPG.d(3)==1) {
244:         dx=RPG.sign(x2-x1);  
245:         dy=RPG.sign(y2-y1);  
246:       } else {
247:         dx=RPG.r(3)-1;  
248:         dy=RPG.r(3)-1;  
249:       }
250:       switch (RPG.d(3)) {
251:         case 1: dx=0; break;
252:         case 2: dy=0; break;  
253:       }
254:       x1+=dx;
255:       y1+=dy;
256:       x1=RPG.middle(1,x1,width-1);
257:       y1=RPG.middle(1,y1,height-1);
258:     }
259:     return true;  
260:   }

262:   // make a random twisty tunnel
263:   public boolean makeTunnel(int x, int y, int dx, int dy) {
264:     if (getTile(x,y)==0) {
265:       int ndx=dx; int ndy=dy;
266:       setTile(x,y,floortile);
267:       if (RPG.d(3)==1) {ndx=-dy; ndy=dx;}
268:       if (RPG.d(4)==1) {ndx=dy; ndy=-dx;}
269:       makeTunnel(x+ndx,y+ndy,ndx,ndy);  
270:       return true;
271:     }  
272:     return false;
273:   }

276:   public void addBeasties(int x, int y) {
277:     Creature m=Creature.createCreature(Creature.spawn,getLevel());
278:     for (int i=RPG.d(Creature.numbers[m.creaturetype]); i>0; i--) {
279:       int bx=x-2+RPG.r(5);
280:       int by=y-2+RPG.r(5);
281:       if (isClear(bx,by)) {
282:         addThing(m.cloneType(),bx,by);
283:       }  
284:     }
285:   }

312:         addBaddie(x,y);

314:         addBeasties(x,y);

316:     }
