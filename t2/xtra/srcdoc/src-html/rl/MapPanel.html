062:       public void mousePressed(MouseEvent e) {
063:         int dir=(e.getModifiers() & InputEvent.BUTTON3_MASK)
064:             == InputEvent.BUTTON3_MASK ? -1 : 1;
065: 
066:         Rectangle rect=getBounds();
067:         int rw=rect.width;
068:         int rh=rect.height;
069:         int x=(e.getX()-(rw-width*TILEWIDTH)/2)/TILEWIDTH+scrollx;
070:         int y=(e.getY()-(rh-height*TILEHEIGHT)/2)/TILEHEIGHT+scrolly;
071: 
072:         if (dir==1) {
073:           
074:           Mobile h=Game.hero;
075:           int rx=x-h.x;
076:           int ry=y-h.y;
077:           if ((rx==0)&&(ry==0)) return;
078:           int dx=-1;
079:           if (ry<(2*rx)) dx++;
080:           if (ry>(-2*rx)) dx++;
081:           int dy=-1;
082:           if (rx<(2*ry)) dy++;
083:           if (rx>(-2*ry)) dy++;
084:           Game.simulateDirection(dx,dy);
085:         } else {
086:           if (QuestApp.debug) {
087:             map.addThing(Game.hero,x,y);
088:             repaint();
089:           }
090:         }  
091:       } 
092:             
093:         });

113:   // draws tiles in box (x1,y1)-->(x2,y2) to back buffer
114:   public void drawTiles(int x1, int y1, int x2, int y2) {
115:       //do LOS calculations      
116:       map.calcVisible(Game.hero.x,Game.hero.y,map.visibleRange());
117:       
118:       //draw the specified area in buffer
119:       for (int y=y1; y<=y2 ; y++ ) {
120:           for (int x=x1 ; x<=x2 ; x++) {
121:               drawTile(x,y);
122:           }
123:       }
124:     }

126:     // draws tile at (x,y)
127:     // includes logic for mapping tile number to image
128:     // draw blank if outside map area
129:     public void drawTile(int x, int y) {
130:     int m=map.getTile(x,y);
131: 
132:     if (!map.isDiscovered(x,y)) m=0;
133:     Image source=map.isVisible(x,y)?QuestApp.tiles:QuestApp.greytiles; 
134:       
135:     int tile=m&65535;
136:     int px=(x-scrollx)*TILEWIDTH;
137:     int py=(y-scrolly)*TILEHEIGHT;
138:     
139:     switch (tile) {
140:       case 0:    //blank
141:         buffergraphics.setColor(Color.black);
142:         buffergraphics.fillRect((x-scrollx)*TILEWIDTH,(y-scrolly)*TILEHEIGHT,TILEWIDTH,TILEHEIGHT);
143:         break;
144:     
145:       // default method
146:       default: {
147:         int image;
148:         if (map.isDiscovered(x,y+1)&&Tile.filling[map.getTile(x,y+1)&65535]) {
149:           image=Tile.imagefill[tile];
150:         } else {
151:           image=Tile.images[tile];
152:         }
153:          
154:         int sx=(image%20)*TILEWIDTH; 
155:         int sy=(image/20)*TILEHEIGHT; 
156:         buffergraphics.drawImage(source,px,py,px+16,py+16,sx,sy,sx+TILEWIDTH,sy+TILEHEIGHT,null);      
157:       
158:         // draw in coastlines
159:         if ((Tile.borders[tile]>0)&&(source==QuestApp.tiles)) {
160:           if ((x>0)              && (map.getTile(x-1,y)!=m)) buffergraphics.drawImage(QuestApp.scenery,px,py,px+16,py+16,0,256,16,256+16,null);
161:           if ((x<(map.width-1))  && (map.getTile(x+1,y)!=m)) buffergraphics.drawImage(QuestApp.scenery,px,py,px+16,py+16,16,256,32,256+16,null);
162:           if ((y>0)              && (map.getTile(x,y-1)!=m)) buffergraphics.drawImage(QuestApp.scenery,px,py,px+16,py+16,32,256,48,256+16,null);
163:           if ((y<(map.height-1)) && (map.getTile(x,y+1)!=m)) buffergraphics.drawImage(QuestApp.scenery,px,py,px+16,py+16,48,256,64,256+16,null);
164:         }
165:         break;
166:       }
167:       //end of switch   
168:     }
169:         
170:     if (map.isVisible(x,y)) drawThings(x,y);
171:     }

173:     // Draw all visible objects on map to back buffer
174:     // side effect: sorts map objects in increasing z-order    
175:     private void drawThings(int x,int y) {
176:     Image im=null;
177:     Thing head=map.sortZ(x,y);
178:     if (head==null) return;
179:     int px=(x-scrollx)*TILEWIDTH;
180:     int py=(y-scrolly)*TILEHEIGHT;
181:     do {
182:       int i=head.getImage();
183:       int sx=(i%20)*TILEWIDTH;
184:       int sy=(i/20)*TILEHEIGHT;    
185:       if (!head.isInvisible()) {
186:         if (head instanceof Special)    im=QuestApp.scenery;        
187:         if (head instanceof Mobile) im=QuestApp.creatures;        
188:             if (head instanceof Item) im=QuestApp.items;  
189:             buffergraphics.drawImage(im, px,py,px+16,py+16, sx ,sy,sx+TILEWIDTH,sy+TILEHEIGHT,null);        
190:       }
191:       head=head.next;
192:     } while (head!=null);
193:     }

209:   // animates a shot from (x1,y1) to (x2,y2)
210:   public void doShot(int x1, int y1, int x2, int y2, int c, double speed) {
211:     int rx=x2-x1;
212:     int ry=y2-y1;
213:     if ((rx==0)&&(ry==0)) return;
214:     double d=Math.sqrt(rx*rx+ry*ry);
215:     double dx=rx*(speed/d);
216:     double dy=ry*(speed/d);
217:     double x=x1;
218:     double y=y1;
219:     int sx=TILEWIDTH*(c%20);
220:     int sy=TILEHEIGHT*(c/20);
221:     drawTiles(scrollx,scrolly,scrollx+width-1,scrolly+height-1);
222:     for (int i=0; i<(int)(d/speed); i++) {
223:       int px=(int) ((x-scrollx)*TILEWIDTH*zoomfactor/100);
224:       int py=(int) ((y-scrolly)*TILEHEIGHT*zoomfactor/100);
225:       Graphics g=getGraphics();
226:       if (g!=null) {
227:         Rectangle rect=getBounds();
228:         int ox=(rect.width-width*TILEWIDTH*zoomfactor/100)/2;
229:         int oy=(rect.height-width*TILEHEIGHT*zoomfactor/100)/2;
230:         drawMap(g);
231:         g.drawImage(QuestApp.effects,px+ox,py+oy,px+ox+TILEWIDTH*zoomfactor/100,py+oy+TILEHEIGHT*zoomfactor/100,sx,sy,sx+TILEWIDTH,sy+TILEHEIGHT,null);
232:       }
233:       x+=dx;
234:       y+=dy;
235:     }
236:     repaint();
237:   }

239:     // makes an explosion (3*3) of the specified style
240:     public void doExplosion(int x, int y, int c, int r) {
241:         if (!map.isVisible(x,y)) return;
242:         int explosiontype=c;
243:         for (int i=0;i<=5;i++) {
244:             drawTiles(scrollx,scrolly,scrollx+width-1,scrolly+height-1);
245:       int px=(x-scrollx)*TILEWIDTH;
246:       int py=(y-scrolly)*TILEHEIGHT;
247:       int sx=i*TILEWIDTH;
248:       int sy=TILEHEIGHT*explosiontype;
249:       for (int dy=-1;dy<=1;dy++) for (int dx=-1;dx<=1;dx++) {
250:         buffergraphics.drawImage(QuestApp.effects,px+3*r*dx*i,py+3*dy*r*i,px+3*r*dx*i+TILEWIDTH,py+3*r*dy*i+TILEHEIGHT,sx,sy,sx+TILEWIDTH,sy+TILEHEIGHT,null);
251:       }
252:            Graphics g=getGraphics();
253:            if (g!=null) drawMap(g);
254:         }
255:         repaint();
256:     }

258:     public void doSpark(int x, int y, int c) {
259:         if (!map.isVisible(x,y)) return;
260:         int explosiontype=c;
261:         for (int i=0; i<=5; i++) {
262:             drawTiles(scrollx,scrolly,scrollx+width-1,scrolly+height-1);
263:       int px=(x-scrollx)*TILEWIDTH;
264:       int py=(y-scrolly)*TILEHEIGHT;
265:       int sx=i*TILEWIDTH;

267:       buffergraphics.drawImage(QuestApp.effects,px,py,px+TILEWIDTH,py+TILEHEIGHT,sx,sy,sx+TILEWIDTH,sy+TILEHEIGHT,null);
268:            Graphics g=getGraphics();
269:            if (g!=null) drawMap(g);
270:         }

272:     }

300:   // standard paint method
301:     // - builds map image in back buffer then copies to screen
302:     public void paint(Graphics g) {
303:     Rectangle rect=getBounds();
304:         if (zoomfactor!=lastzoomfactor) {
305:       int w=rect.width;
306:       int h=rect.height;
307:       g.setColor(Color.black);
308:       g.fillRect(0,0,w,h);
309:       lastzoomfactor=zoomfactor;
310:     }
311:         
312:         // create back buffer if needed
313:     if (buffer==null) {
314:           buffer = createImage(width*TILEWIDTH,height*TILEHEIGHT);
315:           buffergraphics = buffer.getGraphics();
316:         }
317:         
318:         Rectangle b=rect; //g.getClipBounds();
319:         
320:         // draw area to back buffer
321:         drawTiles((b.x)/TILEWIDTH+scrollx,(b.y/TILEHEIGHT+scrolly),(b.x+b.width)/TILEWIDTH+scrollx,((b.y+b.height)/TILEHEIGHT+scrolly));
322:         if (cursor) drawCursor(curx,cury);
323:             
324:     drawMap(g);
325:      }
