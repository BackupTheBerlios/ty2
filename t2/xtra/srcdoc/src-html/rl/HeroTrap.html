001: package rl;
002: 
003: public class HeroTrap extends Trap {
004:   protected int type;
005:   protected Portal portal;
006:   
007:   public static final int ICETRAP=1;
008:   public static final int PITTRAP=2;
009:   public static final int BEARTRAP=3;
010:   public static final int FIRETRAP=4;
011:   public static final int GASTRAP=5;
012:   public static final int TELEPORTTRAP=6;
013:   public static final int AMBUSHTRAP=7;
014:   
015:   private static final Description DESC_HEROTRAP = 
016:         new Describer("trap","It's rather lucky that you spotted this dangerous trap.");
017:   private static final Description DESC_ICETRAP = 
018:     new Describer("ice trap","A magical ice trap.");
019:   private static final Description DESC_PITTRAP = 
020:     new Describer("pit trap","A deep and dangerous pit. Someone could have a nasty accident.");
021:   private static final Description DESC_BEARTRAP = 
022:     new Describer("bear trap","A bear trap.");
023:   private static final Description DESC_FIRETRAP = 
024:     new Describer("fire trap","A fire trap.");
025:   private static final Description DESC_GASTRAP = 
026:     new Describer("gas trap","A gas trap.");
027:   private static final Description DESC_TELEPORTTRAP = 
028:     new Describer("teleport trap","A teleport trap.");
029:   private static final Description DESC_AMBUSHTRAP = 
030:     new Describer("ambush trap","A magical ambush trap.");
031: 
032:   public static final int[]         images= {0,             47,           41,           42,            40,            45,           40,                44};
033:   public static final Description[]  descs= {DESC_HEROTRAP, DESC_ICETRAP, DESC_PITTRAP, DESC_BEARTRAP, DESC_FIRETRAP, DESC_GASTRAP, DESC_TELEPORTTRAP, DESC_AMBUSHTRAP};
034:   public static final int[]       usecount= {0,             6,            0,            3,             6,             4,            5,                 1};
035: 
036:   public HeroTrap() {
037:     this(Game.hero.getStat(RPG.ST_LEVEL));
038:   }
039:   
040:   public HeroTrap(int l) {
041:     this(RPG.d(images.length-1),l); 
042:   }
043: 
044:   public HeroTrap(int t, int l) {
045:     type=t;
046:     level=l;
047:     uses=RPG.d(usecount[type]);
048:   }
049:   
050:   public int getImage() {return images[type];}
051:     
052:   public void discover() {
053:         if (!discovered) Game.message("Whoa! That looks like a nasty trap!");
054:         appear();
055:     }
056:     
057:   // make trap appear, with any necessary modifications!
058:   public boolean appear() {
059:     if (!super.appear()) return false;
060:     switch (type) {
061:       case PITTRAP:
062:         Map map=getMap();
063:         portal=new Portal("pit");
064:         map.addThing(portal,x,y);
065:         Pit thepit=new Pit(level);
066:         thepit.entrance.setDestination(portal);
067:         Point ts=thepit.findFreeSquare(); 
068:         portal.setDestination(thepit,ts.x,ts.y);
069:         remove();
070:         break; 
071:     }
072:     return true;
073:   }
074:   
075:     public void affectSquare(Map map, int tx, int ty) {
076:     Hero h=Game.hero;
077:     if (map!=h.getMap()) return;
078:     
079:     Thing t=map.getMobile(tx,ty);
080:     boolean ishero=((h.x==tx)&&(h.y==ty));
081:     
082:     if (!discovered) {
083:       if (ishero) {
084:         Game.message("You have stepped on a trap!");
085:       }
086:     }
087: 
088:       switch (type) {
089:       case ICETRAP:
090:         if (ishero) Game.message("There is a sudden blast of deathly cold!");    
091:         Game.mappanel.doExplosion(x,y,1,4+level,RPG.DT_ICE);
092:         remove();
093:         break;
094:         
095:       case PITTRAP:
096:         if (discovered&&RPG.test(t.getStat(RPG.ST_AG),2*level)) {
097:           appear();
098:           if (ishero) Game.message("You sidestep the pit");
099:         } else {
100:           if (ishero) Game.message("You take a painful fall down a deep pit....."); 
101:           appear();
102:           portal.travel(t);
103:           t.damage(RPG.DT_IMPACT,RPG.a(2));
104:         } 
105:         break;
106:         
107:       case GASTRAP:
108:         if (ishero) Game.message("The air is filled with poisonous fumes!");  
109:         new Spell(Spell.POISONCLOUD).castAtLocation(null,map,x,y);
110:         appear();;
111:         break;
112: 
113:       case FIRETRAP:
114:         if (ishero) Game.message("There is a sudden blast of fire!");  
115:         new Spell(Spell.FIREBALL).castAtLocation(null,map,x,y);
116:         appear();;
117:         break;
118: 
119:       case BEARTRAP:
120:         if (ishero&&((!discovered)||RPG.test(50,h.getStat(RPG.ST_AG)))) {
121:           Game.message("The bear trap snaps shut around you leg!");
122:           h.addAttribute(new TemporaryAttribute("Stuck",15000/h.getStat(RPG.ST_ST),new int[] {RPG.ST_IMMOBILIZED,1},"You break free from the trap"));
123:         }
124:         appear();;
125:         break;
126:  
127:       case TELEPORTTRAP:
128:         new Spell(Spell.TELEPORT).castAtSelf(t);
129:         appear();;
130:         break;
131: 
132:       case AMBUSHTRAP:
133:         if (ishero) {
134:           int added=0;
135:           Creature c = Lib.createCreature(level);
136:           for (int dx=h.x-1; dx<=h.x+1; dx++) for (int dy=h.y-1; dy<=h.y+1; dy++) {
137:             if ((!map.isBlocked(dx,dy))&&(RPG.d(2)==1)) {
138:               map.addThing(c.cloneType(),dx,dy);
139:               added++; 
140:             } 
141:           }
142:           
143:           if (added>0) Game.message("Ambush!!"); else Game.message("You feel a strange draft of cold air");  
144:           remove();
145:         }
146:         break;
147: 
148:       default:
149:         if (ishero) Game.message("The trap appears to be broken");
150:         break;
151:     }
152:   }
153: 
154:   public Description getDescription() {return descs[type];}
155: }
