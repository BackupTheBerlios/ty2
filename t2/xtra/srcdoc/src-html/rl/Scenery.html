01: // Root class for scenic items and decorations on maps
02: 
03: package rl;
04: 
05: public class Scenery extends Special {
06:     protected int image;
07:     protected boolean blocking;
08:     protected Description desc;
09:     
10:     public Scenery() {
11:       image=40;
12:       blocking=false;    
13:     }
14:     
15:     public Scenery(int imagenumber, boolean solid, Description d) {
16:         image = imagenumber;
17:         blocking = solid;
18:         desc=d;
19:     }
20:     
21:     public void use() {
22:     }
23:     
24:   // consider scenery owned
25:   public boolean isOwned() {return true;}
26:   
27:   public boolean isBlocking() {return blocking;}
28:   
29:   // Can the item be pushed around?
30:   //   0 = definitely not
31:   //   1 = always
32:   //   x = yes, with difficulty increasing with x (compare to character ST)
33:   public int getPushability() {return 0;}
34:     
35:   public boolean isInvisible() {return false;}
36:   
37:   public int getImage() {return image;}
38:     
39:     public Description getDescription() {return desc;}
40:   
41:   public void die() {
42:     // local folk don't approve of vandalism!
43:     if (Game.actor==Game.hero) {
44:       getMap().areaNotify(x,y,6,AI.EVENT_THEFT,1,Game.hero);
45:     }
46:     super.die();
47:   }
48: }
