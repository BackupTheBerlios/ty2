003: // the critter is a neutral beast
004: // it hunts out food
005: // may attack anyone who comes too close....
006: 
007: // critters of similar type band together and all turn 
008: // nasty (NastyCritterAI) if one of them is attacked
009: 
010: public class CritterAI extends BaseAI {
011:     public static final CritterAI instance=new CritterAI();
012:     
013:   // action routine
014:   // MUST use up all aps
015:     public boolean doAction(Mobile m) {
016:       Hero h=Game.hero;
017:     Map map=m.getMap();
018: 
019:     // sniff for food - this is a slow algorithm 
020:     // so only do occasionally or if visible
021:     if (m.isVisible()||(RPG.d(40)==1)) {
022:       int dis=4; //search distance
023:       Thing[] search = map.getThings(m.x-dis,m.y-dis,m.x+dis,m.y+dis);
024:       for (int i=0; i<search.length; i++) {
025:         Thing t=search[i];
026:         if ((t.getUse()&Thing.USE_EAT)>0) {
027:           m.setStat(RPG.ST_TARGETX,t.x);
028:           m.setStat(RPG.ST_TARGETY,t.y);
029:           m.setStat(RPG.ST_AIMODE,1);
030:         }
031:       }
032:       m.aps-=50;
033:     }
034:     
035:     while (m.aps>=0) {
036:       int d=RPG.dist(h.x,h.y,m.x,m.y);
037:           
038:       int state=m.getStat(RPG.ST_AIMODE);
039:       
040:       int dx; int dy;
041:       
042:       if (state==0) {
043:         // wandering
044:         dx=RPG.r(3)-1;
045:         dy=RPG.r(3)-1;
046:   
047:         // do search for food
048:         
049:         // randomize movement
050:         if (d>5) {
051:           // could make them run... probably not worth it
052:           //if (RPG.d(2)==1) dx=-RPG.sign(h.x-m.x);
053:           //if (RPG.d(2)==1) dy=-RPG.sign(h.y-m.y);
054:         } else {
055:           // make them charge vaguely at hero
056:           if (RPG.d(2)==1) dx=RPG.sign(h.x-m.x);
057:           if (RPG.d(2)==1) dy=RPG.sign(h.y-m.y);
058:         }
059:       } else {
060:         // scrounging!
061:         int tx=m.getStat(RPG.ST_TARGETX);
062:         int ty=m.getStat(RPG.ST_TARGETY);
063:       
064:         dx=RPG.sign(tx-m.x);
065:         dy=RPG.sign(ty-m.y);
066:         
067:         
068:         if ((tx==m.x)&&(ty==m.y)) {
069:           Thing[] stuff=map.getThings(tx,ty);
070:           for (int i=0; i<stuff.length; i++) {
071:             Thing t=stuff[i];
072:             if ((t.getUse()&Thing.USE_EAT)>0) {
073:               t.use((Being)m);
074:             } 
075:           }
076:           m.setStat(RPG.ST_AIMODE,0);
077:         }
078:       }
079:       
080:       int nx=m.x+dx;
081:       int ny=m.y+dy;
082:       
083:       if (!map.isBlocked(nx,ny)) {
084:         m.moveTo(map,nx,ny);
085:         m.aps=m.aps-10000/m.getStat(RPG.ST_MOVESPEED);  
086:       } else {
087:         m.setStat(RPG.ST_AIMODE,0);
088:         Mobile t=map.getMobile(nx,ny);
089:         if ((t!=null)&&(t.isHostile(m))) {
090:           m.attack(t);    
091:         }    else {
092:           m.aps-=100; 
093:         }
094:       }
095:       }    
096:     return true;
097:     }    
098:   
099:   public int notify(Mobile m, int eventtype, int ext, Object o) {
100:     if ((eventtype==EVENT_ATTACKED)&&(o==Game.hero)&&(ext==m.getStat(RPG.ST_SIDE))) {
101:       turnNasty(m);
102:     }
103:     else if ((eventtype==EVENT_ALARM)&&(ext==m.getStat(RPG.ST_SIDE))) {
104:       turnNasty(m);
105:     }  
106:     return 0;
107:   }
108:   
109:   public void turnNasty(Mobile m) {
110:     // turn into nasty critter
111:     m.setAI(NastyCritterAI.instance);
112:     m.setStat(RPG.ST_STATE,AI.STATE_HOSTILE);
113:     
114:     //Map map=m.getMap();
115:     //if (map!=null) map.areaNotify(m.x,m.y,7,EVENT_ATTACKED,m.getStat(RPG.ST_SIDE),Game.hero);
116:   } 
117: }
