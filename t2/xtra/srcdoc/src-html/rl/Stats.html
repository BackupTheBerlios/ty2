01: // Represents an array of attributes
02: //
03: // A stats object is attacked to each Being to store character attributes,
04: // and may also be used as a handy way to store item modifiers
05: //
06: // Use getStat(index) and setStat(index,value) to modify stats
07: //
08: // The array automatically grows to accomodate new values
09: 
10: package rl;
11: 
12: public class Stats extends Object implements Cloneable, java.io.Serializable {
13:     // define default size for new array
14:     private static final int DEFAULTSIZE=5;
15:     private static final int DELTA=3;
16:     
17:     // fields are private to prevent any interference
18:     private int[] stats=new int[DEFAULTSIZE*2];
19:     private int size=DEFAULTSIZE;  // size of array
20:     private int count=0; // number of array elements used
21:     
22:     // default constructor - empty stats
23:     public Stats() {
24:     }
25:     
26:   // deep copy
27:   public Object clone() {
28:     try {
29:       Stats s=(Stats)super.clone();
30:       s.stats=(int[])s.stats.clone();
31:       return s;
32:     } catch (Exception e) {throw new Error("Stats clone error");}
33:   }
34:   
35:     // create Stats object with the first attributes specified
36:     public Stats(int[] mainstats) {
37:       ensureSize(mainstats.length+DELTA);
38:       count=mainstats.length;
39:       for (int i=0; i<count; i++) {
40:           stats[i*2]=i;
41:           stats[i*2+1]=mainstats[i];
42:       }
43:     }
44:     
45:     // this function makes sure that the array is large enough
46:     // must have room for newsize*(index,value) pairs
47:     private void ensureSize(int newsize) {
48:       if (newsize>size) {
49:         int[] newstats=new int[newsize*2];
50:         System.arraycopy(stats,0,newstats,0,count*2);
51:         stats=newstats;
52:         size=newsize;    
53:       }    
54:     }
55: 
56:     // returns the attribute with given index number
57:     public int getStat(int index) {
58:     for (int i=0; i<count; i++) {
59:         if (stats[i*2]==index) return stats[i*2+1];
60:     }
61:       return 0;
62:     }
63:     
64:     public void incStat(int index, int value) {
65:     // search the array, adding value if found
66:     for (int i=0; i<count; i++) {
67:         if (stats[i*2]==index) {
68:             stats[i*2+1]+=value; // increment
69:           return;    
70:         }
71:     }
72:       // grow the array and append new value
73:       ensureSize(count+DELTA);
74:       stats[count*2]=index;
75:       stats[count*2+1]=value;
76:       count=count+1;
77:     }
78:     
79:     
80:     // set the attribute with give index number
81:     public void setStat(int index, int value) {
82:     // search the array, updating value if found
83:     for (int i=0; i<count; i++) {
84:         if (stats[i*2]==index) {
85:             stats[i*2+1]=value;
86:           return;    
87:         }
88:     }
89:       // grow the array and append new value
90:       ensureSize(count+DELTA);
91:       stats[count*2]=index;
92:       stats[count*2+1]=value;
93:       count=count+1;
94:     }
95: }
