001: //
002: // Dungeon in traditional rogue style
003: //

005: package rl;
006: 
007: import java.util.Random;
008: 
009: class RogueDungeon extends Map {
010:   // size of map in terms of blocks
011:   protected int bh;
012:   protected int bw;
013: 
014:   // size of blocks in number of tiles
015:   protected int th;
016:   protected int tw;
017:   
018:   // room links
019:   protected static final int LN=1;
020:   protected static final int LE=2;
021:   protected static final int LS=4;
022:   protected static final int LW=8;
023:   protected int[][] links;
024:   protected int[][] linkedzones;
025:   
026:   // room dimensions
027:   protected int[][] rx;
028:   protected int[][] ry;
029:   protected int[][] rw;
030:   protected int[][] rh;
031:   
032:   
033:   private RogueDungeon(int width, int height, int tilewidth, int tileheight) { 
034:     super(width*tilewidth,height*tileheight);
035:     try {
036:       th=tileheight;
037:       tw=tilewidth;
038:       bw=width;
039:       bh=height;
040:     
041:       rx=new int[width][];
042:       ry=new int[width][];
043:       rw=new int[width][];
044:       rh=new int[width][];
045:       links=new int[width][];
046:       linkedzones=new int[width][];
047:     
048:       for (int w=0; w<width; w++) {
049:         rx[w]=new int[height]; 
050:         ry[w]=new int[height]; 
051:         rw[w]=new int[height]; 
052:         rh[w]=new int[height]; 
053:         links[w]=new int[height]; 
054:         linkedzones[w]=new int[height]; 
055:       } 
056:     } catch (Exception e) {
057:       Game.message("Failed to construct RogueDungeon"); 
058:       Game.message(e.toString());
059:     }
060:   }  
061:   
062:   public static Map create(int l, Theme theme) {
063:     try {
064:       RogueDungeon map=new RogueDungeon(3+RPG.r(l/3),3+RPG.r(l/3),8+RPG.d(l),8+RPG.d(l));
065:       map.setLevel(l);
066:       map.setTheme(theme);
067:       map.build();
068:       return map;
069:     } catch (Exception e) {
070:       Game.message("Failed to create RogueDungeon"); 
071:       Game.message(e.toString());
072:       return new Dungeon(50,50,l);
073:     }
074:   }
075:   
076:   public void replaceLinkedZones(int fromzone, int tozone) {
077:     for (int lx=0; lx<bw; lx++) for (int ly=0; ly<bh; ly++) {
078:       if (linkedzones[lx][ly]==fromzone) linkedzones[lx][ly]=tozone; 
079:     }
080:     // System.out.println(fromzone+" -> "+tozone);
081:   }
082:   
083:   // link two adjacent linkedzones
084:   public void makeLink(int x, int y, int dx, int dy) {
085:     if (dx==1) {
086:       links[x][y]|=LE; 
087:       links[x+dx][y+dy]|=LW; 
088:     }
089:     if (dx==-1) {
090:       links[x][y]|=LW; 
091:       links[x+dx][y+dy]|=LE; 
092:     }
093:     if (dy==1) {
094:       links[x][y]|=LS; 
095:       links[x+dx][y+dy]|=LN; 
096:     }
097:     if (dy==-1) {
098:       links[x][y]|=LN; 
099:       links[x+dx][y+dy]|=LS; 
100:     }
101:     
102:     try {
103:       int fromzone=linkedzones[x][y];
104:       int tozone=linkedzones[x+dx][y+dy];
105:       replaceLinkedZones(fromzone,tozone);
106:     } catch (Exception e) {
107:       Game.message(e.toString()+" ("+x+","+y+") ("+(x+dx)+","+(y+dy)+")"); 
108:     }
109:   }
110:   
111:   // ensure the whole map is a single zone
112:   public void makeConnected() {
113:     
114:     boolean connected=false;
115:     for (int i=0; (i<100000)&&(!connected); i++) {
116:       int tx=RPG.r(bw);
117:       int ty=RPG.r(bh);
118:       int dx=RPG.r(3)-1;
119:       int dy=RPG.r(3)-1;
120:       
121:       while (
122:               ( ((dy==0)&&(dx==0)) || ((dx*dy)!=0) )
123:            || ((tx+dx)<0)||((tx+dx)>=bw)||((ty+dy)<0)||((ty+dy)>=bh)
124:            || (linkedzones[tx][ty]==linkedzones[tx+dx][ty+dy])
125:         ) 
126:       {
127:         tx=RPG.r(bw);
128:         ty=RPG.r(bh);
129:         dx=RPG.r(3)-1;
130:         dy=RPG.r(3)-1;
131:       }
132:       
133:       makeLink(tx,ty,dx,dy);
134:       
135:       // test to see if everything is connected
136:       int minzone=10000;
137:       int maxzone=0;
138:       for (int x=0; x<bw; x++) for (int y=0; y<bh; y++) {
139:         int z=linkedzones[x][y];
140:         if (z<minzone) minzone=z;
141:         if (z>maxzone) maxzone=z;
142:       }   
143:       connected=(minzone==maxzone);
144:     };
145:     
146:   }
147:   
148:   public void createDoor(int tx, int ty) {
149:     switch(RPG.d(3)) {
150:       case 1: 
151:         setTile(tx,ty,floortile);
152:         break; 
153:       
154:       default:
155:         theme.createDoor(this,tx,ty,getLevel());
156:     } 
157:   }
158:   
159:   public void build() {
160:     if (constructed) return;
161:     
162:     // set up lots of individual linkedzones
163:     for (int x=0; x<bw; x++) for (int y=0; y<bh; y++) {
164:       linkedzones[x][y]=x+bw*y; 
165:     }
166:     
167:     // connect all the linkedzones
168:     makeConnected();
169:     
170:     // set the room dimensions (ex. wall)
171:     for (int x=0; x<bw; x++) for (int y=0; y<bh; y++) {
172:       rw[x][y]=RPG.rspread(2,tw-3);
173:       rh[x][y]=RPG.rspread(2,th-3);
174:       rx[x][y]=x*tw+RPG.d(tw-rw[x][y]-2); 
175:       ry[x][y]=y*th+RPG.d(th-rh[x][y]-2); 
176:     }
177:     
178:     // make all the rooms
179:     for (int x=0; x<bw; x++) for (int y=0; y<bh; y++) {
180:       fillArea(rx[x][y],ry[x][y],rx[x][y]+rw[x][y]-1,ry[x][y]+rh[x][y]-1,floortile); 
181:     }
182:     
183:     // make all the corridors, looking E
184:     for (int x=0; x<(bw-1); x++) for (int y=0; y<bh; y++) {
185:       if ((links[x][y]&LE)>0) {
186:         int py1=ry[x][y]+RPG.r(rh[x][y]); 
187:         int py2=ry[x+1][y]+RPG.r(rh[x+1][y]); 
188:         
189:         int minx=rx[x][y]+rw[x][y]+1;
190:         int maxx=rx[x+1][y]-2;
191:         
192:         createDoor(minx-1,py1);
193:         createDoor(maxx+1,py2);
194: 
195:         int px=RPG.rspread(minx,maxx);
196:         
197:         fillArea(minx,py1,px,py1,floortile);
198:         fillArea(px,py1,px,py2,floortile);
199:         fillArea(px,py2,maxx,py2,floortile);
200:       }      
201:     }
202:     
203:     // make all the corridors, looking S
204:     for (int x=0; x<bw; x++) for (int y=0; y<(bh-1); y++) {
205:       if ((links[x][y]&LS)>0) {
206:         int px1=rx[x][y]+RPG.r(rw[x][y]); 
207:         int px2=rx[x][y+1]+RPG.r(rw[x][y+1]); 
208:         
209:         int miny=ry[x][y]+rh[x][y]+1;
210:         int maxy=ry[x][y+1]-2;
211:         
212:         createDoor(px1,miny-1);
213:         createDoor(px2,maxy+1);
214:         
215:         int py=RPG.rspread(miny,maxy);
216:         
217:         fillArea(px1,miny,px1,py,floortile);
218:         fillArea(px1,py,px2,py,floortile);
219:         fillArea(px2,py,px2,maxy,floortile);
220:       }      
221:     }
222:     
223:     replaceTiles(0,walltile);
224:     
225:     constructed=true; 
226:   }
227:   
228: }
