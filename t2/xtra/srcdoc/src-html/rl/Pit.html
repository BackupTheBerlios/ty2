128:     if (m==Game.hero) switch (type) {
129:       case 1000: if ((x==1)&&(y>1)) {
130:         clearArea(2,1,width-2,height-2);
131:         for (int px=2; px<width-1; px++) for (int py=1; py<(height-1); py++) {
132:           Item it=Lib.createItem(RPG.r(10));
133:           // it.flags|=Item.ITEM_IDENTIFIED;
134:           addThing(it,px,py); 
135:         } 
136:         break;
137:       }  
138: 
139:       case 1001: if ((x==1)&&(y>1)) {
140:         clearArea(2,1,width-2,height-2);
141:         for (int px=2; px<width-1; px++) for (int py=1; py<(height-1); py++) {
142:           Item it;
143:           switch(y) {
144:             case 2: case 3: it=CommonWeapon.createWeapon(RPG.d(px*4)); break;
145:             default: it=Lib.createItem(RPG.d(px*4)); break;
146:           }
147:           // it.flags|=Item.ITEM_IDENTIFIED;
