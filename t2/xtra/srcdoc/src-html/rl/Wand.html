001: // A magic wand casts a particular type of spell
002: // It has it's own supply of magic points (mps)
003: //
004: // It can be recharged, although this will occasionally destroy the wand
005: //
006: // If broken, the wand will fire it's ramianing charge randomly
007: 
008: package rl;
009: 
010: public class Wand extends Item implements Description {
011:     private static final Description DESC_WILLOWWAND=
012:         new Describer("willow wand","A willow wand");
013:     private static final Description DESC_METALWAND=
014:         new Describer("metal wand","A metal wand");
015:     private static final Description DESC_CRYSTALWAND=
016:         new Describer("crystal wand","A crystal wand");
017:     private static final Description DESC_BLUEWAND=
018:         new Describer("blue wand","A painted blue wand");
019:     
020:     protected Spell spell;
021:     protected Description desc;
022:     protected int mps;
023:     
024:     public Wand() {
025:       switch (RPG.d(4)) {
026:         case 1:
027:             desc=DESC_WILLOWWAND;    image=288;
028:           switch (RPG.d(10)) {
029:               case 1: spell=new Spell(Spell.SPARK); break;
030:               case 2: spell=new Spell(Spell.POISONBOLT); break;
031:               case 3: spell=new Spell(Spell.ACIDBLAST); break;
032:               case 4: spell=new Spell(Spell.FORCEBOLT); break;
033:               case 5: spell=new Spell(Spell.THUNDERBOLT); break;
034:          case 6: spell=new Spell(Spell.IDENTIFY); break;
035:            default: spell=new Spell(Spell.randomSpell()); break;
036:         }
037:           break;
038:         case 2:
039:             desc=DESC_METALWAND;    image=292;
040:           switch (RPG.d(5)) {
041:           case 1: spell=new Spell(Spell.FIREBALL); break;
042:           case 2: spell=new Spell(Spell.FLAMECLOUD); break;
043:           case 3: spell=new Spell(Spell.BLAST); break;
044:           case 4: spell=new Spell(Spell.BLAZE); break;
045:           default: spell=new Spell(Spell.randomSpell()); break;
046:           }
047:         break;            
048:       case 3:
049:         desc=DESC_CRYSTALWAND; image=290;    
050:         switch (RPG.d(4)) {
051:           case 1: spell=new Spell(Spell.TELEPORT); break;
052:           case 2: spell=new Spell(Spell.SUMMON); break;
053:           default: spell=new Spell(Spell.randomSpell()); break;
054:           }
055:         break;            
056:       case 4:
057:         desc=DESC_BLUEWAND; image=289;    
058:         switch (RPG.d(4)) {
059:           case 1: spell=new Spell(Spell.IDENTIFY); break;
060:           case 2: spell=new Spell(Spell.RECHARGE); break;
061:           case 3: spell=new Spell(Spell.REPAIR); break;
062:           default: spell=new Spell(Spell.randomSpell()); break;
063:         }
064:       break;            
065:       }
066:     
067:     // charge the wand
068:     mps=RPG.d(20)+RPG.d(3)*spell.cost;
069:     }
070:     
071:   public void recharge(int power) {
072:     if (RPG.r(10*spell.cost)<(mps-power)) {
073:       Game.message(Text.capitalise(getTheName())+" is destroyed by overcharging!");    
074:       die();
075:     } else {
076:       mps=mps+RPG.d(power);
077:       Game.message("You recharge "+getTheName()); 
078:     }
079:   }
080:     
081:     public int damage(int dam, int damtype) {
082:         if (RPG.d(3,10)>dam) return 0;
083:        ThingOwner p=place;
084:        die();
085:       return dam;
086:     }
087:     
088:   public void die() {
089:     Map map=getMap();
090:     if ((map!=null)&&(spell!=null)) {
091:       spell.castAtLocation(null,map,getMapX(),getMapY());  
092:     }
093:     super.die();
094:   }
095:   
096:   public int getWeight() {return 300;}
097:     
098:     public Spell getSpell() {return spell;}
099:     
100:     public int getUse() {return USE_NORMAL;}
101:     
102:     public void use(Being user) {
103:     boolean ishero=(user==Game.hero);
104:     
105:     if ((spell!=null)&&(mps>=spell.cost)) {
106:       mps=mps-spell.cost;
107:       if (ishero) {
108:         QuestApp.questapp.screen.castSpell(spell); 
109:         flags=flags|ITEM_IDENTIFIED;
110:       }
111:       user.aps-=120;
112:     } else {
113:       if (ishero) Game.message("This wand appears to be useless");    
114:     }
115:     }
116: 
117:   public int getStat(int s) {
118:     switch (s) {
119:       case RPG.ST_ITEMVALUE: {
120:         if (isIdentified()) {
121:           return (spell!=null)?(2000+(200*mps)):50;
122:         } else {
123:           return 1000; 
124:         }
125:       }
126:       default: return super.getStat(s); 
127:     } 
128:   }
129: 
130:   public Description getDescription() {return this;}
131:   
132:   // Description interface 
133:   public String getName(int number, int article)    {
134:     if (isIdentified()) return Describer.getArticleName(((number>1)?"wands":"wand")+((spell==null)?"":(" of "+spell.getName()+" ("+mps/spell.cost+" charges)")),number,article);    
135:     else return desc.getName(number,article);
136:   }
137:   
138:      public String getPronoun(int number, int acase) {
139:        return "it";    
140:      }
141:      
142:   public String getDescriptionText() {
143:     if (!isIdentified()) return desc.getDescriptionText();
144:     else return desc.getDescriptionText()+" ("+Text.capitalise(getName(1,ARTICLE_INDEFINITE))+")";
145:   }
146: }
