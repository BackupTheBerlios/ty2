001: // Represents a stack of similar items that can be treated as a group
002: //
003: // Typically used for small or common items that you don't want to
004: // appear individually in the inventory (e.g. arrows, gold coins)
005: //
006: // Different stackable item classes should extend Stack
007: //
008: // Note that all stackable items must be the same, so they can't have 
009: // different properties or qualities. The canStackWith() method is
010: // responsible for determining if two groups of stackable items
011: // are the same and can therefore be combined into a single stack.
012: 
013: package rl;
014: 
015: public class Stack extends Item {
016:   public int number=1;
017:   protected int type=0;
018:   
019:   public Stack(int n) {
020:     number=n;
021:   }
022:   
023:   public int getNumber() {
024:     return number; 
025:   }
026:   
027:   public void setNumber(int num) {
028:       number=num;
029:       if (number<=0) remove();    
030:   }
031: 
032:   public int getIndividualWeight() {
033:     return 1; 
034:   }
035:   
036:   public int getWeight() {
037:     return number*getIndividualWeight();    
038:   }
039: 
040:   // descriptions etc.
041:   public String getName() {
042:     if (number==1) return super.getName();
043:     else return number + " "+ getDescription().getName(number,Description.ARTICLE_NONE);
044:   }
045:   
046:   public String getAName() { 
047:     if (number==1) return super.getAName();
048:     else return getName();
049:   }
050:   
051:   public String getTheName() { 
052:     if (number==1) return super.getTheName();
053:     else return "the "+getName();
054:   }
055: 
056:   // check if stacking possible
057:     public boolean canStackWith(Thing t) {
058:       if ((t.getClass())!=(getClass())) return false;
059:     Stack l=(Stack)t;
060:     return (l.type==type)&&(l.quality==quality)&&(l.flags==flags);
061:     }
062:     
063:   // split n items from stack
064:   public Stack split(int n) {
065:     if (n>=number) {
066:       remove();
067:       return this;
068:     }
069:     if (n<1) return null;
070:     Stack s=(Stack)this.clone();
071:     s.number=n;    
072:     number=number-n; 
073:     return s; 
074:   }
075:   
076:   public Thing remove(int n) {
077:     if (n>=number) {
078:       remove();
079:       return this;
080:     } else {
081:       return split(n); 
082:     }
083:   }
084:   
085:     // transfers contents to other stack
086:     public boolean stackWith(Thing thing) {
087:       if (!canStackWith(thing)) return false;
088:       ((Stack)thing).number += number;
089:       number=0;
090:       remove();
091:     return true;
092:     }
093:   
094:   // INTERACTION HANDLING
095:   
096:   public int damage(int dam, int damtype) {
097:     int f=getStat(RPG.ST_ITEMFRAGILITY);
098:     if (RPG.d(1000)<=(dam*f)) {
099:       die();
100:     }
101:     return dam; 
102:   }
103:   
104:   // called when player tries specific action on object
105: 
106:   public int getZ() {return Z_ITEM;}
107: 
108: }
