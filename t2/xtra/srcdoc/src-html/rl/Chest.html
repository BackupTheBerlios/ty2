001: //
002: // A chest, possibly trapped, possibly containing something useful
003: //
004: 
005: package rl;
006: 
007: public class Chest extends Scenery {
008:   private int trap;
009:   private boolean trapdetected;
010:   private boolean locked;
011:   private int level;
012:   private Thing contents;
013:     
014:   private static final Description DESC_CHEST =
015:         new Describer ("chest","A solid wooden chest.");
016:   private static final Description DESC_OPENCHEST =
017:         new Describer ("chest","An open chest.");
018:   private static final Description DESC_TRAPPEDCHEST =
019:         new Describer ("chest","A solid chest protected by a devious trap.");
020:     
021:     public Description getDescription() {
022:       if (!locked) return DESC_OPENCHEST;
023:       if ((trap>0)&&trapdetected) return DESC_TRAPPEDCHEST;
024:     return DESC_CHEST;      
025:     }
026:     
027:   // create a chest 
028:   // value determines usefulness of contents....
029:   // ....and also potential danger of traps involved
030:     public Chest(int l) {
031:         super(120+RPG.r(2),true,null);
032:         level=l;
033:     contents=Lib.createItem(level+RPG.d(6));
034:     if (RPG.d(10)<=level) trap=RPG.d(3);
035:     trapdetected=false;
036:     locked=true;
037:     }
038:     
039:     public int getImage() {
040:         return super.getImage()+(locked?0:2);
041:     }
042:     
043:   public void die() {
044:     Map map=getMap();
045:     if (map!=null) getMap().addThing(contents,x,y);
046:     super.die(); 
047:   }
048:   
049:   public int damage(int dam, int damtype) {
050:     if (RPG.d(2,10)<dam) {
051:       Game.message("The chest is destroyed");
052:       triggerTrap();
053:       die();
054:     } else {
055:       triggerTrap();
056:     }
057:     return dam;
058:     }
059:     
060:     public void use(Being user) {
061:     // chest are only for heroes....
062:     Hero h=Game.hero;
063:     if (user!=h) return;
064:     
065:     if (trap==0) {
066:             if (!locked) {
067:                 if (contents!=null){
068:                     Game.message("You find "+contents.getAName());
069:                     h.getMap().addThing(contents,h.x,h.y);
070:                     contents=null;
071:                 } else {
072:                   Game.message("You find nothing in the chest");    
073:                 }
074:             } else {
075:               Game.message("You unlock the chest....");
076:               locked=false;    
077:             }
078:         } else {
079:           if (!trapdetected) {
080:             // have blundered into this one....
081:             Game.message("The chest is trapped!");
082:             if (RPG.test(h.getStat(RPG.ST_CR),2)&&RPG.test(h.getStat(RPG.ST_IN),10)) {
083:                 Game.message("But you manage to disarm it");
084:               trap=0;    
085:             } else {
086:               triggerTrap();    
087:             }
088:           } else {
089:               // try to disarm
090:             if (RPG.test(h.getStat(RPG.ST_CR),2)) {
091:                 Game.message("You manage to disarm the trap");
092:               trap=0;    
093:             } else {
094:               Game.message("You trigger the trap by accident!");
095:               triggerTrap();    
096:             }
097:           }
098:         }
099:     }
100:     
101:   public void triggerTrap () {
102:     Hero h=Game.hero;
103:     Map map=getMap();
104:         int dist=(h.x-x)*(h.x-x)+(h.y-y)*(h.y-y);
105:         switch (trap) {
106:           case 0:
107:               return;    
108:             case 1:
109:         trap=0;
110:         if (dist<=2) {
111:             Game.message("You are hit by a violent explosion!");                
112:         }
113:         Game.mappanel.doExplosion(x,y,0,4,RPG.DT_NORMAL);
114:         trap=0;
115:         break;
116:             case 2:
117:         trap=0;
118:         if (dist<=6) {
119:           Game.message("You are surrounded by poisonous fumes!");                
120:         }
121:         Game.mappanel.doExplosion(x,y,2,2,RPG.DT_POISON);
122:         break;
123:       case 3:
124:         trap=0;
125:         Being c=Lib.createFoe(level+1);
126:         if (isVisible()) Game.message("Ambush!");
127:         die();
128:         map.addThing(c,x,y);
129:         break;
130:        }
131:     }
132:     
133:     public void discover() {
134:       if ((trap==0)||trapdetected) return;
135:       
136:       if (RPG.d(3)==1) {
137:         Game.message("You have discovered a trap!");
138:         trapdetected=true;
139:       }    
140:     }    
141: }
