001: package rl;
002: 
003: public class OldCaves extends Map {
004:   private int level;
005:   
006:   public OldCaves (int t, int l) {
007:     super(65,65);
008:     
009:     level=l;
010:     
011:     floortile=Tile.CAVEFLOOR;
012:     walltile=Tile.CAVEWALL;
013:     
014:     switch (t) {
015:       // top entrance of caves
016:       case 1: {
017:         fillArea(0,0,64,64,0);
018:         setTile(64,32,floortile);
019:         setTile(60,32,floortile);
020:         setTile(56,32,floortile);
021:         buildCaves(4);
022:         replaceTiles(0,walltile);
023:         fractalize(0,0,64,64,4);
024:         break;
025:       }
026: 
027:       default: {
028:         break;
029:       }
030:     }
031:     
032:     {
033:       Point ts=findFreeSquare();
034:       addBeasties(ts.x,ts.y);
035:     }
036:      
037:     entrance=new Portal(0);
038:   } 
039:   
040:   
041:   // random cave construction
042:   // extends non-blank areas randomly
043:   //   g = size of cave units (4 expected)
044:   public void buildCaves(int g) {
045:     for (int i=0; i<5000; i++) {
046:       int x=RPG.d(width/g - 2)*g;
047:       int y=RPG.d(height/g - 2)*g;
048:     
049:       if (getTile(x,y)!=0) continue;
050:       
051:       int dir=1+RPG.r(4)*2;
052:       int dx=g*DX[dir];
053:       int dy=g*DY[dir];
054:       
055:       if (getTile(x+dx,y+dy)!=0) continue;
056:       
057:       int c = getTile(x+2*dx,y+2*dy);
058:       
059:       if (c!=0) {
060:         switch(RPG.d(15)) {
061:           case 1: c=Tile.RIVER; break;
062:           case 2: case 3: c=walltile; break;
063:         }
064:         setTile(x,y,c);
065:         setTile(x+dx,y+dy,c);
066:       }
067:     }
068:   }
069:   
070:   // add outdoor critters
071:   public void addBeasties(int x, int y) {
072:     Creature m=Creature.createCreature(Creature.spawn,getLevel());
073:     for (int i=RPG.d(Creature.numbers[m.creaturetype]); i>0; i--) {
074:       int bx=x-2+RPG.r(5);
075:       int by=y-2+RPG.r(5);
076:       if (isClear(bx,by)) {
077:         addThing(m.cloneType(),bx,by);
078:       }  
079:     }
080:   }
081: 
082:   // can exit if there are no nearby hostiles
083:   public boolean canExit() {
084:     Hero h=Game.hero;
085:     Mobile m=getNearbyMobile(h.x,h.y,10);
086:     if (Game.hero.x!=(width-1)) return false;
087:     if ((m==null)||!(h.isHostile(m))) {
088:       return true;
089:     } else {
090:       Game.message("You cannot leave with enemy near!");
091:       return false; 
092:     }
093:   }
094:   
095:   public int getLevel() {
096:     return level; 
097:   }
098:   
099:   // set the message for this particular map
100:   public String getEnterMessage() {
101:     return "These caves are damp and sinister";  
102:   }
103: 
104: 
105:   // Level name for display on status panel
106:   public String getLevelName() {
107:     return "Caves";  
108:   }
109: }
