003: public class Inventory extends Object implements Active, Cloneable, java.io.Serializable {
004:   private Thing[] inv;
005:   private int invsize;
006:   private int invcount;
007:   private ThingOwner place;
008: 
009:   public Inventory(ThingOwner owner) {
010:     place=owner;    
011:   }
012: 
013:   // deep clone of inventory
014:   public Inventory buildcopy(ThingOwner owner) {
015:     Inventory i;
016:     try {
017:       i=(Inventory)super.clone();
018:     } catch (CloneNotSupportedException e) {throw new Error("Inventory clone failed");} 
019:     if (inv!=null) {
020:       i.inv=(Thing[])i.inv.clone();
021:       for (int t=0; t<i.invcount ; t++) {
022:         i.inv[t]=(Thing)i.inv[t].clone();
023:         i.inv[t].place=owner; 
024:       }
025:     }
026:     i.place=owner;
027:     return i;
028:   }
029: 
030:   public void setUsage(Thing t,int wt) {
031:     if (t.place!=place) return;
032:     t.y=wt;    
033:   }
034:   
035:   // get total weight of inventory items
036:   public int getWeight() {
037:     int result=0;
038:     for (int i=0; i<invcount; i++) {
039:       result=result+inv[i].getWeight();    
040:     }    
041:     return result;
042:   }
043:   
044:   // get all contents
045:   public Thing[] getContents() {
046:     if (invcount==0) return new Thing[0];
047:     Thing[] temp = new Thing[invcount];
048:     int tempcount=0;
049:     for (int i=0; i<invcount; i++) {
050:       Thing t=inv[i];
051:       if (t instanceof Item) {
052:         temp[tempcount]=t;
053:         tempcount++;    
054:       }
055:     }
056:     Thing[] result = new Thing[tempcount];
057:     if (tempcount>0) System.arraycopy(temp,0,result,0,tempcount);
058:     return result;        
059:   }
060: 
061:   public Thing getContents(String s) {
062:     Thing[] stuff=getContents();
063:     for (int i=0; i<stuff.length; i++) {
064:       if ((stuff[i].getName()==s)) return stuff[i];
065:     }
066:     return null;
067:   }
068:   
069:   // get all contents of specified class
070:   public Thing[] getContents(Class c) {
071:     if (invcount==0) return null;
072:     Thing[] temp = new Thing[invcount];
073:     int tempcount=0;
074:     for (int i=0; i<invcount; i++) {
075:       Thing t=inv[i];
076:       if (c.isInstance(t)) {
077:         temp[tempcount]=t;
078:         tempcount++;  
079:       }
080:     }
081:     Thing[] result = new Thing[tempcount];
082:     if (tempcount>0) System.arraycopy(temp,0,result,0,tempcount);
083:     return result;      
084:   }
085: 
086:   public Thing getWeapon(int wt) {
087:     for (int i=0; i<invcount; i++) {
088:       if (inv[i].y==wt) return inv[i];    
089:     }
090:     return null;
091:   }
092:   
093:   public Thing[] getUsableContents(int use) {
094:     if (invcount==0) return null;
095:     Thing[] temp = new Thing[invcount];
096:     int tempcount=0;
097:     for (int i=0; i<invcount; i++) {
098:       Thing t=inv[i];
099:       if (t instanceof Item) {
100:         if ((t.getUse()&use)>0) {
101:           temp[tempcount]=t;
102:           tempcount++;    
103:         }    
104:       }
105:     }
106:     Thing[] result = new Thing[tempcount];
107:     if (tempcount>0) System.arraycopy(temp,0,result,0,tempcount);
108:     return result;        
109:   }
110:   
111:   // get all contents with a given stat value
112:   public Thing[] getStatContents(int stat, int val) {
113:     if (invcount==0) return null;
114:     Thing[] temp = new Thing[invcount];
115:     int tempcount=0;
116:     for (int i=0; i<invcount; i++) {
117:       Thing t=inv[i];
118:       if (t.getStat(stat)==val) {
119:         temp[tempcount]=t;
120:         tempcount++;  
121:       }
122:     }
123:     Thing[] result = new Thing[tempcount];
124:     if (tempcount>0) System.arraycopy(temp,0,result,0,tempcount);
125:     return result;      
126:   }
127: 
128:   // get all contents that can be worn/wielded with 'w'
129:   public Item[] getWieldableContents() {
130:     if (invcount==0) return null;
131:     Item[] temp = new Item[invcount];
132:     int tempcount=0;
133:     for (int i=0; i<invcount; i++) {
134:       Thing t=inv[i];
135:       if (t instanceof Item) {
136:         if (((Item)t).wieldType()>RPG.WT_NONE) {
137:           temp[tempcount]=(Item)t;
138:           tempcount++;    
139:         }    
140:       }
141:     }
142:     Item[] result = new Item[tempcount];
143:     if (tempcount>0) System.arraycopy(temp,0,result,0,tempcount);
144:     return result;        
145:   }
146:   
147:   // get the net modifier for all used inventory items
148:   public int getModifier(int stat) {
149:     int result=0;
150:     for (int i=0; i<invcount; i++) {
151:         Thing t=inv[i];
152:         if (t.y!=RPG.WT_NONE) {
153:           result=result+t.getModifier(stat);
154:         }         
155:     }    
156:     return result;
157:   }
158:   
159:   // time effect - pass to active sub-items
160:   public void action(int time) {
161:     for (int i=0; i<invcount;i++)    {
162:         Thing t=inv[i];
163:         if (t instanceof Active) ((Active)t).action(time);
164:     }
165:   }
166: 
167:   public Thing getThing(int i) {
168:     return ((i>=0)&(i<invcount))?inv[i]:null;
169:   }
170: 
171:   public int getUsage(int i) {
172:     return ((i>=0)&(i<invcount))?inv[i].y:RPG.WT_ERROR;
173:   }
174:   
175:   public void ensureSize (int s) {
176:     if (s<=invsize) return;
177:     invsize=s+5;
178:     Thing[] newinv=new Thing[invsize];
179:     if (inv!=null) {
180:       System.arraycopy(inv,0,newinv,0,invcount);
181:     }    
182:     inv=newinv;
183:   } 
184:     
185:   public void removeThing(Thing thing) {
186:     int pos=thing.x;
187:       if (inv==null) throw new Error("Empty Inventory bug!");
188:       if (thing.place!=place) throw new Error("Thing in wrong place!");
189:       if (inv[pos]!=thing) throw new Error("Thing in wrong position!");
190:     if (pos<(invcount-1)) System.arraycopy(inv,pos+1,inv,pos,invcount-pos-1);
191:     inv[invcount-1]=null;
192:     thing.place=null;
193:     thing.next=null;
194:     thing.y=RPG.WT_NONE;   // definitely not using anymore
195:     invcount=invcount-1;
196:     for (int i=0; i<invcount; i++) inv[i].x=i; // new positions
197:   }
198: 
199:   public void swapItems(int a, int b) {
200:     if ((a>=invcount)||(b>=invcount)||(a<0)||(b<0))    return;
201:     Thing ti=inv[a];
202:     inv[a]=inv[b];
203:     inv[b]=ti;
204:       inv[a].x=a;
205:       inv[b].x=b;
206:   }
207: 
208:     public void addThing(Thing thing) {
209:       if (thing==null) return;
210:       thing.remove();
211:     
212:     // see if we can stack with existing items
213:     if (thing instanceof Stack) {
214:       Stack sthing=(Stack)thing;
215:       for (int i=0; i<invcount; i++) {
216:         if (sthing.stackWith(inv[i])) return;    
217:       }    
218:     }
219:     
220:     ensureSize(invcount+1);
221:     inv[invcount]=thing;
222:     thing.place=place;
223:     thing.next=null;   // don't care about list 
224:     thing.x=invcount;  // might be useful to know slot!
225:     thing.y=RPG.WT_NONE; // item not in use
226:     invcount=invcount+1;
227:   }
228: 
229:   public boolean clearUsage(int wt) {
230:       for (int i=0; i<invcount; i++) {
231:           Thing t=inv[i];
232:       int it = t.y;
233:         if ((it==wt)&&(t instanceof Item)) {
234:             if ((place!=Game.hero)||((((Item)t).flags&Item.ITEM_CURSED)==0)) {
235:           inv[i].y=RPG.WT_NONE;
236:         } else {
237:           Game.message("You are unable to remove your "+t.getName()+"!"); 
238:           return false;
239:         }
240:       }
241:       }
242:     return true;
243:   }
