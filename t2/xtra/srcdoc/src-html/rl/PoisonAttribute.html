01: package rl;
02: 
03: public class PoisonAttribute extends ActiveAttribute {
04:     protected int damage;
05:     protected int probability;
06:     protected int duration; 
07:     protected int damtype;
08:     
09:     public PoisonAttribute(int d, int dt, int dur, int prob) {
10:         damage=d;
11:         damtype=dt;
12:         duration=dur;
13:         probability=prob;
14:     }
15:     
16:     public void action(int time) {
17:     if (duration<=0) {
18:       remove();  
19:       return;
20:     }
21:     
22:     duration=duration-time;
23: 
24:     for (int i=RPG.po(probability*time,10000); i>0; i--) {
25:       Mobile m=getMobile();
26:       int dam=m.damage(RPG.a(damage),damtype);
27:       if (m==Game.hero) {
28:         if (dam>0) Game.message("You feel poisoned");    
29:       }
30:     }
31:   }
32:     
33:     public String getAttributeString() {return "Poisoned";}
34:     
35: }
