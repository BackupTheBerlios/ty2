013: public class Potion extends Stack implements Description {
014:   protected int desc;

018:   private static String[] potioncolours={"red","burgundy","orange","yellow","green","blue","aquamarine","cyan","sky blue","brown","indigo","violet","purple","pink","grey","white","black"};
019:   private static int[]    images =      {245,  245,       240,     240,     247,    246,   248,         248,   248,       243,    244,     244,     244,     244,   243,   242,    241};

023:   public static final int HEALING=1;
024:   public static final int GAINSTAT=2;
025:   public static final int WATER=3;
026:   public static final int POWER=4;
027:   public static final int FRENZY=5;
028:   public static final int POISON=6;
029:   public static final int CUREPOISON=7;
030:   public static final int LOSESTAT=8;
031:   public static final int EXPERIENCE=9;

036:   public static final int AMNESIA=14;

039:   private static final String[]           effects = {"null",  "healing", "advancement", "water",  "power", "berserker fury", "poison", "cure poison", "weakening", "experience", "hunger", "speed", "fate", "clumsiness", "amnesia"};
040:   public static final  int[] potions =              {0,       HEALING,   GAINSTAT,      WATER,    POWER,   FRENZY,           POISON,   CUREPOISON   , LOSESTAT,    EXPERIENCE,   HUNGER,   SPEED,   FATE,   CLUMSINESS,   AMNESIA};

042:   public int getUse() {return Thing.USE_NORMAL|Thing.USE_DRINK;}
043:     
044:   public static Stack createPotion() {
045:     return new Potion(potions[RPG.d(potions.length-1)]); 
046:   }
047:   
048:   public int getStat(int s) {
049:     switch (s) {
050:       case RPG.ST_ITEMVALUE: return 2000;
051:       default: return super.getStat(s); 
052:     } 
053:   }
054:   
055:   public Potion(int t, int num) {
056:     this(t);
057:     number=num; 
058:   }
059:   
060:   public Potion(int t) {
061:     super(1);
062:     if (t==0) t=potions[RPG.r(potions.length)];
063:     
064:     type=t;
065:   }
066:     
067:   public int getIndividualWeight() {
068:     return 400; 
069:   }
070:   
071:   public void use(Being user) {
072:       use(user, USE_DRINK);    
073:     }
074:     
075:     public void use(Being user, int usetype) {
076:     Hero h=Game.hero;
077:     boolean ishero=(user==Game.hero);
078:     
079:     if (usetype!=Thing.USE_DRINK) return;
080:     
081:     switch (type) {
082:       case POWER: {
083:         user.mps=RPG.min(user.getStat(RPG.ST_MPSMAX)+1,user.mps+RPG.d(2,5));
084:         Game.message("You feel energized");
085:         break;
086:       }      
087: 
088:       case GAINSTAT: {
089:         user.stats.incStat(RPG.d(8),1);
090:         Game.message("You feel more capable");
091:         break;
092:       }
093:     
094:       case WATER: {
095:         Game.message("You feel refreshed");
096:         break;
097:       }
098:     
099:       case POISON: {
100:         int dam=user.damage(RPG.a(3),RPG.DT_POISON);
101:         if (dam>0) {
102:           Game.message("You feel really sick");
103:           user.addThing(new PoisonAttribute(3,RPG.DT_POISON,RPG.d(3,6)*250,10));
104:         } else Game.message("This potion tastes foul");
105:         break;
106:       }
107: 
108:       case HEALING: {
109:         int max=user.getStat(RPG.ST_HPSMAX);
110:         if (user.hps<max) {
111:           user.hps=RPG.min(max,user.hps+RPG.d(2,5));
112:           if (ishero) Game.message("You feal healthier");
113:         } else {
114:           if (ishero) Game.message("You feel a warm glow inside");
115:         }
116:         break;
117:       }
118: 
119:       case FRENZY: {
120:         if (ishero) Game.message("Your blood is filled with an unstoppable rage!"); 
121:         user.addAttribute(new TemporaryAttribute("frenzied",RPG.d(5,1000),new int[] {RPG.ST_ATTACKSPEED,RPG.d(2,100)},"You start to calm down"));
122:         break;
123:       }
124:     
125:       case CUREPOISON: {
126:         if (ishero) Game.message("You feel cleansed."); 
127:         Thing[] poisons=user.inv.getContents(PoisonAttribute.class);
128:         for (int i=0; i<poisons.length; i++) {
129:           poisons[i].remove(); 
130:         }
131:         break;
132:       }
133: 
134:       case LOSESTAT: {
135:         user.stats.incStat(RPG.d(8),-1);
136:         Game.message("You feel less capable");
137:         break;
138:       }
139: 
140:       case EXPERIENCE: {
141:         if (ishero) {
142:           h.gainExperience(RPG.d(2,100));
143:           Game.message("You feel a sudden surge of knowledge");
144:         }
145:         break;
146:       }
147: 
148:       case HUNGER: {
149:         if (ishero) Game.message("You suddenly feel very hungry"); 
150:         user.setStat(RPG.ST_HUNGER,200000);
151:         break; 
152:       }
153:       
154:       case SPEED: {
155:         if (ishero) Game.message("You suddenly feel the urge to run very fast"); 
156:         user.addAttribute(new TemporaryAttribute("Accelerated",RPG.d(5,1000),new int[] {RPG.ST_MOVESPEED,RPG.d(2,100)},"You start to slow down"));
157:         break;
158:       }
159: 
160:       case FATE: {
161:         if (ishero) Game.message("You feel invincible"); 
162:         user.stats.incStat(RPG.ST_FATE,1);
163:         break;
164:       }
165: 
166:       case CLUMSINESS: {
167:         if (ishero) Game.message("You seem less able to control your limbs"); 
168:         user.addAttribute(new TemporaryAttribute("Clumsy",RPG.d(10,2000),new int[] {RPG.ST_SK,-RPG.d(6),RPG.ST_AG,-RPG.d(6)},"You stop feeling clumsy"));
169:         break;
170:       }
171: 
172:       case AMNESIA: {
173:         if (ishero) {
174:           Art[] artlist=h.arts.getArts();
175:           if (artlist.length>0) {
176:             Art a=artlist[RPG.r(artlist.length)];
177:             Game.message("You forget about "+a.getName()); 
178:             // todo: delete art here
179:           } else {
180:             Game.message("Doh. You feel very stupid"); 
181:           }
182:         }
183:         break;
184:       }
185:     }
186: 
187:     identify(1000);
188:     remove(1);
189:   }

195:   private String getPotionString() {
196:     return 
197:       potionstyles[(Game.seed()+type*3)%potionstyles.length]
198:      +" " 
199:      +potioncolours[(Game.seed()+type)%potioncolours.length];
200:   }
201:   
202:   public int getImage() {
203:     return images[(Game.seed()+type)%potioncolours.length];
204:   }
205:   
206:   // description stuff
207:   public String getSingularName()    {
208:        if (isIdentified()) {
209:        return "potion of "+effects[type];    
210:     } else {
211:        return getPotionString()+" potion";
212:     }
213:   }
214:   
215:   // description stuff
216:   public String getPluralName()  {
217:     if (isIdentified()) {
218:        return "potions of "+effects[type];  
219:     } else {
220:        return getPotionString()+" potions";
221:     }
222:   }
