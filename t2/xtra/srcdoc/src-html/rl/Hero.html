698:     public void kick(int dx, int dy) {
699:         aps=aps-50000/getStat(RPG.ST_ATTACKSPEED);
700:         Map m=(Map)place;
701:         Thing t=m.getObjects(x+dx,y+dy);
702:         while (t!=null) {
703:           if (t.isBlocking()) {
704:         kick(t,dx,dy);
705:             return;    
706:           }    
707:           t=t.next;
708:         }
709:         if (m.isBlocked(x+dx,y+dy)) {
710:           Game.message("You kick the wall - Ouch!");
711:           return;    
712:         }
713:         t=m.getObjects(x+dx,y+dy);
714:         
715:     while (t!=null) {
716:          if ((t instanceof Item)||(t instanceof Scenery)) {
717:            if (!(t instanceof Decoration)) {
718:           kick(t,dx,dy);
719:           return;
720:            }
721:          }
722:          t=t.next;
723:         }    
724:         Game.message("You kick thin air");
725:     }
