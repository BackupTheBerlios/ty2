050:   public int notify(Mobile m, int eventtype, int ext, Object o) {
051:          int state=m.getStat(RPG.ST_AIMODE);
052:     switch (eventtype) {
053:         case EVENT_ALARM: case EVENT_VISIBLE:
054:              reTarget(m,Game.hero.x,Game.hero.y);
055:            if (state!=1) {
056:                 m.getMap().areaNotify(m.x,m.y,5,AI.EVENT_ALARM,0,null);
057:             }
058:             return 1;
059:             
060:       case EVENT_ATTACKED: {
061:         if (o==Game.hero) m.setStat(RPG.ST_AIMODE,AI.STATE_HOSTILE); 
062:         return 1;
063:       }
064:       
065:       default:
066:           return super.notify(m,eventtype, ext, o);    
067:     }
068:   }

070:   public void reTarget(Mobile m, int tx, int ty) {
071:          m.setStat(RPG.ST_TARGETX,tx);
072:         m.setStat(RPG.ST_TARGETY,ty);
073:        m.setStat(RPG.ST_AIMODE,1);
074:   }
075:   
076:   // creature attacks towards target tx,ty
077:   public void doAttack(Mobile m, int tx, int ty) {
078:     Map map=m.getMap();
079:     if (map==null) return;
080:     boolean visible=m.isVisible();
081: 
082:     while (m.aps>=0) {
083:       if (RPG.d(100)<=m.getStat(RPG.ST_CASTCHANCE)) {
084:         if (CasterAI.instance.doAction(m)) continue; 
085:       }
086:           
087:       if (visible&&(RPG.d(3)==1)&&(m.getWielded(RPG.WT_MISSILE)!=null)) {
088:         ShooterAI.instance.doAction(m);
089:         continue; 
090:       }
091:           
092:       int dx=RPG.sign(tx-m.x);
093:       int dy=RPG.sign(ty-m.y);
094:           
095:       // run away?
096:       if ((m.getStat(RPG.ST_IN)>=6)&&((m.getStat(RPG.ST_HPS)*3)<=m.getStat(RPG.ST_HPSMAX))) {
097:         if (tryMove(m,m.x-dx,m.y-dy)) continue; 
098:         if (tryMove(m,m.x-RPG.sign(dx-dy),m.y-RPG.sign(dy+dx))) continue; 
099:         if (tryMove(m,m.x-RPG.sign(dx+dy),m.y-RPG.sign(dy-dx))) continue; 
100:       }
101:            
102:       if (tryMove(m,m.x+dx,m.y+dy)) continue; 
103:       if (tryMove(m,m.x+RPG.sign(dx-dy),m.y+RPG.sign(dy+dx))) continue; 
104:       if (tryMove(m,m.x+RPG.sign(dx+dy),m.y+RPG.sign(dy-dx))) continue; 
105:            
106:       if (RPG.d(2)==1) dx=RPG.r(3)-1;
107:       if (RPG.d(2)==1) dy=RPG.r(3)-1;
108:       if (!tryMove(m,m.x+dx,m.y+dy)) {
109:         m.aps-=100;
110:         m.xdirection=0;
111:         m.ydirection=0;
112:       }
113:     }
114:   }

116:   public boolean doAction(Mobile m) {
117:       int state=m.getStat(RPG.ST_AIMODE);
118:       Hero h=Game.hero;
119:       Map map=m.getMap();
120:       if (map==null) return true;
121:       boolean visible=m.isVisible();
122:       
123:       switch (state) {
124:         case 0:
125:             if (visible) {
126:               reTarget(m,Game.hero.x,Game.hero.y);
127:                 map.areaNotify(m.x,m.y,6,AI.EVENT_ALARM,0,null);
128:             } else {
129:             // wander randomly
130:             while (m.aps>=0) {
131:               int dx=RPG.r(3)-1;
132:               int dy=RPG.r(3)-1;

135:             if (!tryMove(m,nx,ny)) m.aps-=100;;                
136:           }
137:
