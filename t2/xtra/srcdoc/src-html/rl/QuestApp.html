059:   // image filter object to create greyed tiles
060:   class GreyFilter extends RGBImageFilter {
061:     public GreyFilter() {
062:       canFilterIndexColorModel = true;
063:     }
064: 
065:     public int filterRGB(int x, int y, int rgb) {
066:       return (rgb & 0xff000000)
067:         | (0x10101 * (   ((rgb & 0xff0000) >> 18)
068:                        + ((rgb & 0xff00) >> 10)
069:                        + ((rgb & 0xff) >> 2)      ));
070:     }
071:   }

253:   // calculates the Hero's final score
254:   public int calcScore(Hero h) {
255:     int score=0;
256:     score=h.getStat(RPG.ST_SCORE);
257:     score+=2*h.getStat(RPG.ST_LEVEL)*h.getStat(RPG.ST_LEVEL); 
258:     score+=5*h.getStat(RPG.ST_SCORE_BESTLEVEL)*h.getStat(RPG.ST_SCORE_BESTLEVEL);
259:     score+=3*h.getStat(RPG.ST_SCORE_BESTKILL)*h.getStat(RPG.ST_SCORE_BESTKILL);
260:     score+=h.getStat(RPG.ST_SCORE_KILLS)/100;
261:     return score;
262:   }

373:   // loads an image from wherever possible
374:   // ideally, from the .jar resource bundle
375:   // not sure if all of this is necessary
376:   // but try to cover all possible environments
377:   public Image getImage(String filename) {
378:     Toolkit toolkit = Toolkit.getDefaultToolkit();
379: 
380:     Image image=null;
381: 
382:     try {
383:       URL imageURL=getClass().getResource(filename);
384:       if (imageURL!=null) {
385:         image = toolkit.getImage(imageURL);
386:       }
387:     } catch (Exception e) {
388:       System.out.println(e);
389:     }
390: 
391:     if (image==null) {
392:       image=toolkit.getImage(filename);
393:       if (image==null) {
394:         image=getImage(getCodeBase(),filename);
395:       }
396:     }
397: 
398:     return image;
399:   }
