001: // Object representing the Tyrant World Map
002: //
003: // Contains code for handling:
004: //   long-distance travel
005: //   wilderness encounters
006: //   creating outdoor map locations

009: package rl;
010: 
011: public class WorldMap extends Map {
012:   
013:   protected int[] terrains = {Tile.PLAINS, Tile.PLAINS,Tile.FORESTS,Tile.HILLS,Tile.MOUNTAINS,Tile.SEA,Tile.SWAMP,Tile.SEA};
014:   
015:   public WorldMap(int w, int h) {
016:     super(w,h);
017:     
018:     
019:     for (int x=0; x<w; x+=4) {
020:       setTile(x,0,Tile.MOUNTAINS); 
021:       setTile(x,h-1,Tile.MOUNTAINS); 
022:     }
023:     for (int y=0; y<h; y+=4) {
024:       setTile(0,y,Tile.MOUNTAINS); 
025:       setTile(w-1,y,Tile.SEA); 
026:     }
027: 
028:     for (int y=4; y<h-4; y+=4) for (int x=4; x<w-4; x+=4){
029:       setTile(x,y,terrains[RPG.r(terrains.length)]);
030: 
031:     }
032:     
033:     setTile(16,8,Tile.FORESTS);
034:     setTile(16,12,Tile.FORESTS);
035:     setTile(16,16,Tile.PLAINS);
036:     setTile(12,12,Tile.FORESTS);
037:     setTile(8,12,Tile.HILLS);
038:     
039:     fractalizeBlock(0,0,w-1,h-1,4); 
040:     
041:     // starting town + quest
042:     addThing(new PlacePortal(PlacePortal.TOWN),16,16);
043:     addThing(new PlacePortal(PlacePortal.RUIN),16,8);
044: 
045:     // some other towns
046:     addThing(new PlacePortal(PlacePortal.TOWN),0,0,width-1,height-1);
047:     addThing(new PlacePortal(PlacePortal.TOWN),0,0,width-1,height-1);
048: 
049:     addThing(new PlacePortal(PlacePortal.GOBLINTRIBE),0,0,width-1,height-1);
050:   }
051:   
052:   
053:   
054:   // can we explore this square?
055:   public boolean canExit() {
056:     Game.message("There is nothing of interest here");
057:     return false;
058:   }
059:   
060:   // enter a world map square
061:   //  - if there is a portal to a special location then use it
062:   //  - else create the appropraite outdoor area map
063:   //
064:   public void exitMap(int tx, int ty) {
065:     Hero h=Game.hero;
066:     Portal p=(Portal)getObject(tx,ty,Portal.class);
067:     
068:     if (p!=null) {
069:       p.travel(h);
070:     } else {
071:       // put hero into a local area map
072:       Map m=createArea(tx,ty);
073:       p=new Portal(0);
074:       p.setDestination(m.entrance);
075:       // could build a portal here so we can return to exact location map
076:       // Don't think we really want this though
077:       //   addThing(p,tx,ty);
078:       //   m.entrance.setDestination(p);
079:       p.travel(h); 
080:     }
081:   }
082: 
083:   // move costs are much higher for outdoor distances
084:   public int getMoveCost(Thing mover,int x, int y) {
085:     return 10*super.getMoveCost(mover,x,y); 
086:   }
087: 
088:   // set the message for this particular map
089:   public String getEnterMessage() {
090:     return "This fertile valley is known as North Karrain";  
091:   }
092: 
093:   // Level name for display on status panel
094:   public String getLevelName() {
095:     return "The Great Outdoors";  
096:   }
097:   
098:   // override hitsqure for random encounters
099:   public void hitSquare(Thing m,int x, int y,boolean touchfloor) {
100:     if (m==Game.hero) {
101:       Thing p=getObject(x,y,Portal.class);
102:       if (p==null) encounter(x,y);
103:     }
104:     super.hitSquare(m,x,y,touchfloor);
105:   }
106:   
107:   // called to check for random encounters
108:   public void encounter(int x, int y) {
109:     if (RPG.d(10)>1) return;
110:     
111:     Hero h=Game.hero;
112:     Game.mappanel.viewPosition(x,y);
113: 
114:     switch (RPG.d(4)) {
115:       case 1: {
116:         Game.message("You see somebody in the distance"); 
117:         break;
118:       }
119:       
120:       case 2: {
121:         Creature c=Lib.createFoe(Game.hero.getStat(RPG.ST_LEVEL)-1);
122:         Game.message("You see "+c.getAName()+" waiting to ambush travellers"); 
123:         Game.message("Do you want to attack? (y/n)");
124: 
125:         if (Game.getOption("yn")=='y') {
126:           Game.message("");
127:           
128:           // create location map 
129:           Map m=createArea(x,y);
130:           
131:           m.addThing(Game.hero,32,32);
132:           m.addThing(c,32,25);
133:           
134:           // companions
135:           if (RPG.d(2)==1) {
136:             for (int i=RPG.d(6); i>0; i--) {
137:               int gx=28+RPG.r(9);
138:               int gy=18+RPG.r(7);
139:               if (!m.isBlocked(gx,gy)) {
140:                 m.addThing(c.cloneType(),gx,gy); 
141:               } 
142:             }
143:             if ((RPG.d(2)==1)&&(!m.isBlocked(32,20))) {
144:               m.addThing(Lib.createFoe(Game.hero.getStat(RPG.ST_LEVEL)+1),32,20); 
145:             }
146:           }
147:         } else {
148:           Game.message("You avoid the encounter"); 
149:         }
150:         break;
151:       }
152:       
153:       case 3: {
154:         Creature c=Lib.createFoe(Game.hero.getStat(RPG.ST_LEVEL)-1);
155:         Game.message("You see some villagers being attacked by "+c.getDescription().getName(2,Description.ARTICLE_INDEFINITE)); 
156:         Game.message("Do you want to aid them? (y/n)");
157: 
158:         if (Game.getOption("yn")=='y') {
159:           // create location map 
160:           Map m=createArea(x,y);
161: 
162:           m.addThing(Game.hero,32,32);
163:           
164:           // villages
165:           for (int i=RPG.d(2,3); i>0; i--) {
166:             int gx=29+RPG.r(7);
167:             int gy=25+RPG.r(7);
168:             if (!m.isBlocked(gx,gy)) switch(RPG.d(3)) {
169:               case 1: m.addThing(new Person(Person.TOWNIE),gx,gy); break;
170:               case 2: m.addThing(new Person(Person.GIRL),gx,gy); break;
171:               case 3: m.addThing(new Person(Person.WOMAN),gx,gy); break;
172:             } 
173:           }
174:           
175:           // critters
176:           for (int i=RPG.d(2,6); i>0; i--) {
177:             int gx=28+RPG.r(9);
178:             int gy=18+RPG.r(7);
179:             if (!m.isBlocked(gx,gy)) {
180:               m.addThing(c.cloneType(),gx,gy); 
181:             } 
182:           }
183:           // boss critter
184:           m.addThing(Lib.createFoe(Game.hero.getStat(RPG.ST_LEVEL)+1),32,20); 
185:         } else {
186:           Game.message("You leave them to their fate"); 
187:         }
188:         break;
189:       }
190: 
191:       case 4: {
192:         Game.message("You are slowed by bad weather conditions."); 
193:         h.aps-=1000;
194:         break;
195:       }
196: 
197:     }
198:   }

200:   // create Outdoor Area for specified square
201:   // can drop hero strainght into this
202:   public Map createArea(int tx, int ty) {
203:     Outdoors m=new Outdoors(getTile(tx,ty));
204:     
205:     // link back to worldmap
206:     m.entrance.setDestination(new TargetPortal(this,tx,ty));
207:     
208:     return m;
209:   }

212:     super.action(time); 
213:   }
