01: package rl;
02: 
03: public class TypeStack extends Stack {
04:   public ItemType type;
05:   
06:   public TypeStack(ItemType t, int n) {
07:     super(n);
08:     type=t; 
09:   }
10:   
11:   public int getWeight() {
12:     return type.getWeight(this)*number;    
13:   }
14: 
15:   public int getStat(int s) {return type.getStat(this,s);}
16:   
17:   public void throwAt(Being thrower, Map m, int tx, int ty) {
18:     type.throwAt(this,thrower,m,tx,ty);
19:   }
20: 
21:   // descriptions etc.
22:   public String getName() {
23:     if (number==1) return super.getName();
24:     else return number + " "+ getDescription().getName(number,Description.ARTICLE_NONE);
25:   }
26:   
27:   public String getAName() { 
28:     if (number==1) return super.getAName();
29:     else return getName();
30:   }
31:   
32:   public String getTheName() { 
33:     if (number==1) return super.getTheName();
34:     else return "the "+getName();
35:   }
36: 
37:   public Description getDescription() {
38:     return type.getDescription(this); 
39:   }
40: 
41:   // check if stacking possible
42:     public boolean canStackWith(Thing t) {
43:       if (!(t instanceof TypeStack)) return false;
44:     TypeStack l=(TypeStack)t;
45:     return (l.type==type)&&super.canStackWith(t);
46:     }
47:     
48:   // INTERACTION HANDLING
49:   
50:   // try to identify item
51:   public boolean identify(int idskill) {
52:     return type.identify(this,idskill);
53:   }
54: 
55:   public boolean isIdentified() {
56:     return type.isIdentified(this); 
57:   }
58:   
59:   public int damage(int dam, int damtype) {
60:     int f=getStat(RPG.ST_ITEMFRAGILITY);
61:     if (RPG.d(1000)<=(dam*f)) {
62:       remove();
63:     }
64:     return dam; 
65:   }
66:   
67:   public int getUse() {return type.getUse(this);}
68: 
69:   public void use(Being user) {type.use(this,user);}
70: 
71:   // called when player tries specific action on object
72:   public void use(Being user, int usetype) {type.use(this,user,usetype);}
73: 
74:   public int getZ() {return Z_ITEM;}
75: 
76:   public int getImage() {return type.getImage(this);}
77:   
78:   public int wieldType() {return type.wieldType(this);}
79:   
80: }
