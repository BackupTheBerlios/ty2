01: // This is the root object for all Quests that can be assigned to the hero
02: //
03: // Use as follows:
04: //   Game.hero.addThing(new Quest(Quest.SOMEQUEST));
05: 
06: package rl;
07: 
08: 
09: public class Quest extends Thing implements Active {
10:   public static final int ASSIGNED=0;
11:   public static final int COMPLETE=1;
12:   public static final int FAILED=2;
13:   
14:   // the quest target (item, creature, person etc.)
15:   protected Thing target;
16:   
17:   
18:   public Quest() {
19:   }
20: 
21:   protected int state=ASSIGNED;
22:   
23:   public int getState() {
24:     return state; 
25:   }
26:   
27:   public Thing getTarget() {return target;}
28:   
29:   public void setTarget(Thing t) {target=t;}
30:   
31:   public void setState(int a) {
32:     state=a; 
33:     switch(state) {
34:       case COMPLETE: complete(); break;
35:       case FAILED: failed(); break; 
36:     }
37:   }
38: 
39:   public String questName() {
40:     return "null";
41:   }
42:   
43:   // return the written description of the quest
44:   // Can be two or three lines long as required to give full detail
45:   public String questString() {
46:     return "You must cut down the tallest tree in the forest with.... a herring!";
47:   }
48:   
49:   // called to initialise quest, make map modifications etc.
50:   public void setup() {}
51:   
52:   // called when quest is complete
53:   public void complete() {}
54:   
55:   // called if quest is failed
56:   public void failed() {}
57: 
58:   public void action (int t) {}
59: }
