01: package rl;
02: 
03: public class RuneTrap extends Trap {
04:   protected int image;
05:   protected int spell;
06:   protected Thing actor;
07:   
08:   private static final Description DESC_RUNETRAP = 
09:         new Describer("runetrap","A magical runetrap carefully inscribed on the floor.");
10:     
11:   public RuneTrap() {
12:     // runetraps are clearly visible by default
13:     discovered=true;
14:     
15:     actor=null;
16:     image=160+RPG.r(3);
17:     spell=Spell.FIREBALL;
18:     }
19:     
20:   public RuneTrap(Thing owner, int thespell) {
21:     this();
22:     actor=owner;
23:     spell=thespell; 
24:   }
25:   
26:   // target adjacent hero directly if spell is not a ball spell
27:   public boolean actAdjacten() {
28:     return Spell.radii[spell]==0; 
29:   }
30:   
31:     public int getImage() {return image;}
32:     
33:     public void affectSquare(Map map, int tx, int ty) {
34:     remove();
35:     Thing temp=Game.actor;
36:     
37:     if (map!=null) {
38:       Spell sp=new Spell(spell);
39:       sp.castAtLocation(null,map,tx,ty);
40:       if (actor!=null) {
41:         ((Being)actor).mps-=sp.cost;
42:         if (actor==Game.hero) Game.message("Your power is drained by the runes"); 
43:       }
44:     }
45:     
46:     Game.actor=temp;
47:   }
48: 
49:   public Description getDescription() {return DESC_RUNETRAP;}
50: }
