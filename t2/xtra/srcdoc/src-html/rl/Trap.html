003: public class Trap extends Special {
004:   protected boolean discovered=false;
005:   protected boolean activated=true;
006:   protected int uses=1;
007:   protected int level=1;
008: 
009:   public static Trap createTrap(int l) {
010:     switch (RPG.d(3)) {
011:       case 1: {
012:         return new RuneTrap(null,Spell.randomOffensiveSpell(l)); 
013:       }
014:       default: return new HeroTrap(l);
015:     }
016:   }
017:   
018:   
019:   public int getLevel() {
020:     return level; 
021:   }
022:   
023:   // trap is invisible until discovered
024:   public boolean isInvisible() {return !discovered;}
025:   
026:   public boolean isDiscovered() {return discovered;}
027:   
028:   // returns true if t is "wary" of this trap
029:   public boolean isWarning(Thing t) {
030:     return ((!discovered)&&(t instanceof Creature))||(t.getStat(RPG.ST_IN)>5); 
031:   }
032:   
033:   public boolean tryDisarm(Thing b) {
034:     return RPG.test(b.getStat(RPG.ST_CR)*b.getStat(RPG.ST_DISARM),15*getLevel());
035:   }
036:   
037:   // trap is right on the floor
038:   public int getZ() {return Z_FLOOR;}
039: 
040:   // trigger, activated by Thing t
041:   public void trigger(Thing t) {
042:     if (t!=null) {
043:       Map map=t.getMap();
044:       if (map!=null) {
045:         int tx=t.getMapX();
046:         int ty=t.getMapY();
047:         trigger(map,tx,ty);
048:         return;  
049:       } 
050:     }
051:     
052:     Map map=getMap();
053:     if (map!=null) {
054:       int tx=getMapX();
055:       int ty=t.getMapY();
056:       trigger(map,tx,ty);
057:       return;
058:     }
059:     
060:     throw new Error("Trap triggered off-map");
061:   }
062:   
063:   public void setActivated(boolean b) {
064:     activated=b; 
065:   }
066:   
067:   public boolean isActivated() {
068:     return activated; 
069:   }
070:   
071:   // return true if the trap should target an adjacent actor
072:   // that triggers the trap
073:   // trap types that should probably return true are:
074:   //    pit traps
075:   //    dart traps
076:   //    single-square spell traps (but not ball spells)
077:   public boolean actAdjacent() {
078:     return false; 
079:   }
080:   
081:   // trigger in specific location
082:   public void trigger(Map map, int tx, int ty) {
083:     if (activated) {
084:       // turn off trap to prevent self-triggering
085:       setActivated(false);
086:       affectSquare(map,tx,ty);
087:       
088:       // turn trap back on
089:       setActivated(true);
090:       
091:       // deactivate if uses are exhausted
092:       if (uses>0) {
093:         uses=uses-1;
094:         if (uses==0) {
095:           setActivated(false); 
096:           remove();
097:         }
098:       } 
099:     }
100:   }
101:   
102:   // actually do the trap affect
103:   public void affectSquare(Map map, int tx, int ty) {
104:     
105:   }
106:   
107:   // make the trap appear
108:   public boolean appear() {
109:     if (discovered) return false;
110:     discovered=true;
111:     return true;
112:   }
113: }
