01: package rl;
02: 
03: import java.applet.*;
04: import java.awt.*;
05: import java.awt.event.*;
06: 
07: public class ListScreen extends Screen {
08:     String title="List";
09:     String[] strings=null;
10:     static int page=0;
11:     Object result;
12:   
13:     private static int pagesize=12;
14:     private static int lineheight=20;
15:     
16:   public Object getObject() {
17:     boolean end=false;
18:     while (!end) {
19:       KeyEvent k=Game.getInput();
20:       if (k!=null) {
21:         if (k.getKeyCode()==k.VK_ESCAPE) return null;
22:       
23:         char c=Character.toLowerCase(k.getKeyChar());
24: 
25:         if ((c=='p'||c=='-')&&(page>0)) {
26:           page--;
27:           repaint();  
28:         }
29: 
30:         if ((c=='n'||c=='+'||c==' ')&&(((page+1)*pagesize)<strings.length)) {
31:           page++;
32:           repaint();
33:         }
34: 
35:         int kv = (int)c - (int)'a';
36:         if ((kv>=0)&&(kv<pagesize)) {
37:           kv=kv+page*pagesize;
38:           if ((kv>=0)&&(kv<strings.length)) return strings[kv];
39:         }
40:       }
41:     } 
42:     return null;
43:   }
44: 
45:   public void activate() {
46:       // questapp.setKeyHandler(keyhandler);
47:   }
48:     
49:   public ListScreen(String s,String[] invstrings) {
50:     setLayout(new GridLayout(15,1));  
51:     setBackground(new Color(150,75,20));
52:     setFont(QuestApp.mainfont);
53:   
54:     // get the strings
55:     strings=invstrings;
56:     if (strings==null) strings=new String[0];
57:     
58:     // set the list title
59:     title=s;
60:   }
61: 
62:     public void paint(Graphics g) {
63:         if ((page*pagesize)>strings.length) page=0;
64:         
65:         Dimension d = getSize();
66:     
67:     g.drawString(title,20,1*lineheight);
68:         
69:       for(int i=0; i<Math.min(strings.length-(page*pagesize),pagesize); i++) {
70:             String t=strings[i+page*pagesize];
71:             String s="["+(char)((int)'a'+i)+"] "+t;
72:             g.drawString(s,50,lineheight*(i+2));
73:         }
74:         
75:         String bottomstring="ESC=Cancel ";
76:          if (page>0) bottomstring=bottomstring+" P=Previous ";
77:          if (((page+1)*pagesize)<strings.length) bottomstring=bottomstring+" N=Next ";
78:     g.drawString(bottomstring,20,d.height-20);
79:     }
80:     
81: }
