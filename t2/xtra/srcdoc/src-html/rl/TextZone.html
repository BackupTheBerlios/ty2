01: //
02: // Graphical component to display an area of text
03: //
04: 
05: package rl;
06: 
07: import java.awt.*;
08: 
09: public class TextZone extends Component {
10:   public String text;
11:   public Font font;
12:   
13:   private int border=5;
14:   
15:   public TextZone() {
16:     this("");
17:   }
18:   
19:   public TextZone(String t) {
20:     text=t;
21:     font=QuestApp.mainfont;
22:     setFont(font);
23:   }
24:   
25:   public void setText(String s) {
26:     text=new String(s);    
27:   }
28:     
29:   public String getText() {
30:     return text;    
31:   }  
32:   
33:   public void paint(Graphics g) {
34:     FontMetrics met=g.getFontMetrics(g.getFont());
35:     
36:     g.setColor(QuestApp.infotextcolor);
37: 
38:     Rectangle r=getBounds();
39:     int charsize=met.charWidth(' ');
40:     
41:     int linelength; int y; int lineinc;
42:     {
43:       linelength = (r.width-border*2)/charsize;
44:       y=border+met.getMaxAscent();
45:       lineinc=met.getMaxAscent()+met.getMaxDescent();
46:     }
47:         
48:     String[] st=Text.separateString(text,'\n');
49:     
50:     for (int i=0; i<st.length ;i++) { 
51:       String s=st[i];      
52:       
53:       String[] lines=Text.wrapString(s,linelength);
54:       
55:       for (int l=0; l<lines.length ; l++) {
56:         String line=lines[l];
57:         
58:         g.drawString(line,border,y);
59:         y+=lineinc;
60:       }
61:     }
62:   }
63:   
64: }
