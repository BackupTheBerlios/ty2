029:   public static RangedWeapon createRangedWeapon(int level) {
030:     int t=1;
031:     int q=1;
032:     
033:     // adjust level slightly
034:     level=level+RPG.r(3)-RPG.r(3);
035:     
036:     for (int i=0; i<100; i++) {
037:       t=RPG.d(names.length-1);
038:       if (RPG.d(rarity[t])!=1) continue;
039:       
040:       int ql=levels[t];     // level of average weapon
041:       if ((ql+qualitylevels[3])>level) continue; // can't make it bad enough
042:       
043:       q=3;
044:       
045:       // improve the weapon if luck and level allows it
046:       //
047:       //  could use : (RPG.d(13)>q) &&   to limit qualities?
048:       while ( (ql+qualitylevels[q+1]<=level) ) {q++;}
049:       break;
050:     }
051:     return new RangedWeapon(t,q);
052:   }

138:   public void fireAt(Being shooter, Missile missile, Map m, int tx, int ty) {
139:     boolean ishero=(shooter==Game.hero);
140:     
141:     if (missile!=null) {
142:       shooter.aps-=(speeds[type]*10)/(10+shooter.getStat(RPG.ST_ARCHERY));
143:       
144:        // find where thing hits
145:       Point p=m.tracePath(shooter.x,shooter.y,tx,ty);
146:     
147:       // check for current location
148:       if ((p.x==x)&&(p.y==y)) {
149:         shooter.dropThing(missile); 
150:         return;
151:       } else {
152:         // calculate ranged skill
153:         int rsk=(shooter.getStat(RPG.ST_SK)*(3+shooter.getStat(RPG.ST_ARCHERY)))/3;
154:         rsk = (rsk * rskmuls[type] * missile.getStat(RPG.ST_RSKMULTIPLIER)) / 10000;
155:         rsk = rsk + rskbonuses[type] + missile.getStat(RPG.ST_RSKBONUS);
156:  
157:         // calculate distance and offset
158:         int dx=tx-shooter.x; 
159:         int dy=ty-shooter.y;
160:         int dist=(int)Math.sqrt(dx*dx+dy*dy);
161: 
162:         // what are we firing at?
163:         Thing target = m.getMobile(tx,ty);
164:     
165:         int range=missile.getStat(RPG.ST_RANGE);
166:         
167:         Game.mappanel.doShot(shooter.x,shooter.y,tx,ty,100,0.5);
168: 
169:         int shot_difficulty=missile.shotDifficulty(shooter,target,dist);
170:         
171:         // get the hit factor
172:         // 0          = miss
173:         // 1          = normal hit
174:         // 2 or above = critical hit
175:         int factor = RPG.hitFactor(rsk,shot_difficulty);
176:           
177:         if ((target!=null)&&RPG.test(rsk,shot_difficulty)) {
178:           //have scored a hit!
179:           int rst=shooter.getStat(RPG.ST_ST)+shooter.getStat(RPG.ST_RANGER)*4;
180:           rst = (rst * rstmuls[type] * missile.getStat(RPG.ST_RSTMULTIPLIER)) / 10000;
181:           rst = rst + rstbonuses[type] + missile.getStat(RPG.ST_RSTBONUS);
182:           
183:           if (target.isVisible()) {
184:             if (shooter==Game.hero) {
185:               switch (factor) {
186:                 case 1: Game.message("You fire and hit "+target.getTheName()); break;
187:                 case 2: Game.message("You fire and score a great hit on "+target.getTheName()); break;
188:                 default: Game.message("You fire and score a perfect hit on "+target.getTheName()); break;
189:               }              
190:             } else {
191:               Game.message(shooter.getTheName()+" fires and hits "+target.getTheName());
192:             }
193:           }
194:           missile.hit(target,rst*factor);
195:         
196:           // drop item if missile survives
197:           if (RPG.d(100)<=missile.survivals[missile.type]) missile.moveTo(m,tx,ty);
198:         
199:           return;
200:         } else {
201:           Game.message(missile.getTheName()+" misses"); 
202:           missile.moveTo(m,tx,ty);
203:         }
204:      
205:         
206:       }
207:     } else {
208:       if (ishero) Game.message("You have no ammunition!"); 
209:     }
210:   }
