109:   // describe damage
110:   public static String damageDescription(int dam, int max) {
111:     if (dam<=0) return "no damage";
112:     if (max<=0) max=10;
113:     double d=((double)dam)/max; 
114:     if (d<0.2) return "minor damage";
115:     if (d<0.4) return "moderate damage";
116:     if (d<0.7) return "serious damage";
117:     return "critical damage";
118:   }

169:     if (code<=0) code=RPG.d(2);
170:     switch (code) {
171:       case 1: if (checkArtifact("\"Nitrozac\"")==null) {
172:         return addArtifact("Nitrozac Boots",new StandardArmour(StandardArmour.NITROZACBOOTS)); 
173:       }
174:       case 2: if (checkArtifact("\"Ker-Ombai\"")==null) {
175:         return addArtifact("Ker-Ombai Boots",new StandardArmour(StandardArmour.KEROMBAIBOOTS)); 
176:       }
177:     }    
178:     return createMagicItem(Game.hero.getStat(RPG.ST_LEVEL));
179:   }

233:     level=RPG.middle(0,level+RPG.d(6)-RPG.d(6),15);
234:     switch (level) {
235:       case 5: return Potion.createPotion();  
236:       case 8: return new Ring();
237:       case 12: return new Ring();
238:       
239:       default: switch (RPG.d(9)) {
240:         case 1: return new SpellBook();
241:         case 2: return new Wand();
242:         case 3: case 4: return Potion.createPotion();
243:         case 5: return StandardWeapon.createWeapon(StandardWeapon.specialsource,level);  
244:         default: return Scroll.createScroll(level); 
245:       }
