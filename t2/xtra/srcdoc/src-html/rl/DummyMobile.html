01: package rl;
02: 
03: public class DummyMobile extends Being {
04:     private static final Description dummydesc =
05:         new Describer("evil critter","An obnoxious inhabitant of the dungeon");
06:     
07:   public Description getDescription() { return dummydesc;}
08:   
09:   public DummyMobile(String s, int i) {
10:     super();
11:     image=i;    
12:     hps=RPG.d(4);
13:   }
14:   
15:   public void die() {
16:       if (place==null) return;
17:       Map m=getMap();
18:     
19:     if (m.getObject(x,y,Decoration.class)==null)    {
20:         m.addThing(new Decoration(Decoration.DEC_REDBLOOD),x,y);
21:     }
22:     if (RPG.sometimes())    {
23:         m.addThing(StandardArmour.createArmour(0),x,y);
24:     }
25:     super.die();
26:   }
27:       
28:     public void action() {
29:       if (!(place instanceof Map)) throw new Error("Lost mobile....");
30:     Map m=(Map) place;
31:     Mobile h=Game.hero;
32:         
33:       while (aps>=0) {
34:       
35:         int nx=x-1+RPG.r(3);
36:         int ny=y-1+RPG.r(3);
37:       
38:         if (RPG.often()) nx=x+RPG.sign(h.x-x);
39:         if (RPG.often()) ny=y+RPG.sign(h.y-y);
40:       
41:         // move the mobile
42:         if (!m.isBlocked(nx,ny)) moveTo(m,nx,ny);
43:         aps=aps-100;
44:       }    
45:     }
46:     
47:     public int getStat(int s) {
48:         switch (s) {
49:       case RPG.ST_ASK: return 4;            
50:       case RPG.ST_AST: return 4;            
51:       case RPG.ST_DSK: return 4;            
52:             default: return super.getStat(s);
53:         }
54:     }
55: }
