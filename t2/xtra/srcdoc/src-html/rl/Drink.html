01: package rl;
02: 
03: public class Drink extends ItemType {
04:   protected int desc;
05:     
06:   private static final Description[] descriptions=
07:      {new Describer("bottle of wine","bottles of wine","A bottle of wine. Good vintage.")
08:      ,new Describer("keg of beer","kegs of beer","A keg of strong beer. It seems a lusty brew.")
09:      };
10: 
11:   public static final Drink WINE=     new Drink(223,0);
12:   public static final Drink BEERKEG=  new Drink(263,1);
13:     
14:   public Drink(int im, int d) {
15:     image=im;
16:     desc = d; 
17:   }
18:    
19:   public static TypeStack createDrink(int f) {
20:     if (f==0) f=RPG.d(2);
21:         
22:     switch (f) {
23:       case 1:
24:         return new TypeStack(WINE,1); 
25:       case 2:
26:         return new TypeStack(BEERKEG,1); 
27:       default:
28:         throw new Error("Drink type not recognised");
29:     }
30:   }
31:     
32:   public int getUse(TypeStack s) {return Thing.USE_NORMAL|Thing.USE_DRINK;}
33:     
34:   public void use(TypeStack s, Being user) {
35:     use(s,user,Thing.USE_DRINK);    
36:   }
37:     
38:     public int getWeight(TypeStack s) {return 750;}
39:     
40:     public void use(TypeStack s, Being user, int usetype) {
41:     boolean ishero=(user==Game.hero);
42:     
43:     if (usetype!=Thing.USE_DRINK) return;
44: 
45:     Thing t=s.remove(1);
46: 
47:     if (ishero) {
48:           Game.message("You drink "+t.getTheName());
49:           Game.message("Very refreshing");
50:     }
51:     user.aps-=500;
52:     
53:     if ((s.type==WINE)||(s.type==BEERKEG)) {
54:       user.mps=RPG.max(user.mps-RPG.d(2,6),0);
55:       if (ishero) Game.message("You feel a little giddy");
56:     }
57:   }
58:     
59:     public Description getDescription(TypeStack s) {return descriptions[desc];}
60: }
