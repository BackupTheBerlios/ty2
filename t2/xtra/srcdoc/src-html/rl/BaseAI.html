003: // the root AI object
004: public class BaseAI implements AI, java.io.Serializable {
005:     public boolean doAction(Mobile m) {
006:     // don't do anything
007:     m.aps=RPG.min(m.aps,-1);
008:     return true;    
009:     }    
010:     
011:     // don't need these except for complex AI objects
012:     public void register(Mobile m) {}
013:     public void unregister(Mobile m) {}
014:     
015:     public int getState(Mobile m) {return m.getStat(RPG.ST_STATE);}
016:     
017:   public boolean isHostile(Mobile m, Mobile b) {
018:     int bstate=b.getStat(RPG.ST_STATE);
019:     int mstate=m.getStat(RPG.ST_STATE);
020:     
021:     // everyone fights chaotics
022:     // including other chaotics
023:     if (bstate==AI.STATE_CHAOTIC) return true;
024:     
025:     // if it's a follower, decide whether we like the leader
026:     if (bstate==AI.STATE_FOLLOWER) return isHostile(m,b.getAI().getLeader());
027:     
028:     switch (getState(m)) {
029:       case AI.STATE_CHAOTIC: return true;
030:        
031:       case AI.STATE_HOSTILE: {
032:         // attacks hero
033:         if (b==Game.hero) return true;
034:         // attacks townies
035:         if (bstate==AI.STATE_INHABITANT) return true;
036:         // attack anyone not on the same side
037:         if (b.getStat(RPG.ST_SIDE)!=m.getStat(RPG.ST_SIDE)) return true;
038:         
039:         // can't see a reason to attack. damn.
040:         return false;
041:       }
042:     
043:       case AI.STATE_INHABITANT: {
044:         // fight against hostiles
045:         if (bstate==AI.STATE_HOSTILE) return true;
046:         
047:         // fight the hero if the map is angry
048:         if ((b==Game.hero)&&(m.getMap()!=null)&&(m.getMap().isAngry())) return true;
049:         return false; 
050:       }
051:       
052:       case AI.STATE_FOLLOWER: {
053:         // we like our own leader, naturally
054:         if (b==getLeader()) return false;
055:         
056:         // follow leader's example
057:         return isHostile(getLeader(),b);  
058:       }
059:       
060:       default:
061:         // neutrals don't attack anyone
062:         return false;
063:     }
064:   }
065:   
066:   public Mobile getLeader() {
067:     return null; 
068:   }
069:   
070:   public void turnNasty(Mobile m) {
071:     // Could turn townies into nasties
072:     // but prefer changing entire map state
073:     // m.setAI(NastyCritterAI.instance);
074:     m.setStat(RPG.ST_STATE,AI.STATE_HOSTILE);
075:     m.getMap().setAngry(-1);
076:   }
077:   
078:   public int notify(Mobile m, int eventtype, int ext, Object o) {
079:     Map map=m.getMap();
080:     if (map==null) return 0;
081:     if (m.isHostile(Game.hero)) return 0;
082:     switch (eventtype) {
083:       case EVENT_ATTACKED:
084:         if ( (o instanceof Hero) && (ext==m.getStat(RPG.ST_SIDE)) ) {
085:           Game.message(m.getTheName()+" shouts angrily!"); 
086:           Hero h=(Hero)o;
087:           turnNasty(m); 
088:         }
089:         return 0;
090:       case EVENT_THEFT:
091:         if (m.isVisible()) {
092:           if (RPG.test(m.getStat(RPG.ST_IN),((Hero)o).getStat(RPG.ST_SK))) {
093:             Game.message(m.getTheName()+" shouts angrily!"); 
094:             turnNasty(m); 
095:             map.areaNotify(m.x,m.y,10,eventtype,1,o);
096:           } else {
097:             Game.message(m.getTheName()+" doesn't notice"); 
098:           }
099:         }
100:         return 0;
101:     }
102:     return 0;
103:   }
104: }
