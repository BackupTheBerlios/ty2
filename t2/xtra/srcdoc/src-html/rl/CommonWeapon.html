068:   public static Weapon createWeapon(int level) {
069:     // add some randomness to level
070:     level=level+RPG.r(3)-RPG.r(3);
071:     
072:     // decide the material
073:     int m=Material.getWeaponMaterial(level);
074:     
075:     for (int i=0; i<100; i++) {
076:       int t=RPG.d(names.length-1);
077:       if (RPG.d(rarity[t])!=1) continue;
078:       
079:       int ql=Material.getLevel(m)+levels[t];     // level of average weapon
080:       if ((ql+qualitylevels[3])>level) continue; // can't make it bad enough
081:       
082:       int q=3;
083:       
084:       // improve the weapon if luck and level allows it
085:       //
086:       //  could use : (RPG.d(13)>q) &&   to limit qualities?
087:       while ( (ql+qualitylevels[q+1]<=level) ) {q++;}
088:       
089:       return new CommonWeapon(t,m,q);
090:     }
091:     
092:     return new CommonWeapon(1,Material.IRON,4);
093:   }

102:   public int getStat(int s) {
103:     switch(s) {
104:       case RPG.ST_ATTACKCOST: return  attackcosts[type];
105:       case RPG.ST_ASKMULTIPLIER: 
106:         return skillmuls[type]*qualitymuls[quality]*Material.getWeaponSkill(material)/10000;
107:       case RPG.ST_ASTMULTIPLIER: 
108:         return strengthmuls[type]*qualitymuls[quality]*Material.getWeaponStrength(material)/10000;
109:       case RPG.ST_DSKMULTIPLIER: 
110:         return defencemuls[type]*qualitymuls[quality]*Material.getWeaponSkill(material)/10000;
111:       
112:       case RPG.ST_ASKBONUS: 
113:         return skillbonuses[type]*qualitymuls[quality]*Material.getWeaponSkill(material)/10000;
114:       case RPG.ST_ASTBONUS: 
115:         return strengthbonuses[type]*qualitymuls[quality]*Material.getWeaponStrength(material)/10000;
116:       case RPG.ST_DSKBONUS: 
117:         return defencebonuses[type]*qualitymuls[quality]*Material.getWeaponSkill(material)/10000;
118:       
119:       case RPG.ST_ITEMVALUE: {
120:         int val=Material.getValue(material,sizes[type]);
121:         val=val*relativevalues[type]/100;
122:         return RPG.niceNumber(val*Lib.qualityvalues[quality]/100);
123:       }
124:       
125:       default: return super.getStat(s);
126:     }    
127:   }

141:   // return toughness for the weapon
142:   // increases with higher quality and better materials
143:   public int getToughness() {
144:     if ((quality>=13)||isArtifact()) return 0;
145:     
146:     // base toughness for average iron sword
147:     int tuff=200;
148:     
149:     // rusty items are much weaker!
150:     if (isRusty()) tuff=tuff/4;
151:     
152:     // modify for quality
153:     tuff=tuff*quality/5;
154:     
155:     // modify for type and material
156:     tuff = (((tuff*toughnesses[type])/100)*Material.getToughness(material))/100;
157:     
158:     return tuff; 
159:   }
