167:   // move to square, landing on floor
168:     public void moveTo(Map m, int tx, int ty) {
169:         m.addThing(this,tx,ty);
170:         m.hitSquare(this,tx,ty,true);
171:     }
172: 
173:   // move randomly to on side
174:   public void displace() {
175:     if (place instanceof Map) {
176:       Map m= (Map) place;
177:       int nx=x+RPG.r(3)-1;
178:       int ny=y+RPG.r(3)-1;
179:       if ((m.getTile(nx,ny)&Tile.TF_BLOCKED)!=0) return;
180:       if (isBlocking()&&m.isBlocked(nx,ny)) return;
181:       moveTo(m,nx,ny);
182:     } 
183:   }
184: 
185:   public void throwAt(Being thrower, Map m, int tx, int ty) {
186:     moveTo(m,tx,ty); 
187:   }
188: 
189:   public boolean isWithin(ThingOwner thing) {
190:     for (Thing head=this; head!=null; head=(Thing)head.place) {
191:         if (head.place==thing) return true;
192:         if (!(head.place instanceof Thing)) return false;
193:     }
194:     return false;
195:   }
