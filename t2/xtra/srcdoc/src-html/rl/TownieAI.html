01: package rl;
02: 
03: public class TownieAI extends BaseAI {
04:     public static final TownieAI instance=new TownieAI();
05:     
06:   public boolean doAction(Mobile m) {
07:     Hero h=Game.hero;
08:     
09:     int state=m.getStat(RPG.ST_STATE);
10:     
11:     // get target location
12:     int tx=m.getStat(RPG.ST_TARGETX);
13:     int ty=m.getStat(RPG.ST_TARGETY);
14:     
15:     // if initial value, then set target to current location
16:     if (tx==0) {
17:       m.setStat(RPG.ST_TARGETX,m.x);
18:       m.setStat(RPG.ST_TARGETY,m.y);
19:     }
20:     
21:       while (m.aps>=0) {
22:       Map map=m.getMap();
23: 
24:       // go for nearest foe if applicable
25:       Mobile target=(Mobile)map.findNearestFoe(m);
26:       if ((target!=null)&&m.canSee(target)) {
27:         NastyCritterAI.instance.doAttack(m,target.x,target.y);
28:         return true;
29:       };
30:     
31:         int dx=RPG.r(3)-1;
32:         int dy=RPG.r(3)-1;
33: 
34:         if (RPG.d(4)>1) dx=RPG.sign(tx-m.x);
35:          if (RPG.d(4)>1) dy=RPG.sign(ty-m.y);
36:           
37:       int nx=m.x+dx;
38:       int ny=m.y+dy;
39:       
40:       
41:       if (m.canMove(map,nx,ny)) {
42:         NastyCritterAI.instance.doAttack(m,nx,ny);
43:         return true;
44:       } else {
45:         if (RPG.d(6)==1) {
46:           m.setStat(RPG.ST_TARGETX,RPG.rspread(0,map.getWidth()-1));
47:           m.setStat(RPG.ST_TARGETY,RPG.rspread(0,map.getHeight()-1));
48:         }
49:       }
50:         m.aps=m.aps-10000/m.getStat(RPG.ST_MOVESPEED);    
51:       }    
52:     return true;
53:     }    
54:   
55: }
