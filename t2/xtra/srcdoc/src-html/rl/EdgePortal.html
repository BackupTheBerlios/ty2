01: // The EdgePortal class represents a special kind of portal which can
02: // be accessed via any point on specified edges of a Map. 
03: // It is intended mainly for outdoor locations etc.
04: 
05: package rl;
06: 
07: public class EdgePortal extends Portal {
08:   // Constants indicate which map edge is used
09:   // can use ZONE to allow  arrivals anywhere in the map
10:   public static final int NORTH=1;
11:   public static final int SOUTH=2;
12:   public static final int EAST=4;
13:   public static final int WEST=8;
14:   public static final int ALL=15;
15:   public static final int ZONE=16;
16: 
17:   // what side the edge portal is on
18:   protected int side;
19:   
20:   public EdgePortal(int s) {
21:     super(0);  // invisible portal superclass
22:     side=s; 
23:   }
24:   
25:   public boolean isEdge(int tx, int ty) {
26:     if ((side&ZONE)>0) return true;
27:     if (((side&NORTH)>0)&&(ty==0)) return true;
28:     if (((side&WEST)>0)&&(tx==0)) return true;
29:     if (((side&EAST)>0)&&(tx==getMap().getWidth()-1)) return true;
30:     if (((side&SOUTH)>0)&&(ty==(getMap().getHeight()-1))) return true;
31:     return false;
32:   }
33:   
34:   public void arrive(Thing t) {
35:     
36:     Map map=getMap();
37:     
38:     if (map==null) throw new Error("EdgePortal Error - no map specified");
39:     
40:     int tx=-1; int ty=-1; int i=0;
41:     
42:     while (tx==-1) {
43:       switch (RPG.d(5)) {
44:         case 1: if ((side&NORTH)>0) {ty=0; tx=RPG.r(map.getWidth());}; break;
45:         case 2: if ((side&SOUTH)>0) {ty=map.getHeight()-1; tx=RPG.r(map.getWidth());}; break;
46:         case 3: if ((side&WEST)>0) {tx=0; ty=RPG.r(map.getHeight());}; break;
47:         case 4: if ((side&EAST)>0) {tx=map.getWidth()-1; ty=RPG.r(map.getHeight());}; break;
48:         case 5: if ((side&ZONE)>0) {tx=RPG.r(map.getWidth()); ty=RPG.r(map.getHeight());}; break;
49:       } 
50:       
51:       i++; if (i>5000) throw new Error("EdgePortal Error - unable to find space to add "+t.getName());
52:     
53:       if ((tx>=0)&&(map.isBlocked(tx,ty))) tx=-1;
54:     };
55:     
56:     t.moveTo(map,tx,ty);
57:   }
58: }
