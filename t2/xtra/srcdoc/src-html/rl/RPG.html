497:   // Poisson distribution
498:   public static int po(double x) {
499:     int r=0;
500:     double a=rand.nextDouble();
501:     if (a>=0.99999999) return 0;
502:     double p=Math.exp(-x);
503:     while (a>=p) {
504:       r++;
505:       a=a-p;
506:       p=p*x/r;
507:     }
508:     return r;
509:   }

587:   // sum of best r from n s-sided dice
588:     public static int best(int r, int n, int s) {
589:     if ((n<=0)||(r<0)||(r>n)||(s<0)) return 0;
590:     
591:     int[] rolls=new int[n];
592:     for (int i=0; i<n; i++) rolls[i]=d(s);
593:     
594:     boolean found;
595:     do {
596:       found=false;
597:       for (int x=0; x<n-1; x++) {
598:         if (rolls[x]<rolls[x+1]) {
599:           int t=rolls[x];
600:           rolls[x]=rolls[x+1];
601:           rolls[x+1]=t;
602:           found=true; 
603:         } 
604:       }
605:     } while (found);
606:     
607:     int sum=0;
608:     for (int i=0; i<r; i++) sum+=rolls[i];
609:     return sum; 
610:   }

629:     public static int percentile(int var, int base) {
630:         // Don't divide by zero.
631:         if (base == 0) { return 0; }
632:         
633:         // Make the percentile
634:         int p = var*100/base;
635:         
636:         // Sometimes very small percentages (0.80%) will be rounded down
637:         // to 0, which, if var is not 0, may look funky.  Under this condition,
638:         // make the percentile 1%.
639:         if ((var > 0) && (p == 0)) {
640:             p = 1;
641:         }
642:         
643:         return p;
644:     }

646:   public static Object[] subList(Object[] list, Class c) {
647:     Object[] temp = new Object[list.length];
648:     int tempcount=0;
649:     for (int i=0; i<list.length; i++) {
650:       Object t=list[i];
651:       if (c.isInstance(t)) {
652:         temp[tempcount]=t;
653:         tempcount++;  
654:       }
655:     }
656:     Object[] result = new Object[tempcount];
657:     if (tempcount>0) System.arraycopy(temp,0,result,0,tempcount);
658:     return result;
