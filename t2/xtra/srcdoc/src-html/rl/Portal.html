001: // A Portal is a special object which represents a way to travel either
002: //   - from one map to another
003: //   - between distant locations on the same map
004: //
005: // use Portal.travel(thing) to send an object, create or Hero to the
006: // target destination.
007: //
008: // Portals will often operate in pairs, e.g. stairs on two different levels
009: // which link to each other.
010: //
011: // If of Portal is not twinned with another Portal, then a secret marker
012: // is added to the destination location.
013: //
014: // Portals also implement functionality for creatures to follow the hero
015: // through the portal if they are in pursuit.
016: 
017: package rl;
018: 
019: public class Portal extends Scenery implements Active, ThingOwner {
020:   // the target map, which needs to be private
021:   protected Portal target;
022:     
023:   // average number of creatures to spawn 
024:   // per 1000 turns (i.e. 100000 ticks)
025:   public int spawnrate;
026:   
027:   // level of spawned creatures
028:   public int spawnlevel;
029:   
030:   // source of creature IDs to spawn
031:   public int[] spawnsource;
032:   
033:   // linked list of critters about to emerge
034:   // e.g. if they have followed the player
035:   public Thing critters;
036:     
037:     
038:   private int type;
039:   
040:   public static final String[] names =   {"void",  "stairs up",  "stairs down",  "ladder up",  "ladder down",  "small cave",  "signpost", "pit"};
041:   public static final int[] images =     {0,       0,            1,              2,            3,              303,           66,         41};
042:   public static final int[] visible =    {0,       1,            1,              1,            1,              1,             1,          1};
043:   public static final int[] cantravel =  {1,       1,            1,              1,            1,              0,             1,          1};
044:   
045:   public int getImage() {
046:     return images[type]; 
047:   }
048:   
049:   // image=0
050:   public static final Description DESC_UP=
051:       new Describer("stairs up","Stairs leading upwards");
052:   // image=1
053:   public static final Description DESC_DOWN=
054:       new Describer("stairs down","Stairs leading downwards");
055:   // image=2
056:   public static final Description DESC_LADDERUP=
057:       new Describer("ladder up","Ladder leading upwards");
058:   // image=3
059:   public static final Description DESC_LADDERDOWN=
060:       new Describer("ladder down","Ladder leading downwards");
061:   
062:   // image=303
063:   public static final Description DESC_SMALLCAVE=
064:       new Describer("small cave","A small cave. You cannot fit down it.");
065:   
066:   public Portal(String s) {
067:     int i=Text.index(s,names);
068:     type= i>=0 ? i : 0;
069:   }
070:   
071:   // create a dummy portal object
072:   public Portal(int t) {
073:     type=t; 
074:   }
075: 
076:   // doesn't block movement
077:   public boolean isBlocking() {return false;}
078:   
079:   
080:   //return the current target object
081:   public Thing getTarget() {
082:     return target;    
083:   }
084:   
085:   public Map getTargetMap() {
086:        return getTarget().getMap();
087:   }
088:   
089:   // arrive at the portal
090:   
091:   public void arrive(Thing t) {
092:     if (t==Game.hero) {
093:       t.moveTo(getMap(),x,y);
094:       return; 
095:     }
096:      
097:     if (t instanceof Mobile) {
098:       Mobile b=(Mobile)t;
099: 
100:       if (getMap() instanceof WorldMap) {
101:         if (b.getAI().getLeader()==Game.hero) {
102:       
103:           // add to inventory trick if map is a WorldMap
104:           Game.hero.addThing(b); 
105:           return;
106:         } 
107:         
108:         // don't do anything - creatures don't follow onto WorldMaps yet
109:         return;
110:       }
111:     } 
112:      
113:     addThing(t);
114:   }
115:   
116:   
117:   public boolean canTravel(Thing t) {return cantravel[type]>0;}
118:   
119:   // go through the portal
120:   public void travel(Thing t) {
121:     if (!canTravel(t)) return;
122:     
123:     // is there a destination portal?
124:     Portal destportal=getDestination(); 
125: 
126:     // special handling if it is the hero
127:     if (t==Game.hero) { 
128:       // retrieve current map
129:       Map curmap=t.getMap();
130:       int cx=t.x;
131:       int cy=t.y;
132:       
133:       if (curmap!=null) {
134:         // see if there are any critters going to try to follow
135:         Thing[] things=curmap.getThings(cx-10,cy-10,cx+10,cy+10);
136:         // System.out.println(things.length);
137:         for (int i=0; i<things.length; i++) if (things[i] instanceof Being) {
138:           Being a=(Being)things[i];
139:           if (a.isVisible()&&((a.getAI() instanceof FollowerAI)||((Being)a).isHostile(Game.hero))) {
140:             travel(a);           
141:           }
142:         }
143:       }
144:       
145:       // reverse direction of critters in transit
146:       while (critters!=null) {
147:         travel(getCritter()); 
148:       }
149:       
150:       // add followers to destination portal
151:       Thing[] followers= Game.hero.inv.getContents(Being.class);
152:       for (int i=0; i<followers.length; i++) {
153:         travel(followers[i]); 
154:       }
155:     }
156:       
157:     // move traveller to target map
158: 
159:     // add followers to map on arrival
160:     //if ((t==Game.hero)&&(!(map instanceof WorldMap))) {
161:     //  Thing[] followers= Game.hero.inv.getContents(Being.class);
162:     //  for (int i=0; i<followers.length; i++) {
163:     //    Thing f=followers[i];
164:     //    f.moveTo(map,tx,ty);
165:     //    f.displace(); 
166:     //  }
167:     //}
168:     
169:     destportal.arrive(t); 
170:   }
171:   
172:   public void addThing(Thing c) {
173:     if (c==null) return;
174:     c.remove();
175:     c.place=this;
176:     c.next=critters;
177:     critters=c; 
178:   }
179:   
180:   // add a critter to portal monster list
181:   // give to hero if a follower and no map
182:   public void addCritter(Thing c) {
183:     if (place!=null) {
184:       addThing(c); 
185:     } else {
186:       if (c instanceof Being) {
187:         Being b=(Being)c;
188:         if (b.getAI().getLeader()!=null) {
189:           // add to leader's inventory 
190:           b.getAI().getLeader().addThing(c);
191:         } 
192:       }
193:     }
194:   }
195:   
196:   public Thing getCritter() {
197:     Thing c=critters;
198:     c.remove();
199:     return c; 
200:   }
201:   
202:   // link portal to entrance of given map
203:   public void makeLink(Map m) {
204:     if (!(place instanceof Map)) throw new Error("Bad portal link attempted");
205:     if (m.entrance==null) {
206:       throw new Error("Portal make link failed.");
207:       // m.entrance=new Portal(0,false,null); 
208:     }
209:     target=m.entrance;
210:     m.entrance.target=this; 
211:   }
212:   
213:   // Portal may introduce new creatures to dungeon
214:   // or do other stuff based on time elapsed
215:   public void action(int time) {
216:     Map m=getMap();
217:     
218:     // creature spawning
219:     if ( (critters==null) && (RPG.po(time*spawnrate,100000)>0) ) {
220:       addCritter(Creature.createCreature(spawnsource,spawnlevel));
221:     }
222:      
223:     // bring creatures out of list
224:     if (critters!=null) {
225:       int cx=x+RPG.r(3)-1; int cy=y+RPG.r(3)-1;
226:       if (!m.isBlocked(cx,cy)) {
227:         m.addThing(critters,cx,cy);
228:       } 
229:     }
230:   }
231:  
232:   public void setDestination(Portal t) {
233:     target=t; 
234:   }
235: 
236:   public Portal getDestination() {
237:     return target; 
238:   }
239: 
240:   public void removeThing(Thing t) {
241:     if (critters==t) {
242:       // first thing in list
243:       critters=t.next;
244:       t.next=null;
245:       t.place=null; 
246:     } else {
247:       // scan list
248:       Thing h=critters;
249:       while (h!=null) {
250:         if (h.next==t) {
251:           h.next=t.next;
252:           t.next=null;
253:           t.place=null;
254:           return; 
255:         }
256:         h=h.next;
257:       }
258:     }
259:   }
260: 
261:   public void setDestination(Map m, int x, int y) {
262:     target=new Portal(0);
263:     m.addThing(target,x,y);
264:   }
265:   
266:   public boolean isInvisible() {return visible[type]==0;}
267:   
268:   public String getName() {return names[type];}
269:   
270:   public Description getDescription() {return this;}
271: }
