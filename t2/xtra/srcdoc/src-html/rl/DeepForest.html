011:   public DeepForest(int w, int h) {
012:       super(w,h);
013:   
014:     setTheme(new Theme("deepforest"));
015:     
016:     fillArea(0,0,w-1,h-1,walltile);
017:     fillArea(1,1,w-2,h-1,0);
018: 
019:     // put in the entrance
020:     //Point es=new Point(w/4,RPG.rspread(w/4,(w*3)/4));
021:     //if (RPG.d(2)==1) {
022:     //  int t=es.x; es.x=es.y; es.y=t;
023:     //}
024:     //if (RPG.d(2)==1) {
025:     //  es.x=w-es.x; es.y=h-es.y;
026:     //}
027:     entrance=new EdgePortal(EdgePortal.SOUTH);
028:     addThing(entrance,w/2,h-1);
029:     
030:     Point es=new Point(w/2,(3*h)/4);
031:     
032:     makePath(w/2,h-4,es.x,es.y);
033:     fillArea(w/2-1,h-8,w/2+1,h-1,floortile);
034:     addThing(Coin.createMoney(600),es.x,es.y);
035:     // addThing(new Coin(1,20),es.x,es.y);
036:     
037:     // your goal
038:     makeClearing(w-es.x,h-es.y);
039:     makeHut(w-es.x-5,h-es.y-4,w-es.x+5,h-es.y+4);
040:     
041:     //make a route
042:     Point tp=new Point(RPG.r(width-2)+1,RPG.r(height-2)+1);
043:     makePath(es.x,es.y,tp.x,tp.y);
044:     makePath(tp.x,tp.y,w-es.x,h-es.y);
045:     
046:     for (int loop=0; loop<10; loop++) {
047:       Point ls=findFreeSquare();
048:       int cx=RPG.r(width-2)+1;    
049:       int cy=RPG.r(width-2)+1;
050:       makeClearing(cx,cy);    
051:       makePath(ls.x,ls.y,cx,cy);
052:     }
053:     
054:     
055:     replaceTiles(0,walltile);
056:     
057:     decorateForest(0,0,w-1,h-1);
058:   
059:     // fill in the blanks
060:     replaceTiles(0,walltile);
061: 
062:     // drop in some critters
063:     for (int i=0; i<8; i++) {
064:       addBeasties(RPG.r(w),RPG.r(h));
065:       Point p=findFreeSquare();
066:       addBaddie(p.x,p.y);    
067:     }
068: 
069:     // the lair
070:     Point dps=findFreeSquare();
071:     //Point dps=new Point(w/2,h-3);
072:     Portal dp=new Portal("stairs down");
073:     addThing(dp,dps.x,dps.y);
074:     dp.makeLink(new GuardianVault(Lib.createCreature(6), StandardArmour.createArmour(6)));
075: 
076:     // the infinite dungeon
077:     Point ids=findFreeSquare();
078:     //Point dps=new Point(w/2,h-3);
079:     Portal id=new InfinitePortal(2,1);
080:     addThing(id,ids.x,ids.y);
081:   }

083:   private void makePath(int x1,int y1, int x2, int y2) {
084:     int dx; int dy;
085:     while ((x1!=x2)||(y1!=y2)) {
086:       if (getTile(x1,y1)==0) {
087:           setTile(x1,y1,floortile);
088:       }    
089:       if (RPG.d(3)==1) {
090:         dx=RPG.sign(x2-x1);    
091:         dy=RPG.sign(y2-y1);    
092:       } else {
093:         dx=RPG.r(3)-1;    
094:         dy=RPG.r(3)-1;    
095:       }
096:       switch (RPG.d(3)) {
097:         case 1: dx=0; break;
098:         case 2: dy=0; break;    
099:       }
100:       x1+=dx;
101:       y1+=dy;
102:       x1=RPG.middle(1,x1,width-1);
103:       y1=RPG.middle(1,y1,height-1);
104:     }    
105:   }
106: 
107:   private void makeClearing(int x, int y) {
108:     int w=RPG.d(8); 
109:     int h=RPG.d(8);
110:     int x1=x-w;
111:     int y1=y-h;
112:     int x2=x+w;
113:     int y2=y+h;
114:       
115:     int cx=(x1+x2)/2;
116:     int cy=(y1+y2)/2;
117:     
118:     for (int lx=x1; lx<=(x1+w*2); lx++) for (int ly=y1; ly<(y1+h*2); ly++) {
119:       if ( (((lx-cx)*(lx-cx)*100)/(w*w) + ((ly-cy)*(ly-cy)*100)/(h*h)) < 100 ) {
120:           if (getTile(lx,ly)==0) setTile(lx,ly,floortile);
121:       }
122:     }  
123:     
124:     if ((w>=4)&&(h>=4)&&(RPG.d(6)==1)&&(x!=entrance.x)) {
125:       // do a fairy ring
126:       for (int i=0; i<RPG.d(10); i++) {
127:         float a=RPG.random()*100;
128:         int mx=(int)(0.5+cx+3*Math.sin(a));
129:         int my=(int)(0.5+cy+3*Math.cos(a));
130:         if (!isBlocked(mx,my)) addThing(Food.createFood(Food.MUSHROOM),mx,my);    
131:       }
132:       
133:         for (int i=0; i<=RPG.d(2,4); i++) {
134:           Creature c=new Creature(Creature.RABBIT);
135:           c.setStat(RPG.ST_TARGETX,cx);    
136:           c.setStat(RPG.ST_TARGETX,cy);
137:           addThing(c,cx,cy);    
138:         }
139:     }  
140:   }

143:   public void makeHut(int x1,int y1,int x2,int y2) {
144:       fillArea(x1,y1,x2,y2,floortile);
145:       int w=x2-x1-4;
146:       int h=y2-y1-4;
147:       
148:       if ((h<3)||(w<3)) return;
149:       
150:       fillArea(x1+1,y1+1,x2-1,y2-1,Tile.CAVEWALL);
151:       fillArea(x1+2,y1+2,x2-2,y2-2,Tile.FLOOR);
152:       
153:       int doorx; int doory;
154:       
155:       if (RPG.d(2)==1) {
156:           doorx=RPG.rspread(x1+2,x2-2);
157:           doory=(RPG.d(2)==1)?y1+1:y2-1;
158:       } else {
159:           doory=RPG.rspread(y1+2,y2-2);
160:           doorx=(RPG.d(2)==1)?x1+1:x2-1;
161:       }
162:       
163:     setTile(doorx,doory,floortile);
164:     addThing(new Door(),doorx,doory);
165:       
166:     Portal p=new Portal("stairs down");
167:     addThing(p,RPG.rspread(x1+2,x2-2),RPG.rspread(y1+2,y2-2));
168:     p.makeLink(new EmeraldDungeon(71,71));
169: 
170:     addThing(new Creature(Creature.GOBLIN),RPG.rspread(x1+2,x2-2),RPG.rspread(y1+2,y2-2));
171:     addThing(new Creature(Creature.GOBLIN),RPG.rspread(x1+2,x2-2),RPG.rspread(y1+2,y2-2));
172:     addThing(Lib.createItem(0),RPG.rspread(x1+2,x2-2),RPG.rspread(y1+2,y2-2));
173:     addThing(Lib.createItem(0),RPG.rspread(x1+2,x2-2),RPG.rspread(y1+2,y2-2));
174:     addThing(Food.createFood(0),RPG.rspread(x1+2,x2-2),RPG.rspread(y1+2,y2-2));
175:     addThing(Drink.createDrink(0),RPG.rspread(x1+2,x2-2),RPG.rspread(y1+2,y2-2));
176:     addThing(Food.createFood(0),RPG.rspread(x1+2,x2-2),RPG.rspread(y1+2,y2-2));
177:     addThing(Drink.createDrink(0),RPG.rspread(x1+2,x2-2),RPG.rspread(y1+2,y2-2));
178:     addThing(new PowderKeg(),RPG.rspread(x1+2,x2-2),RPG.rspread(y1+2,y2-2));
179:   }

182:   public void decorateForest(int x1,int y1,int x2,int y2) {
183:        for (int x=x1; x<=x2; x++) for (int y=y1;y<=y2; y++) {
184:          if (isBlocked(x,y)) continue;
185:          if (!(getTile(x,y)==floortile)) continue;
186:          else {
187:              switch(RPG.d(60)) {
188:                case 1: case 2:    
189:                    addThing(GameScenery.create(4),x,y);
190:                    break;
191:                case 5:    
192:                    addThing(GameScenery.create(5),x,y);
193:                    break;         
194:                case 6:    
195:                    addThing(Food.createFood(Food.MUSHROOM),x,y);
196:                    break;         
197:                }
198:          }    
199:        }
200: 
201:   }

203:   private void addBeasties(int x, int y) {
204:     Creature m;
205:     switch (RPG.d(5)) {
206:       case 1: m=new Creature(Creature.TRIFFID); break;
207:       default: m=new Creature(Creature.SPIDER); break;
208:     }    
209:         
210:     for (int i=0; i<4; i++) {
211:       int bx=x-2+RPG.r(5);
212:       int by=y-2+RPG.r(5);
213:       if (isClear(bx,by)) {
214:           addThing(m,bx,by);
215:         m=m.cloneType(); // create more of the same
216:       }    
217:     }
218:   }

220:   private void addGuard(int x, int y) {
221:     if (isClear(x,y)) {
222:         Mobile m=new Creature(Creature.SNAKE);
223:         addThing(m,x,y);
224:     }    
225:   }
226: 
227:   private void addWandering() {
228:     Point p=findFreeSquare();
229:     int x=p.x; int y=p.y;
230:     if (!isVisible(x,y)) {
231:       Mobile m=null;
232:       m=Creature.createMonster(getLevel());
233:       if (m!=null) addThing(m,x,y);    
234:     }    
235:   }
236: 
237:   private void addBaddie(int x, int y) {
238:     if (isClear(x,y)) {
239:         Mobile m=new Creature(Creature.MUTANT);
240:         m.addThing(Lib.createItem(0));
241:         addThing(m,x,y);
242:     }    
243:   }

249:     super.action(time);
