017:       super(w,h);
018:     level=lev;
019:     
020:     // randomly choose wall type
021:     switch (RPG.d(6)) {
022:       case 1: 
023:            setTheme(new Theme("labyrinthe")); break;    
024:       case 2: case 3:
025:         setTheme(new Theme("dungeon")); break;    
026:       case 4: 
027:         setTheme(new Theme("caves")); break;  
028:       case 5: 
029:         setTheme(new Theme("sewers")); break;  
030:       case 6: 
031:         setTheme(new Theme("goblins")); break;  
032:     }
033:     
034:     makeDungeon(0,0,w-1,h-1);
035:   
036:     // fill in the blanks
037:     replaceTiles(0,walltile);
038: 
039:     // drop in some critters
040:     for (int i=0; i<((w*h)/500); i++) {
041:       addBeasties(RPG.r(w),RPG.r(h));
042:       Point p=findFreeSquare();
043:       addBaddie(p.x,p.y);    
044:     }
045: 
046:     Point ent=findFreeSquare();
047:     entrance=new Portal("stairs up");
048:     addThing(entrance,ent.x, ent.y);

052:   // scan in direction (dx,dy) to find blank square
053:   public Point findEdge(int x,int y, int dx, int dy) {
054:     while (getTile(x,y)!=0) {
055:       x+=dx;
056:       y+=dy;
057:     }    
058:     if (isBlocked(x-dx,y-dy))    return null;
059:     return new Point(x,y);
060:   }

068:   public void makeDungeon(int x1,int y1,int x2,int y2) {
069:       fillArea(x1,y1,x2,y2,walltile);
070:       fillArea(x1+1,y1+1,x2-1,y2-1,0);
071:       
072:     // create a central room
073:       fillArea((x1+x2)/2-RPG.d(3),(y1+y2)/2-RPG.d(3),(x1+x2)/2+RPG.d(3),(y1+y2)/2+RPG.d(3),floortile);
074:       
075:     for (int buildloop=0;buildloop<300;buildloop++) {
076:       
077:       // first choose a direction
078:       int dx=0;
079:       int dy=0;
080:       switch (RPG.d(4)) {
081:         case 1: dx=1; break;    //W
082:         case 2: dx=-1; break;    //E
083:         case 3: dy=1; break;    //N
084:         case 4: dy=-1; break;    //S
085:       }    
086:       
087:       // now find a free extension point
088:       // Point p=findFreeSquare();
089:       // p=findEdge(p.x,p.y,dx,dy);
090:       Point p=findEdgeSquare(dx,dy,0);
091:       if (p==null) continue;
092:       // advance onto blank square
093:       p.x+=dx;
094:       p.y+=dy;
095:       
096:       // choose new feature to add
097:       switch(RPG.d(18)) {
098:           case 1: case 2: case 3:
099:               makeCorridor(p.x,p.y,dx,dy);
100:               break;
101:        case 4: case 5: case 6:
102:               makeOvalRoom(p.x,p.y,dx,dy);
103:               break;
104:           case 7: case 8: case 9: case 10: case 11: case 12:
105:               makeRoom(p.x,p.y,dx,dy);
106:               break;
107:           case 13: case 14: case 15:
108:                makeChamber(p.x,p.y,dx,dy);
109:                break;
110:         case 16:
111:           makeTunnel(p.x,p.y,dx,dy);
112:           break;
113:         case 17: case 18:
114:           makeSquare(p.x,p.y,dx,dy);
115:           break;  
116:       }
117:     }
118:   }

120:   // make a dungeon door
121:   public void makeDoor(int x,int y, int dx,int dy) {
122:     switch (RPG.d(10)) {
123:         case 1: case 2:
124:             setTile(x,y,walltile);
125:             addThing(new SecretDoor(),x,y);
126:             break;
127:         case 3:
128:             setTile(x,y,floortile);
129:           addThing(new Door("portcullis",getLevel()),x,y);
130:           break;
131:         default:
132:           setTile(x,y,floortile);
133:           addThing(Door.createDoor(getLevel()),x,y);
134:           break;
135:     }

138:   private boolean makeRoom(int x,int y, int dx, int dy) {
139:        // random dimesions and offset
140:        int x1=x-RPG.d(RPG.abs(dx-1),5);
141:        int y1=y-RPG.d(RPG.abs(dy-1),5);
142:        int x2=x+RPG.d(RPG.abs(dx+1),5);
143:        int y2=y+RPG.d(RPG.abs(dy+1),5);
144:        
145:        if (((x2-x1)<3)||((y2-y1)<3)||(!isBlank(x1,y1,x2,y2))) return false;
146:     
147:     if (RPG.d(3)==1) fillArea(x1,y1,x2,y2,walltile);
148:     fillArea(x1+1,y1+1,x2-1,y2-1,floortile);
149:     
150:     // make entrance wall
151:     fillArea((dx==0)?x1:x,(dy==0)?y1:y,(dx==0)?x2:x,(dy==0)?y2:y,walltile);
152:     // make the door
153:     setTile(x,y,floortile);
154:     addThing(new Door(getLevel()),x,y);
155:     
156:     addRoomFeatures(x1+1,y1+1,x2-1,y2-1);
157:     
158:     return true;
159:   }

161:   // adds spice to room area.... assumes blank floor
162:   private void addRoomFeatures(int x1, int y1, int x2, int y2) {
163:     int w=x2-x1+1; int h=y2-y1+1;
164:     switch (RPG.d(30))  {
165:       case 1: case 2:
166:         if ((w*h)<66) for (int x=x1; x<=x2; x++) for (int y=y1;y<=y2; y++) {
167:           if (RPG.d(5)==1) addGuard(x,y);    
168:         } 
169:         if (RPG.d(20)==1) addThing(new PowderKeg(),x1+RPG.r(w),y1+RPG.r(h));
170:         break;
171: 
172:       case 3: case 4:
173:           if ((w>=5)&&(h>=5)) {
174:             if ((w%2)==1) for (int lx=1; lx<w; lx+=2) {
175:               setTile(x1+lx,y1+1,walltile);    
176:               setTile(x1+lx,y2-1,walltile);    
177:             }    
178:             if ((h%2)==1) for (int ly=1; ly<h; ly+=2) {
179:               setTile(x1+1,y1+ly,walltile);    
180:               setTile(x2-1,y1+ly,walltile);    
181:             }    
182:           }
183:           break;
184: 
185:       case 5: case 6:
186:           int planttype=RPG.d(3); // 1, 2 or 3 plant types
187:           for (int x=x1; x<=x2; x++) for (int y=y1;y<=y2; y++) {
188:             if (RPG.d(5)==1) addThing(GameScenery.create(planttype),x,y);    
189:           }
190:           break; 
191: 
192:       case 7: case 8:
193:           addThing(Lib.createItem(0),x1+RPG.r(w),y1+RPG.r(h));
194:           break;
195: 
196:       case 9: // unfilled central area
197:           fillArea(x1+1,y1+1,x2-1,y2-1,0);
198:           break;
199: 
200:       case 10: case 11: // central room
201:           if ((w>5)&&(h>5)) {
202:               fillArea(x1+1,y1+1,x2-1,y2-1,walltile);
203:               fillArea(x1+2,y1+2,x2-2,y2-2,floortile);
204:               addRoomFeatures(x1+2,y1+2,x2-2,y2-2);
205:               switch (RPG.d(4)) {
206:             case 1: makeDoor(x1+1,y1+1+RPG.d(h-4),1,0); break;             
207:             case 2: makeDoor(x2-1,y1+1+RPG.d(h-4),1,0); break;             
208:             case 3: makeDoor(x1+1+RPG.d(w-4),y1+1,0,1); break;             
209:             case 4: makeDoor(x1+1+RPG.d(w-4),y2-1,0,1); break;             
210:               }
211:           }    
212:           break;
213: 
214:       case 12: // room with runetraps;
215:           for (int x=x1; x<=x2; x++) for (int y=y1;y<=y2; y++) {
216:             if (RPG.d(4)==1) addThing(new RuneTrap(),x,y);    
217:           }
218:           break; 
219: 
220:       case 13: 
221:           addThing(Lib.createItem(0),x1+RPG.r(w),y1+RPG.r(h));
222:           addThing(new HeroTrap(),x1+RPG.r(w),y1+RPG.r(h));
223:           break;
224: 
225:       case 14: case 15: case 16: // vertically partitioned room
226:           if (w>3) {
227:             int x=x1+RPG.d(w-2);
228:             if (isBlank(x,y1-1,x,y1-1)&&isBlank(x,y2+1,x,y2+1)) {
229:               fillArea(x,y1,x,y2,walltile);    
230:                 int y=y1+RPG.r(h);
231:                 addThing(new SecretDoor(),x,y); // opens walltile
232:                 addGuard(x+1,y);
233:                 addGuard(x-1,y);
234:              }    
235:           }    
236:           break;
237:       
238:       case 17: case 18: case 19: // horizontally partitioned room
239:           if (h>3) {
240:             int y=y1+RPG.d(h-2);
241:             if (isBlank(x1-1,y,x1-1,y)&&isBlank(x2+1,y,x2+1,y)) {
242:               fillArea(x1,y,x2,y,walltile);    
243:                 int x=x1+RPG.r(w);
244:                 addThing(new SecretDoor(),x,y); // opens walltile
245:                 addGuard(x,y+1);
246:                 addGuard(x,y-1);
247:              }    
248:           }    
249:           break;
250: 
251:       case 20:
252:         addThing(new SpellBook(),x1+RPG.r(w),y1+RPG.r(h));
253:         break;    
254: 
255:       case 21:
256:         fillArea(x1+1,y1+1,x2-1,y2-1,Tile.POOL);
257:         break;
258:         
259:       case 22: case 23:
260:         addThing(Food.createFood(0),x1,y1,x2,y2);
261:         break;
262:       
263:       case 24:
264:         // traps and bones
265:         for (int x=x1; x<=x2; x++) for (int y=y1;y<=y2; y++) {
266:           if (RPG.d(4)==1) addThing(new HeroTrap(),x,y);  
267:         }
268:         addThing(new Food("corpse"),x1,y1,x2,y2);
269:         addThing(new Food("corpse"),x1,y1,x2,y2);
270:         break; 
271:         
272:         
273:       default:
274:         break;
275:     }
276:   }

278:   public void addBeasties(int x, int y) {
279:     Creature m=Lib.createMonster(getLevel());
280:     for (int i=RPG.d(Creature.numbers[m.creaturetype]); i>0; i--) {
281:       Point p=findFreeSquare(x-2,y-2,x+2,y+2);
282:       if (p!=null) {
283:         addThing(m.cloneType(),p.x,p.y);
284:       }  
285:     }
286:   }

312:   // oval room, pretty cool
313:   private void makeOvalRoom(int x,int y, int dx, int dy) {
314:     int w=RPG.d(2,3); 
315:     int h=RPG.d(2,3);
316:     int x1=x+(dx-1)*w;
317:     int y1=y+(dy-1)*h;
318:     int x2=x+(dx+1)*w;
319:     int y2=y+(dy+1)*h;
320:       
321:     if (!isBlank(x1,y1,x2,y2)) return;
322:     
323:     int cx=(x1+x2)/2;
324:     int cy=(y1+y2)/2;
325:     
326:     for (int lx=x1; lx<=(x1+w*2); lx++) for (int ly=y1; ly<(y1+h*2); ly++) {
327:       if ( (((lx-cx)*(lx-cx)*100)/(w*w) + ((ly-cy)*(ly-cy)*100)/(h*h)) < 100 )
328:              setTile(lx,ly,floortile);
329:     }    
330: 
331:     fillArea(cx,cy,x,y,floortile); 
332:   }

334:   // make a 3*3 special chamber with interesting stuff
335:   public boolean makeChamber(int x,int y, int dx, int dy) {
336:       int x1=x+(dx-1)*2;
337:       int y1=y+(dy-1)*2;
338:       int x2=x+(dx+1)*2;
339:       int y2=y+(dy+1)*2;
340:       int cx=x1+2;
341:       int cy=y1+2;
342:       
343:       if (!isBlank(x1,y1,x2,y2)) return false;
344:       
345:       fillArea(x1+1,y1+1,x2-1,y2-1,floortile);
346:       setTile(x,y,floortile);
347:       makeDoor(x,y,dx,dy);
348:       
349:       boolean continued=false;
350:       
351:       switch (RPG.d(4)) {
352:         case 1: case 2: continued=(makeRoom(cx+2*dx,cy+2*dy,dx,dy)||continued); break;
353:         case 3: continued=(makeChamber(cx+2*dx,cy+2*dy,dx,dy)||continued); break;
354:        }
355:       if (RPG.d(3)==1) continued=(makeChamber(cx+2*dy,cy+2*dx,dy,dx)||continued);
356:       if (RPG.d(3)==1) continued=(makeChamber(cx-2*dy,cy-2*dx,-dy,-dx)||continued);
357:       
358:       if (!continued) {
359:         // we have an ending chamber, so add something interesting
360:         switch (RPG.d(4)) {
361:             case 1: 
362:                 addThing(new HeroTrap(),cx,cy); break;
363:             case 2:
364:                 addThing(new RuneTrap(),cx,cy); 
365:                 addThing(new RuneTrap(),cx+dy,cy-dx); 
366:                 addThing(new RuneTrap(),cx-dy,cy+dx);
367:                 break; 
368:         }
369:         addThing(new Chest(getLevel()),cx+dx,cy+dy);    
370:       }
371:           
372:       completeArea(x1,y1,x2,y2,walltile);
373:       
374:       return true;    
375:   }

377:   // make a 5*5 special chamber with interesting stuff
378:   public boolean makeSquare(int x,int y, int dx, int dy) {
379:     int x1=x+(dx-1)*3;
380:     int y1=y+(dy-1)*3;
381:     int x2=x+(dx+1)*3;
382:     int y2=y+(dy+1)*3;
383:     int cx=x1+3;
384:     int cy=y1+3;
385:     
386:     if (!isBlank(x1,y1,x2,y2)) return false;
387:     
388:     fillArea(x1+1,y1+1,x2-1,y2-1,floortile);
389:     setTile(x,y,floortile);
390:     makeDoor(x,y,dx,dy);
391:     
392:     boolean continued=false;
393:     
394:     // room ahead
395:     switch (RPG.d(4)) {
396:       case 1: case 2: continued=(makeRoom(cx+3*dx,cy+3*dy,dx,dy)||continued); break;
397:       case 3: continued=(makeChamber(cx+3*dx,cy+3*dy,dx,dy)||continued); break;
398:      }
399:     
400:     // side rooms
401:     if (RPG.d(2)==1) {
402:       if (RPG.d(2)==1) continued=(makeChamber(cx+3*dy,cy+3*dx,dy,dx)||continued);
403:       else continued=(makeRoom(cx+3*dy,cy+3*dx,dy,dx)||continued);
404:     }
405:     if (RPG.d(2)==1) {
406:       if (RPG.d(2)==1) continued=(makeChamber(cx-3*dy,cy-3*dx,-dy,-dx)||continued);
407:       else continued=(makeChamber(cx-3*dy,cy-3*dx,-dy,-dx)||continued); 
408:     }
409:     
410:     if (continued) {
411:       switch(RPG.d(4)) {
412:         case 1: 
413:           addThing(new Chest(getLevel()),cx,cy); 
414:           addThing(new HeroTrap(),cx+dx,cy+dy);
415:           break;
416:         case 2:
417:           addBeasties(cx,cy);
418:           break;
419:         case 3:
420:           addThing(new Fire(5,0),cx+1,cy+1);
421:           addThing(new Fire(5,0),cx-1,cy+1);
422:           addThing(new Fire(5,0),cx+1,cy-1);
423:           addThing(new Fire(5,0),cx-1,cy-1);
424:           break;
425:         case 4:
426:           addRoomFeatures(x1+1,y1+1,x2-1,y2-1);
427:           break;
428:       }
429:     } else {
430:       // we have an ending chamber, so add something really interesting
431:       switch (RPG.d(1)) {
432:         case 1: 
433:           addThing(Creature.createCreature(Creature.spawn,getLevel()+3),cx,cy); 
434:           addThing(new Chest(getLevel()+3),cx+dx,cy+dy); 
435:           break;
436:       }
437:     }
438:       
439:     completeArea(x1,y1,x2,y2,walltile);
440:     
441:     return true;  
442:   }

450:   // make a random twisty tunnel
451:   public boolean makeTunnel(int x, int y, int dx, int dy) {
452:     if ((getTile(x,y)==0)&&isValid(x,y)) {
453:       int ndx=dx; int ndy=dy;
454:       setTile(x,y,floortile);
455:       if (RPG.d(3)==1) {ndx=-dy; ndy=dx;}
456:       if (RPG.d(4)==1) {ndx=dy; ndy=-dx;}
457:       makeTunnel(x+ndx,y+ndy,ndx,ndy);    
458:       return true;
459:     }    
460:     return false;
461:   }

468:   // make a long corridor
469:   public boolean makeCorridor(int x, int y, int dx, int dy) {
470:     int l=RPG.d(2,8);
471:     
472:     // don't build if there is no space
473:     if (!isBlank(x,y,x+dx*l,y+dy*l)) return false;
474:     
475:     for (int i=0;i<l;i++) {
476:          setTile(x+i*dx,y+i*dy,floortile);    
477:     }
478:      
479:     // add a door if there is space
480:     if ((l>4)&&(RPG.d(2)==1)) {
481:         if (completeTile(x+dy,y+dx,walltile)
482:            &completeTile(x-dy,y-dx,walltile)) 
483:         {
484:              makeDoor(x,y,dx,dy);
485:         }
486:     }
487:     
488:     // add a room to end
489:     if ((l>3)&&(RPG.d(2)==1)) makeRoom(x+dx*l,y+dy*l,dx,dy);
490:   
491:     if (RPG.d(100)==1) addTrap(x+RPG.r(l)*dx,y+RPG.r(l)*dy);
492:   
493:     return true;
494:   }
