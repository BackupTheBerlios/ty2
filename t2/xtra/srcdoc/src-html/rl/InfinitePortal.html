01: package rl;
02: 
03: public class InfinitePortal extends Portal {
04:   public int tolevel;
05:   
06:   public InfinitePortal(int t,int lev) {
07:     super(t);
08:     tolevel=lev;
09:     }
10:     
11:   // go through the portal - create a new dungeon as needed
12:   public void travel(Thing t) {
13:     // use as normal portal if target dungeon already created
14:     // this is important for e.g. followers
15:     if ((t!=Game.hero)&&(getDestination()!=null)) {
16:       super.travel(t);
17:       return; 
18:     }
19:     
20:     // have we esacped the infinite dungeon?
21:     if (tolevel<1) {
22:       Portal outplace=((InfiniteDungeon)getMap()).wayout;
23:       setDestination(outplace);
24:       super.travel(t);
25:       return;
26:     }
27:     
28:     // remember out portal
29:     Portal p;
30:     if (getMap() instanceof InfiniteDungeon) {
31:       p=((InfiniteDungeon)getMap()).wayout;
32:     } else {
33:       p=this; 
34:     }
35:     
36:     try {
37:       InfiniteDungeon nextlevel=new InfiniteDungeon(tolevel,getImage()==1,p);
38:       Portal targ = (getImage()==1) ? nextlevel.entrance : nextlevel.exit;
39:       setDestination(targ);
40:       super.travel(t);
41:     } catch (Exception e) {
42:       Game.message("Failed to build target InfiniteDungeon"); 
43:     }
44:   }
45: }
