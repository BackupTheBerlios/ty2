053:   public CommonArmour(int t, int m, int q) {
054:     super();
055:     type=t;

063:   public static Armour createArmour(int level) {
064:     // add some randomness to level
065:     level=level+RPG.r(3)-RPG.r(3);
066:     
067:     // decide the material
068:     int m=Material.getArmourMaterial(level);
069:     
070:     for (int i=0; i<100; i++) {
071:       int t=RPG.d(names.length-1);
072:       if (RPG.d(rarity[t])!=1) continue;
073:       
074:       int ql=Material.getLevel(m)+levels[t];     // level of average armour
075:       if ((ql+qualitylevels[3])>level) continue; // can't make it bad enough
076:       
077:       int q=3;
078:       
079:       // improve the armour if luck and level allows it
080:       //
081:       //  could use : (RPG.d(13)>q) &&   to limit qualities?
082:       while ( (ql+qualitylevels[q+1]<=level) ) {q++;}
083:       
084:       return new CommonArmour(t,m,q);
085:     }
086:     
087:     return new CommonArmour(1,Material.IRON,4);
088:   }

097:   public int getStat(int s) {
098:     switch(s) {
099:       case RPG.ST_DSKMULTIPLIER: 
100:         return defencemuls[type];
101:       
102:       case RPG.ST_DSKBONUS: 
103:         return defencebonuses[type];
104:       
105:       case RPG.ST_ITEMVALUE: {
106:         int val=Material.getValue(material,sizes[type]);
107:         val=val*relativevalues[type]/100;
108:         return RPG.niceNumber(val*Lib.qualityvalues[quality]/100);
109:       }
110:       
111:       default: return super.getStat(s);
112:     }    
113:   }

115:   public int getArmour() {
116:     return (basearmour[type]*qualitymuls[getEffectiveQuality()]*Material.getArmourStrength(material))/10000;
117:   }

119:   public int getModifier(int s) {
120:     
121:     switch(s) {
122:       
123:       case RPG.ST_ARMUNARMED: 
124:         return getArmour()*2;
125:       case RPG.ST_ARMICE: 
126:       case RPG.ST_ARMFIRE:
127:       case RPG.ST_ARMNORMAL:  
128:       case RPG.ST_ARMIMPACT:  
129:         return getArmour();
130:       case RPG.ST_ARMPIERCING:
131:       case RPG.ST_ARMMAGIC:
132:         return getArmour()/2;
133:       
134:       case RPG.ST_ENCUMBERANCE:
135:         return encumberance[type];
136: 
137:       case RPG.ST_DSKBONUS:
138:         return defencebonuses[type]*qualitymuls[getEffectiveQuality()]/100;
139: 
140:       case RPG.ST_DSKMULTIPLIER: 
141:         return defencemuls[type]*qualitymuls[getEffectiveQuality()]/100;
142:       
143:       default: {
144:         return super.getModifier(s);
145:       }
146:     }  
147:   }

149:   // return toughness for the armour
150:   // increases with higher quality and better materials
151:   public int getToughness() {
152:     if ((quality>=13)||isArtifact()) return 0;
153:     
154:     // base toughness for average thing
155:     int tuff=100*basearmour[type];
156:     
157:     // rusty items are much weaker!
158:     if (isRusty()) tuff=tuff/4;
159:     
160:     // modify for quality
161:     tuff=tuff*quality/5;
162:     
163:     // modify for type and material
164:     tuff = ((tuff)*Material.getToughness(material))/100;
165:     
166:     return tuff; 
167:   }
168:   
169:   // damage routine causes armour to deteriorate
170:   public int damage(int dam, int damtype) {
171:     int tuff=getToughness();
172:     if (tuff<=0) return 0;
173:     
174:     switch (damtype) {
175:       case RPG.DT_ACID:
176:         dam=dam*10; break;
177:     } 
178:     
179:     int det=RPG.po(dam,tuff);
180:     if ((quality<13)&&(det>0)) {
181:       if (isDamaged()) {
182:         quality-=det;
183:       } else {
184:         setDamaged(true);
185:       }
186:       if (quality>0) {
187:         if (place==Game.hero) {
188:           Game.message("Your "+getName()+" is damaged!"); 
189:         }
190:       } else {
191:         die(); 
192:       }
193:     }
194:     return dam; 
195:   }
