010:   public static String roman(int n) {
011:     String r="";
012:     switch (n/1000) {
013:       case 1: r=r+ "M"   ; break;
014:       case 2: r=r+ "MM"  ; break;
015:       case 3: r=r+ "MMM" ; break;
016:     }
017:     n=n%1000;
018:     
019:     switch (n/100) {
020:       case 1: r=r+ "C"        ; break;
021:       case 2: r=r+ "CC"       ; break;
022:       case 3: r=r+ "CCC"      ; break;
023:       case 4: r=r+   "CD"     ; break;
024:       case 5: r=r+    "D"     ; break;
025:       case 6: r=r+    "DC"    ; break;
026:       case 7: r=r+    "DCC"   ; break;
027:       case 8: r=r+    "DCCC"  ; break;
028:       case 9: r=r+       "CM" ; break;
029:     }
030:     n=n%100;
031: 
032:     switch (n/10) {
033:       case 1: r=r+ "X"        ; break;
034:       case 2: r=r+ "XX"       ; break;
035:       case 3: r=r+ "XXX"      ; break;
036:       case 4: r=r+   "XL"     ; break;
037:       case 5: r=r+    "L"     ; break;
038:       case 6: r=r+    "LX"    ; break;

040:       case 8: r=r+    "LXXX"  ; break;
041:       case 9: r=r+       "XC" ; break;
042:     }
043:     n=n%10;

046:       case 1: r=r+ "I"        ; break;

048:       case 3: r=r+ "III"      ; break;

061:     // return index of string s in array ss
062:   public static int index(String s, String[] ss) {
063:     for (int i=0; i<ss.length; i++) {
064:       if (s.equals(ss[i])) return i;
065:     }
066:     return -1;

077:     public static String centrePad(String a, String b, int len) {
078:       len=len-a.length();

083:     public static String capitalise(String s) {
084:         char c=s.charAt(0);
085:         if (Character.isUpperCase(c)) return s;
086:         StringBuffer sb=new StringBuffer(s);
087:         sb.setCharAt(0,Character.toUpperCase(c));
088:         return sb.toString();
089:     }

091:     public static int countChar(String s, char c) {
092:       int count=0;
093:       for (int i=0; i<s.length(); i++) {
094:           if (s.charAt(i)==c) count++;
095:       }    
096:       return count;
097:     }
098:     
099:     public static int wrapLength(String s, int start, int len) {
100:         if ((s.length()-start)<=len) return (s.length()-start);
101:         
102:         for (int i=len; i>=0; i--) {
103:             if (Character.isWhitespace(s.charAt(i+start))) return i;
104:         }
105:         
106:         return len;
107:     }
108:     
109:     public static String[] wrapString(String s, int len) {
110:         String[] working = new String[1+((2*s.length())/len)];
111:         
112:         int end=s.length();
113:         int pos=0;
114:         int i=0;
115:       while (pos<end) {
116:             int inc=wrapLength(s,pos,len);
117:             working[i]=s.substring(pos,pos+inc);
118:             i++;
119:       pos=pos+inc;
120:       while ((pos<end)&&Character.isWhitespace(s.charAt(pos))) pos++;
121:         };
122:         
123:         // return empty string if size zero
124:         if (i==0) {
125:           i=1;
126:           working[0]="";     
127:         }
128:         
129:         String[] result = new String[i];
130:         System.arraycopy(working,0,result,0,i);
131:         return result;
132:     }

134:     public static int pond(String s) {
135:     // mildly tricky encryption hash
136:     // can't be bothered with anything tougher yet
137:     int len=s.length();
138:     
139:     int a=0;
140:     int f=1;
141:     for (int i=len-1; i>=0; i--) {
142:       a+=((int)s.charAt(i))*f;
143:       f*=31; 
144:     }
145:     
146:     return a;

149:   public static String[] separateString(String s, char c) {
150:       int num=countChar(s,c);
151:       int start=0;
152:       int finish=0;
153:       String[] result=new String[num+1];
154:       for (int i=0; i<(num+1); i++) {
155:         finish=s.indexOf(c,start);
156:         if (finish<start) finish=s.length();
157:         if (start<finish) {
158:             result[i]=s.substring(start,finish);    
159:         } else {
160:           result[i]="";    
161:         }
162:           start=finish+1;
163:       }    
164:         return result;
165:     }
