077:   // create scroll of approximate level value
078:   public static Stack createScroll(int level) {
079:     if (RPG.d(2)==1) {
080:       // create from common scroll types
081:       level=level+RPG.d(5)-RPG.d(5);
082:       level=RPG.middle(0,level,spells.length-1);
083:       return new Scroll(spells[level][RPG.r(spells[level].length)],1+RPG.r(2)*RPG.r(4));
084:     } else {
085:       // let's have a random spell!!
086:       // probably better, so give less on average
087:       return new Scroll(Spell.randomSpell(level+RPG.r(6)),1+RPG.r(2)*RPG.r(3)); 
088:     }
089:   }
090:   
091:   public Scroll(int t, int num) {
092:     super(num);
093:     type=t;
094:     spell=new Spell(t);
095:   }
096:     
097:   public int damage(int dam, int damtype) {
098:         switch(damtype) {
099:           case RPG.DT_FIRE: case RPG.DT_DISPELL: case RPG.DT_ACID:
100:               dam=dam*5;
101:           case RPG.DT_NORMAL: case RPG.DT_PIERCING: case RPG.DT_SPECIAL:
102:               break;
103:       case RPG.DT_ICE: case RPG.DT_IMPACT: case RPG.DT_UNARMED:
104:           dam=dam/2;
105:           break;
106:           default:    
107:               dam=0;
108:         }
109:         
110:         if (!RPG.test(dam,3)) return 0;
111:        ThingOwner p=place;
112:        remove(1);
113:       if (p instanceof Map) {
114:            spell.castAtLocation(this,(Map)p,x,y);    
115:       }
116:       return dam;
117:     }
118:     
119:   public int getUse() {return Thing.USE_NORMAL|Thing.USE_READ;}
120:   
121:   public int getWeight() {return 200;}
122:     
123:   public int getStat(int s) {
124:     switch(s) {
125:       case RPG.ST_ITEMVALUE: return isIdentified()?(100*spell.level*spell.level):600;
126:       default: return super.getStat(s); 
127:     } 
128:   }
129:   
130:     public Spell getSpell() {return spell;}
131:     
132:     public void use(Being user) {
133:       use(user, USE_READ);
134:     }
135:     
136:     public void use(Being user, int usetype) {
137:       if (usetype!=USE_READ) return;
138:       
139:     boolean ishero=(user==Game.hero);
140:     if (!ishero) return;
141:     
142:     if (user.getStat(RPG.ST_LITERACY)>0) {
143:       identify(1000);
144:       QuestApp.questapp.screen.castSpell(spell); 
145:       remove(1);
146:     } else {
147:       Game.message("You can't read!");    
148:     }
149: 
150:     }
