057:   // use long type for calculation in case of overflow  
058:   public static int getValue(int type, int quantity) {
059:     return (int) (((long)(values[type])*quantity)/1000); 
060:   }

062:   // modify item flags for material
063:   public static int modifyFlags(int material, int flags) {
064:     switch (material) {
065:       case SAPIENTPEARWOOD: case SILVER: case GOLD: {
066:         flags|=Item.ITEM_MAGIC;
067:         break; 
068:       }
069:       case BLACKSTEEL: {
070:         flags|=Item.ITEM_CURSED; 
071:         break;
072:       }
073:       case REDSTEEL: {
074:         flags|=Item.ITEM_RUSTY; 
075:         break;
076:       }
077:       case EMERALD: {
078:         flags|=Item.ITEM_ARTIFACT; 
079:         break;
080:       }
081:     } 
082:     return flags;
083:   }
084:   
085:   // return a material appropriate for given level or below
086:   public static int getWeaponMaterial(int level) {
087:     for (int i=1; i<100; i++) {
088:       int r=RPG.d(numbers.length-1);
089:       if (((levels[r])<=level)) {
090:         // check if this is a valid weapon material
091:         if (weaponrarity[r]==0) continue;
092:         
093:         // randomise away from rare materials
094:         if (RPG.d(weaponrarity[r])!=1) continue;
095:         
096:         return r; 
097:       }
098:     }  
099:     return STEEL;
100:   }
101:   
102:   // return a material appropriate for given level or below
103:   public static int getArmourMaterial(int level) {
104:     for (int i=1; i<100; i++) {
105:       int r=RPG.d(numbers.length-1);
106:       if (((levels[r])<=level)) {
107:         // check if this is a valid weapon material
108:         if (armourrarity[r]==0) continue;
109:         
110:         // randomise away from rare materials
111:         if (RPG.d(armourrarity[r])!=1) continue;
112:         
113:         return r; 
114:       }
115:     }  
116:     return STEEL;
117:   }

119:   // can a character identify this material? 
120:   public static boolean canIdentifyMaterial(Thing person, int mat) {
121:     if (person.getStat(RPG.ST_SMITHING)>0) {
122:       if (levels[mat]<10) return true;
123:       if (person.getStat(RPG.ST_SMITHING)>1) return true;
124:     }
125:     return false; 
126:   }
