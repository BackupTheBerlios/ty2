001: package rl;
002: 
003: public class Describer implements Description, java.io.Serializable {
004:   // Description variables
005:   
006:   protected String description;
007:   protected String name;
008:   protected String pluralname;
009:   protected int nametype;
010:   protected int gender;
011:     
012:     public Describer(String n,String pn,String d,int nt,int gen) {
013:         name=n;
014:         pluralname=pn;
015:         description=d;
016:         nametype=nt;
017:         gender=gen;
018:     }
019:     
020:     public Describer(String n,String d) {
021:         this(n,null,d,NAMETYPE_NORMAL,GENDER_NEUTER);
022:     }
023: 
024:     public Describer(String n,String pn,String d) {
025:         this(n,pn,d,NAMETYPE_NORMAL,GENDER_NEUTER);
026:     }
027: 
028:   public Describer(String n,String pn, int nt,String d) {
029:     this(n,pn,d,nt,GENDER_NEUTER);
030:   }
031: 
032:     public String getName(int number, int article) {
033:         switch(article) {
034:             case ARTICLE_NONE:
035:                 return getName(number);
036:             default:
037:                 String s=getArticle(number,(article==ARTICLE_DEFINITE));
038:                 return (s==null) ? getName(number) : s+" "+getName(number);
039:         }
040:     }
041: 
042:     public String getName() {return getName(NUMBER_SINGULAR);}
043:     
044:     public String getName(int number) {
045:         if (number==NUMBER_SINGULAR) {
046:             return name;
047:         } else {
048:             return (pluralname!=null) ? pluralname : (name+"s");
049:         }
050:     }
051:     
052:     public String getDescriptionText() {
053:       return (description==null) ? (Text.capitalise(name)+".") : description;    
054:     }
055:     
056:     public String getArticle(int number, boolean definite) {
057:         switch (nametype) {
058:           case NAMETYPE_PROPER:
059:               return null;    
060:             case NAMETYPE_NORMAL:
061:                 if (number==NUMBER_SINGULAR) return definite?"the": (Text.isVowel(name.charAt(0))?"an":"a");
062:                 // otherwise continue to next case
063:             case NAMETYPE_QUANTITY:
064:                 return definite?"the":"some";
065:         }
066:         return "*article error*";
067:     }
068:     
069:     // static function to return normal articles. Useful for other Description classes
070:     public static String getArticleName(String s, int number, int article) {
071:         if (article==ARTICLE_NONE) return s;
072:         if (number==NUMBER_SINGULAR) return (article==ARTICLE_DEFINITE) ? "the "+s : (Text.isVowel(s.charAt(0))?"an ":"a ")+s;
073:       // otherwise we have a normal plural
074:          return (article==ARTICLE_DEFINITE)?"the "+s:"some "+s;
075:     }
076:     
077:   public String getPronoun(int number, int acase) {
078:     return getPronoun(number,acase,nametype,gender); 
079:   }
080:   
081:     public static String getPronoun(int number, int acase, int nametype, int gender) {
082:         if (number==NUMBER_SINGULAR) {
083:             if ((nametype==NAMETYPE_PROPER)&&(gender!=GENDER_NEUTER)) {
084:             switch (acase) {
085:               case CASE_NOMINATIVE: return (gender==GENDER_MALE)?"he":"she";
086:               case CASE_ACCUSATIVE: return (gender==GENDER_MALE)?"him":"her";
087:               case CASE_GENITIVE: return (gender==GENDER_MALE)?"his":"her";
088:             } 
089:             } else {
090:             switch (acase) {
091:               case CASE_NOMINATIVE: return "it";
092:               case CASE_ACCUSATIVE: return "it";
093:               case CASE_GENITIVE: return "its";
094:             } 
095:             }
096:         } else {
097:             if (nametype==NAMETYPE_QUANTITY) {
098:             switch (acase) {
099:               case CASE_NOMINATIVE: return "it";
100:               case CASE_ACCUSATIVE: return "it";
101:               case CASE_GENITIVE: return "its";
102:             } 
103:             } else {
104:             switch (acase) {
105:               case CASE_NOMINATIVE: return "they";
106:               case CASE_ACCUSATIVE: return "them";
107:               case CASE_GENITIVE: return "their";
108:             } 
109:             }
110:         }
111:         return "*Pronoun Error*";
112:     }
113:     
114: }
