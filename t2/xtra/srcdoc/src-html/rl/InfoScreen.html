01: package rl;
02: 
03: import java.awt.*;
04: import java.awt.event.*;
05: 
06: public class InfoScreen extends Screen {
07:     public String text;
08:     public Font font;
09:     
10:     private int border=20;
11:     
12:     public InfoScreen(String t) {
13:     text=t;
14:     font=QuestApp.mainfont;
15:     setForeground(QuestApp.infotextcolor);  
16:     setBackground(QuestApp.infoscreencolor);
17:     setFont(font);
18:   }
19:   
20:     public void activate() {
21:         // questapp.setKeyHandler(keyhandler);
22:     }
23:     
24:   public void paint(Graphics g) {
25:     FontMetrics met=g.getFontMetrics(g.getFont());
26:     
27:     Rectangle r=getBounds();
28:     int charsize=met.charWidth(' ');
29:     
30:     int linelength; int y; int lineinc;
31:     {
32:       linelength = (r.width-border*2)/charsize;
33:       y=border+met.getMaxAscent();
34:       lineinc=met.getMaxAscent()+met.getMaxDescent();
35:     }
36:             
37:       String[] st=Text.separateString(text,'\n');
38:         
39:       for (int i=0; i<st.length ;i++) { 
40:       String s=st[i];      
41:       
42:       String[] lines=Text.wrapString(s,linelength);
43:       
44:       for (int l=0; l<lines.length ; l++) {
45:           String line=lines[l];
46:           
47:           g.drawString(line,border,y);
48:         y+=lineinc;
49:       }
50:         }
51:     }
52: }
