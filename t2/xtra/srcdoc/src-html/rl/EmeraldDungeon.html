05: class EmeraldDungeon extends Dungeon {
06:   
07:   public EmeraldDungeon(int w, int h) {
08:       super(w,h,3);
09:   
10:     // now add the Emerald Sword
11:     Point es=findFreeSquare();
12:     {
13:         for (int dx=-1; dx<=1; dx++) for (int dy=-1; dy<=1; dy++) {
14:           if (!isBlocked(es.x+dx,es.y+dy)) addThing(new MapStateTrap(1),es.x+dx,es.y+dy);    
15:         }
16:       Portal dp=new Portal("stairs down");
17:       addThing(dp,es.x,es.y);
18:       dp.makeLink(new GuardianVault(new Creature(Creature.SHAMAN),StandardWeapon.createSpecificWeapon(StandardWeapon.EMERALDSWORD)));
19:     }
20:   }
21: 
22:   
23:   private void addGuard(int x, int y) {
24:     if (isClear(x,y)) {
25:         Mobile m=Lib.createCreature(3);
26:         addThing(m,x,y);
27:     }    
28:   }
29: 
30:   protected void addWandering() {
31:     Point p=findFreeSquare();
32:     int x=p.x; int y=p.y;
33:     if (!isVisible(x,y)) {
34:       if (RPG.d(3)==1) {
35:         Mobile m=Lib.createCreature(4);
36:         if (m!=null) addThing(m,x,y);    
37:       }
38:     }    
39:   }
40: 
41:   private void addBaddie(int x, int y) {
42:     if (isClear(x,y)) {
43:         Mobile m=Lib.createCreature(3);
44:         m.addThing(Lib.createItem(0));
45:         addThing(m,x,y);
46:     }    
47:   }
48:   
49:   public void action(int time) {
50:     super.action(time);
51:     if (mapstate>0) {
52:       int count=RPG.po(time,300);
53:       for (int i=0; i<count; i++) addWandering();  
54:     }
55:   }
56: 
57:   public String getEnterMessage() {
58:     return "There's something scary about this place.";  
59:   }
60:   
61:   public String getLevelName() {
62:     return "Old ruin";  
63:   }
