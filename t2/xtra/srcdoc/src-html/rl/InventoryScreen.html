007: public class InventoryScreen extends Screen {
008:   SelectCommand selectcommand;
009:     String title="List";
010:     Thing[] things=null;
011:     static int page=0;
012:     
013:     private static final int pagesize=12;
014:     private static final int lineheight=20;
015:     
016:     KeyAdapter keyhandler = new KeyAdapter () {
017:         public void keyPressed(KeyEvent k) {
018:           if (k.getKeyCode()==k.VK_ESCAPE) select(null);
019:           if (k.getKeyCode()==k.VK_SPACE) select(null);
020:           
021:           char c=Character.toLowerCase(k.getKeyChar());
022: 
023:           if ((c=='p'||c=='-')&&(page>0)) {
024:             page--;
025:             repaint();    
026:             return;    
027:           }
028: 
029:           if ((c=='n'||c=='+')&&(((page+1)*pagesize)<things.length)) {
030:             page++;
031:             repaint();
032:             return;    
033:           }
034: 
035:           int kv = (int)c - (int)'a';
036:           if ((kv<0)||(kv>=pagesize)) return;
037:           kv=kv+page*pagesize;
038:           if ((kv>=0)&&(kv<things.length)) select(things[kv]);
039:         }
040:     };
041:     
042:   public Object getObject() {
043:     boolean end=false;
044:     while (!end) {
045:       KeyEvent k=Game.getInput();
046:       if (k!=null) {
047:         if (k.getKeyCode()==k.VK_ESCAPE) return null;
048:       
049:         char c=Character.toLowerCase(k.getKeyChar());
050: 
051:         if ((c=='p'||c=='-')&&(page>0)) {
052:           page--;
053:           repaint();  
054:         }
055: 
056:         if ((c=='n'||c=='+'||c==' ')&&(((page+1)*pagesize)<things.length)) {
057:           page++;
058:           repaint();
059:         }
060: 
061:         int kv = (int)c - (int)'a';
062:         if ((kv>=0)&&(kv<pagesize)) {
063:           kv=kv+page*pagesize;
064:           if ((kv>=0)&&(kv<things.length)) return things[kv];
065:         }
066:       }
067:     }
068:     return null;    
069:   }
070:   
071:     public void select(Object ob) {
072:     getParent().remove(this);
073:       selectcommand.select(ob);    
074:     }

109:     public void paint(Graphics g) {
110:         super.paint(g);
111:         
112:         if ((page*pagesize)>things.length) page=0;
113:         
114:         g.drawString(title,20,1*lineheight);
115:         
116:       for(int i=0; i<Math.min(things.length-(page*pagesize),pagesize); i++) {
117:             Thing t=things[i+page*pagesize];
118:             String s="["+(char)((int)'a'+i)+"] "+t.getFullName();
119:           String ws=Lib.wieldDescription(t.y);
120:           if (ws!=null) ws="("+ws+")"; else ws="";
121:           s=Text.centrePad(s,ws,50);
122:           s=s+Text.centrePad("  ",t.getWeight()/100+"s",7);
123:             g.drawString(s,50,lineheight*(i+2));
124:       if (t instanceof Item) {
125:         int image=t.getImage();
126:         int sx=(image%20)*MapPanel.TILEWIDTH;
127:         int sy=(image/20)*MapPanel.TILEHEIGHT;
128:         int px=30;
129:         int py=lineheight*(i+1)+8;    
130:         g.drawImage(QuestApp.items,px,py,px+MapPanel.TILEWIDTH,py+MapPanel.TILEHEIGHT,sx,sy,sx+MapPanel.TILEWIDTH,sy+MapPanel.TILEHEIGHT,null);
131:       }
132:         }
133:         
134:         String bottomstring="ESC=Cancel ";
135:          if (page>0) bottomstring=bottomstring+" P=Previous ";
136:          if (((page+1)*pagesize)<things.length) bottomstring=bottomstring+" N=Next ";
137:         String weightstring="Weight: "+Game.hero.inv.getWeight()/100+" / "+Game.hero.getStat(RPG.ST_MAXWEIGHT)/100+"s";
138:         g.drawString(Text.centrePad(bottomstring,weightstring,60),20,getSize().height-10);
139:     }
