021:   private String statString(int s) {
022:     int bs=character.getBaseStat(s);
023:     int ms=character.getStat(s)-bs;
024:     return Text.centrePad(new Integer(bs).toString(),((ms>=0)?"(+"+ms:"("+ms)+")",8);
025:   }
026:     
027:     public void paint(Graphics g) {
028:         if (character==Game.hero) {
029:           g.drawString(Game.hero.getName(),40,20);
030:           g.drawString(Game.hero.race+" "+Game.hero.profession,240,20);
031:         } else {
032:           g.drawString(character.getName(),40,20);
033:         }
034:         
035:         {
036:          int image=character.getImage();
037:       int sx=(image%20)*MapPanel.TILEWIDTH;
038:       int sy=(image/20)*MapPanel.TILEHEIGHT;
039:       int px=16;
040:       int py=8;    
041:         g.drawImage(QuestApp.creatures,px,py,px+MapPanel.TILEWIDTH,py+MapPanel.TILEHEIGHT,sx,sy,sx+MapPanel.TILEWIDTH,sy+MapPanel.TILEHEIGHT,null);
042:         }
043:         
044:     g.drawString("SK: "+statString(RPG.ST_SK),40,40);
045:     g.drawString("ST: "+statString(RPG.ST_ST),40,40+1*16);

052:     g.drawString("(a) Artisan : "+character.getBaseStat(RPG.ST_ARTISAN),240,40+7*16);
053:     g.drawString("(b) Bard    : "+character.getBaseStat(RPG.ST_BARD),240,40+6*16);
054:     g.drawString("(f) Fighter : "+character.getBaseStat(RPG.ST_FIGHTER),240,40+1*16);
055:     g.drawString("(r) Ranger  : "+character.getBaseStat(RPG.ST_RANGER),240,40);
056:     g.drawString("(m) Mage    : "+character.getBaseStat(RPG.ST_MAGE),240,40+5*16);

067:     g.drawString("Current Exp.  : "+character.getStat(RPG.ST_EXP),  40,40+15*16);
068: //    g.drawString("Hunger level  : "+character.getStat(RPG.ST_HUNGER)/100,  40,40+16*16);
069:     g.drawString("Hunger level  : "+RPG.percentile(character.getStat(RPG.ST_HUNGER), Hero.HUNGERLEVEL)+"%",  40,40+16*16);
070:     
071:     Thing weapon=character.getWielded(RPG.WT_MAINHAND);
072:     Weapon w=null;
073:     if (weapon instanceof Weapon) {
074:       g.drawString("Wielding "+weapon.getName(),240,40+9*16);
075:       w=(Weapon)weapon;
076:     } else {
077:       g.drawString("Unarmed",240,40+9*16);
078:       w=Lib.WEAPON_MAUL;
079:     }
080:     
081:     g.drawString("Attack Cost     : "+w.getStat(RPG.ST_ATTACKCOST),240,40+10*16);
082:     g.drawString("Attack Skill    : "+w.getASK(character,null),240,40+11*16);
083:     g.drawString("Attack Strength : "+w.getAST(character,null),240,40+12*16);
084: 
085:     g.drawString("Defence Skill   : "+character.getStat(RPG.ST_DSK),240,40+13*16);
086: 
087:     g.drawString("Armour Class    : "+character.getStat(RPG.ST_ARMNORMAL),240,40+14*16);
088:     g.drawString("Encumberance    : "+RPG.middle(0,character.getStat(RPG.ST_ENCUMBERANCE),100)+"%",240,40+15*16);
089:     g.drawString("Wealth          : "+Integer.toString(character.getMoney()),240,40+16*16);
090:     
091:         String bottomstring="ESC=Exit";
092:         g.drawString(bottomstring,20,getSize().height-10);
093:     }
