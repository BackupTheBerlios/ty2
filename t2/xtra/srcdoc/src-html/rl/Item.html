001: // Item is the base class for all in-game objects
002: // Can be picked up and carried by Mobiles
003: 
004: package rl;
005: 
006: public class Item extends Thing {
007:     // 0=identified, 1=unidentified
008:   protected static final int ITEM_IDENTIFIED=1;
009:   protected static final int ITEM_OWNED=2;
010:   protected static final int ITEM_FORSALE=4;
011:   protected static final int ITEM_CURSED=8;
012:   protected static final int ITEM_BLESSED=16;
013:   protected static final int ITEM_ARTIFACT=32;
014:   protected static final int ITEM_DAMAGED=64;
015:   protected static final int ITEM_MAGIC=128;
016:   protected static final int ITEM_BROKEN=256;
017:   protected static final int ITEM_RUSTY=512;
018:   protected static final int ITEM_CHARMED=1024;
019:   protected static final int ITEM_ID_ATTEMPTED=2048;
020:   
021:   protected int flags=0;
022:   protected int image=0;
023:   protected int quality=0;
024:   
025:   // queries whether an item is identified
026:   // if so, the player will see the item's full (true) description
027:   public boolean isIdentified() {
028:     if (QuestApp.debug&&Game.hero.profession.equals("avatar")) return true;
029:     
030:     if ((flags&ITEM_ID_ATTEMPTED)==0) tryIdentify(Game.hero);
031:     
032:     return ((flags&ITEM_IDENTIFIED)!=0);
033:   }
034:   
035:   public boolean tryIdentify(Being viewer) {
036:     flags|=ITEM_ID_ATTEMPTED;
037:     if (RPG.test(viewer.getStat(RPG.ST_IDENTIFY)*viewer.getStat(RPG.ST_CR),40+5*getLevel()+5*getStat(RPG.ST_ITEMRARITY))) {
038:       flags|=ITEM_IDENTIFIED;
039:       return true;
040:     } 
041:     return false;
042:   }
043:   
044:   public void setIdentified(boolean ident) {
045:     if (ident) {
046:       flags |= ITEM_IDENTIFIED;
047:     } else {
048:       flags &= ~ITEM_IDENTIFIED; 
049:     } 
050:   }
051:   
052:   public void setBlessed(boolean bless) {
053:     if (bless) {
054:       flags |= ITEM_BLESSED;
055:     } else {
056:       flags &= ~ITEM_BLESSED; 
057:     } 
058:   }
059:   
060:   public void setDamaged(boolean damage) {
061:     if (damage) {
062:       flags |= ITEM_DAMAGED;
063:     } else {
064:       flags &= ~ITEM_DAMAGED; 
065:     } 
066:   }
067:   
068:   
069:   public boolean isOwned() {
070:     return (flags&ITEM_OWNED)!=0;
071:   }
072:   
073:   public boolean isArtifact() {
074:     return (flags&ITEM_ARTIFACT)!=0; 
075:   }
076:   
077:   public boolean isDamaged() {
078:     return (flags&ITEM_DAMAGED)!=0; 
079:   }
080:   
081:   public boolean isRusty() {
082:     return (flags&ITEM_RUSTY)!=0; 
083:   }
084:   
085:   // what is the item made of?
086:   // no default material
087:   public int getMaterial() {
088:     return 0; 
089:   }
090:   
091:   // returns the toughness of the item
092:   // this the the average amount of damage needed
093:   // to cause the item to deteriorate
094:   // 
095:   // for indestructible items, return zero
096:   public int getToughness() {
097:     return 0; 
098:   }
099:   
100:   public boolean identify(int idskill) {
101:       if (!isIdentified()) {
102:           if (idskill>=getStat(RPG.ST_ITEMRARITY)) {
103:             flags=flags|ITEM_IDENTIFIED;
104:             return true;    
105:           } else {
106:         return false;              
107:           }     
108:       }    
109:       return true;
110:     }
111:     
112:   public void deteriorate(int levels) {
113:     // can't deteriorate artifacts
114:     if (isArtifact()) return;
115:     
116:     // repeat for each stage of damage
117:     for (int i=levels; i>0; i--) {
118:       // destroy item if already damaged
119:       if ((flags&ITEM_DAMAGED)>0) {
120:         die();
121:         return; 
122:       }
123:       
124:       // if charmed, remove charm else set as damaged
125:       if ((flags&ITEM_CHARMED)>0) {
126:         flags&=~ITEM_CHARMED;
127:       } else {
128:         if ((i==1)&&(place==Game.hero)) Game.message("Your "+getName()+" is damaged");
129:         flags|=ITEM_DAMAGED; 
130:       }
131:     } 
132:   }
133:   
134:   public int damage(int dam, int damtype) {
135:     dam=Lib.physicalDamage(dam,damtype);    
136:     
137:     // can't destroy artifacts etc...
138:     if (getQuality()>=13) return 0;
139:     
140:     int t=getToughness();
141:     int d=RPG.po(dam,t);
142:     deteriorate(d);
143:     return (d>0)?dam:0; 
144:   }
145:   
146:   public void die() {
147:     if (place==Game.hero) {
148:       Game.message("Your "+getFullName()+" is destroyed!");
149:     } else {
150:       visibleMessage(getTheName()+" is destroyed"); 
151:     }
152:     super.die();
153:   }
154:   
155:   
156:   public String getAdjectives() {
157:     Hero h=Game.hero;
158:     boolean isidentified=isIdentified();
159:     int q=getQuality();
160:     
161:     String n=super.getAdjectives();
162:     
163:     if ((q>0)) {
164:       if (isidentified||(h.getStat(RPG.ST_APPRAISAL)>=3)) {
165:         n=Lib.qualitystrings[q]+" "+n; 
166:       } else if (h.getStat(RPG.ST_APPRAISAL)>=1) {
167:         switch (q) {
168:           case 1: case 2: case 3: case 4: n="dodgy "+n; break;
169:           case 5: case 6: case 7: n="reasonable "+n; break;
170:           case 8: case 9: case 10: case 11: case 12: case 13: n="nice "+n; break;
171:         }
172:       }
173:     } 
174:     
175:     if ( ((flags&ITEM_CURSED)>0) && (isidentified||(h.getStat(RPG.ST_PRIEST)>0)) ) {
176:       n="cursed "+n; 
177:     }
178: 
179:     if ( ((flags&ITEM_BLESSED)>0) && (isidentified||(h.getStat(RPG.ST_PRIEST)>0)) ) {
180:       n="blessed "+n; 
181:     }
182: 
183:     if ( (flags&ITEM_DAMAGED)>0) {
184:       n="damaged "+n; 
185:     }
186: 
187:     if ( ((flags&ITEM_MAGIC)>0) && (isidentified||(h.getStat(RPG.ST_MAGE)>0)) ) {
188:       n="magic "+n; 
189:     }
190: 
191:     if ( ((flags&ITEM_CHARMED)>0) && (isidentified||(h.getStat(RPG.ST_MAGE)>0)) ) {
192:       n="charmed "+n; 
193:     }
194: 
195:     return n;
196:   }
197:   
198:   public int getQuality() {return quality;}
199:   
200:   // modify effective quality
201:   public int getEffectiveQuality() {
202:     int q=getQuality();
203:     if ((flags&ITEM_CURSED)>0) q=q-2;
204:     if ((flags&ITEM_BLESSED)>0) q=q+1;
205:     if ((flags&ITEM_MAGIC)>0) q=q+1;
206:     if ((flags&ITEM_CHARMED)>0) q=q+1;
207: 
208:     if ((flags&ITEM_DAMAGED)>0) q=q-1;
209:     if ((flags&ITEM_RUSTY)>0) q=q-2;
210:     if ((flags&ITEM_BROKEN)>0) q=0;
211:     return RPG.middle(0,q,13);    
212:   }
213:   
214:   public int getWeight() {return 0;}
215:   
216:   public int getZ() {return Z_ITEM;}
217: 
218:   public int getImage() {return image;}
219:   
220:   public int wieldType() {return RPG.WT_NONE;}
221: 
222:   public int getModifier(int stat) {return 0;}  
223: }
