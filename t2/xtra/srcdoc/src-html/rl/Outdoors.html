009: public class Outdoors extends Map {
010:   
011:   public Outdoors (int t) {
012:     super(65,65);
013:     buildOutdoors(this,0,0,width-1,height-1,t);
014:     
015:     entrance=new EdgePortal(EdgePortal.ZONE);
016:     addThing(entrance,32,32);
017:   }
018:   
019:   public static void buildOutdoors(Map map, int x1, int y1, int x2, int y2, int t) {  
020:     int w=x2-x1+1;
021:     int h=y2-y1+1;
022:     int area=w*h;
023: 
024:     switch (t) {
025:       case Tile.PLAINS: {
026:         map.setTheme("plains");
027:         map.fillArea(x1,y1,x2,y2,map.floortile); 
028:         for (int i=0; i<(area/50); i++) {
029:           Point ts=map.findFreeSquare();
030:           map.addThing(GameScenery.create(1),ts.x,ts.y);
031:         }
032:         for (int i=0; i<(area/400); i++) {
033:           map.addThing(new Missile(Missile.STONE,1),x1,y1,x2,y2);
034:         }
035:         Point ts=map.findFreeSquare(x1,y1,x2,y2);
036:         map.addThing(GameScenery.create(1),ts.x,ts.y);
037:         map.addThing(new SecretItem(Lib.createItem(4)),ts.x,ts.y);
038:         break;
039:       }
040:          
041:       case Tile.HILLS: {
042:         map.setTheme("plains");
043:         map.fillArea(x1,y1,x2,y2,map.floortile); 
044:         for (int i=0; i<(area/10); i++) {
045:           map.setTile(x1+RPG.r(w),y1+RPG.r(h),Tile.HILLS);
046:         }
047:         for (int i=0; i<(area/40); i++) {
048:           Point ts=map.findFreeSquare(x1,y1,x2,y2);
049:           map.addThing(GameScenery.create(1),ts.x,ts.y);
050:         }
051:         for (int i=0; i<(area/80); i++) {
052:           Point ts=map.findFreeSquare(x1,y1,x2,y2);
053:           map.addThing(GameScenery.create(3),ts.x,ts.y);
054:         }
055:         break;
056:       }
057: 
058:       case Tile.SWAMP: {
059:         map.setTheme("swamp");
060:         map.fillArea(x1,y1,x2,y2,map.floortile); 
061:         
062:         for (int lx=x1; lx<=x2; lx+=8) for (int ly=y1; ly<=y2; ly+=8) {
063:           int r=RPG.pick(new int[] {Tile.PLAINS,map.floortile,map.floortile});
064:           map.setTile(lx,ly,r);
065:         }
066:         map.fractalize(x1,y1,x2,y2,8);
067:         
068:         //for (int i=0; i<(area/10); i++) {
069:         //  int px=x1+RPG.r(w);
070:         // int py=y1+RPG.r(h);
071:         //  map.setTile(px,py,Tile.CAVEFLOOR);
072:         //}
073:         for (int i=0; i<(area/40); i++) {
074:           Point ts=map.findFreeSquare(x1,y1,x2,y2);
075:           map.addThing(GameScenery.create(3),ts.x,ts.y);
076:         }
077:         break;
078:       }
079: 
080:       case Tile.FORESTS: {
081:         map.setTheme("woods");
082:         map.fillArea(x1,y1,x2,y2,map.floortile); 
083:         for (int i=0; i<(area/20); i++) {
084:           Point ts=map.findFreeSquare(x1,y1,x2,y2);
085:           map.addThing(GameScenery.create(3),ts.x,ts.y);
086:         }
087:         for (int i=0; i<(area/80); i++) {
088:           Point ts=map.findFreeSquare(x1,y1,x2,y2);
089:           map.addThing(GameScenery.create(5),ts.x,ts.y);
090:         }
091:         break;
092:       }
093: 
094:       default: {
095:         map.fillArea(x1,y1,x2,y2,t); 
096:         break;
097:       }
098:     }
099:     
100:     {
101:       Point ts=map.findFreeSquare(x1,y1,x2,y2);
102:       addBeasties(map,ts.x,ts.y);
103:     }
104:      
105:   } 
106:   
107:   // add outdoor critters
108:   public static void addBeasties(Map map, int x, int y) {
109:     Creature m=Creature.createCreature(Creature.spawn,map.getLevel());
110:     for (int i=RPG.d(Creature.numbers[m.creaturetype]); i>0; i--) {
111:       int bx=x-2+RPG.r(5);
112:       int by=y-2+RPG.r(5);
113:       if (map.isClear(bx,by)) {
114:         map.addThing(m.cloneType(),bx,by);
115:       }  
116:     }
117:   }
118: 
119:   // can exit if there are no nearby hostiles
120:   public boolean canExit() {
121:     Hero h=Game.hero;
122:     Mobile m=getNearbyMobile(h.x,h.y,10);
123:     if ((m==null)||!(h.isHostile(m))) {
124:       return true;
125:     } else {
126:       Game.message("You cannot leave with enemy near!");
127:       return false; 
128:     }
129:   }

134:   }

138:     return "Somewhere in the Wild";
