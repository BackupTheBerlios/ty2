01: package rl;
02: 
03: public class StringList {
04:   private String[] strings=new String[3];
05:   private int size=3;
06:   private int count=0;  
07:   private static final int delta=3;
08:   
09:   public void ensureSize(int s) {
10:     if (s>size) {
11:       s=s+delta;
12:       String[] newstrings = new String[s];
13:       for (int i=0; i<count; i++) newstrings[i]=strings[i];
14:       size=s;
15:       strings=newstrings; 
16:     }
17:   }
18:   
19:   public int getCount() {
20:     return count; 
21:   }
22:   
23:   public void add(String s) {
24:     ensureSize(count+1);
25:     strings[count]=s;
26:     count=count+1; 
27:   }
28:   
29:   public void setString(int i, String s) {
30:     if ((i<0)||(i>=count)) return;
31:     if (s==null) throw new NullPointerException();
32:     strings[i]=s; 
33:   }
34:   
35:   public String getString(int i) {
36:     if ((i<0)||(i>=count)) return null;
37:     return strings[i];
38:   }
39:   
40:   public String getText(String delimiter) {
41:     if (count<=0) return "";
42:     StringBuffer sb=new StringBuffer(strings[0]);
43:     for (int i=1; i<count; i++) {
44:       sb.append(delimiter);
45:       sb.append(strings[i]);
46:     }
47:     return sb.toString();
48:   }
49:   
50:   public void insert(String s, int p) {
51:     if (s==null) throw new NullPointerException();
52:     ensureSize(count+1);
53:     if (p<0) p=0;
54:     if (p>count) p=count;
55:     if (count>p) System.arraycopy(strings,p,strings,p+1,count-p);
56:     strings[p]=s;
57:     count++;
58:   }
59:   
60:   public void swap(int a, int b) {
61:     if ((a>=0)&&(b>=0)&&(a<count)&&(b<count)) {
62:       String s=strings[a];
63:       strings[a]=strings[b];
64:       strings[b]=s; 
65:     } 
66:   }
67:   
68:   // sort the strings
69:   public void sort() {
70:     boolean end=false;
71:     while (end==false) {
72:       end=true;
73:       for (int i=0; i<(count-1); i++) {
74:         if (strings[i].compareTo(strings[i+1])>0) {
75:           swap(i,i+1);
76:           end=false;
77:         }
78:       } 
79:     }
80:   }
81:   
82:   // compresses identical items and adds count
83:   public StringList compress() {
84:     sort();
85:     StringList s=new StringList();
86:     if (count<=0) return s;
87:     int c=1;
88:     String d=strings[0];
89:     for (int i=1; i<count; i++) {
90:       if (strings[i].compareTo(d)!=0) {
91:         s.add((c==1)?d:d+" (x"+Integer.toString(c)+")"); 
92:         d=strings[i];
93:         c=0;
94:       }
95:       c++;
96:     }
97:     s.add((c==1)?d:d+" (x"+Integer.toString(c)+")"); 
98:     return s;
99:   }
100: }
