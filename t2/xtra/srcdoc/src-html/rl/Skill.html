052:   // gets list of skills trainable by being b
053:   public static String[] getTrainableSkills(Being a, Being b) {
054:     int maxcost=a.getStat(RPG.ST_SKILLPOINTS);
055:     
056:     // make temporary array with space for all the skills
057:     int skillcount=names.length;
058:     String[] temp=new String[skillcount];
059:     int count=0;
060:     
061:     for (int i=0; i<skillcount; i++) {
062:       int code=stats[i];
063:       if (b.arts.getModifier(code)>a.arts.getModifier(code)) {
064:         if ((maxcost>=costs[i])&&canLearnSkill(a,i)) {
065:           temp[count]=names[i];
066:           count++;
067:         }
068:       }
069:     }
070:     String[] result=new String[count];
071:     System.arraycopy(temp,0,result,0,count);
072:     return result;
073:   }

076:   public static int increaseSkill(Being a, String skill, int inc) {
077:     int i=Text.index(skill,names);
078:     Skill s=a.arts.getSkill(stats[i]);
079:     if (s==null) {
080:       a.arts.addArt(new Skill(stats[i],inc));
081:     } else {
082:       s.setLevel(s.getLevel()+inc);
083:     }
084:     int cost=inc*costs[i];
085:     a.setStat(RPG.ST_SKILLPOINTS,a.getStat(RPG.ST_SKILLPOINTS)-cost);
086:     
087:     if (a==Game.hero) Game.message("You improve you ability in "+skill);
088:     
089:     return cost;
090:   }

120:   // returns any attribute modifiers associated with the skill
121:   public int getModifier(int s) {
122:     if (s==stats[skill]) return level;
123:     
124:     int result=0;
125:     switch (s) {
126:       case RPG.ST_MOVESPEED:
127:         if (code==RPG.ST_ATHLETICS) result += level*5;
128:         if (code==RPG.ST_DODGE)     result += level*1;
129:         break;
130:         
131:       case RPG.ST_ATTACKSPEED:
132:         if (code==RPG.ST_FEROCITY)  result += level*10;
133:         if (code==RPG.ST_ATTACK)    result += level*2;
134:         break;
135: 
136:       case RPG.ST_ASKMULTIPLIER:
137:         if (code==RPG.ST_ATTACK)  result += level*15;
138:         break;
139:         
140:       case RPG.ST_DSKMULTIPLIER:
141:         if (code==RPG.ST_DEFENCE)  result += level*15;
142:         break;
143:         
144:       case RPG.ST_DODGEMULTIPLIER:
145:         if (code==RPG.ST_DODGE)  result += level*20;
146:         if (code==RPG.ST_DEFENCE)  result += level*3;
147:         if (code==RPG.ST_ATHLETICS)  result += level*5;
148:         break;
149: 
150:       case RPG.ST_ASKBONUS:
151:         if (code==RPG.ST_ATTACK)  result += level*1;
152:         if (code==RPG.ST_BRAVERY)  result += level*2;
153:         break;
154: 
155:       case RPG.ST_DSKBONUS:
156:         if (code==RPG.ST_DEFENCE)  result += level*1;
157:         if (code==RPG.ST_DODGE)    result += level*1;
158:         break;
159: 
160:       case RPG.ST_REGENERATE:
161:         if (code==RPG.ST_HEALING)  result += level/2;
162:         break;
163: 
164:       case RPG.ST_RECHARGE:
165:         if (code==RPG.ST_MEDITATION)  result += level;
166:         break;
167:         
168:       case RPG.ST_PROTECTION:
169:         if (code==RPG.ST_DEFENCE) result += level;
170:         if (code==RPG.ST_DODGE) result += level;
171:         break;
172:     }
173:     return result;  
174:   }

187:           
188:           int skill=(h.getStat(RPG.ST_SK)*level)/2;
189:                   
190:           if (RPG.test(skill,b.getStat(RPG.ST_SK))) {
191:             Thing[] stuff=b.inv.getContents(Item.class);
192:             if (stuff!=null) {
193:               Thing nick=stuff[RPG.r(stuff.length)];
194:               if (nick.y==0) {

196:                 h.addThing(nick);

203:           } else if (RPG.test(skill,b.getStat(RPG.ST_IN))) {
