057:   public int offerTeaching(Thing t) {
058:     if (!(t instanceof Being)) return 0;
059:     
060:     Being b=(Being) t;
061:     Being h=Game.hero;
062:     
063:     String[] options=Skill.getTrainableSkills(h,b);
064:     
065:     String choice=Game.selectString("Choose a skill to train:",options);
066:     
067:     if (choice==null) return 0;
068:     
069:     return Skill.increaseSkill(h,choice,1);    
070:   }
071: 
072:   public void talk(Thing t) {
073:     talk(t,type); 
074:   }
075:   
076:   public void talk(Thing t, int p) {
077:     Hero h=Game.hero;
078:     try { 
079:       switch (p) {
080:         case CHATTER:
081:           Game.message(chattercomments[subtype][RPG.r(chattercomments[subtype].length)]); 
082:           return;        
083:         
084:         case TEACHER:
085:           if (Skill.getTrainableSkills(h,(Being)t).length>0) {
086:             Game.message("I see you have potential");
087:             Game.message("Would you like to enhance your skills? (y/n)");
088:             if (Game.getOption("yn")=='y') {
089:               offerTeaching(t);
090:             } else {
091:               Game.message("Fare thee well!"); 
092:             }
093:           } else {
094:             Game.message("There is nothing I can teach you now."); 
095:           }
096:           return;
097:           
098:         default: 
099:           Game.message("You get no response."); 
100:       } 
101:     } catch (Exception e) {
102:       e.printStackTrace(); 
103:       Game.message("Mumble mumble mumble"); 
104:     }
105:   }
