001: package rl;
002: 
003: import java.awt.*;
004: import java.applet.*;
005: import java.awt.event.*;
006: 
007: public class CoastApp extends Applet {
008:   
009:   // map size: must be a power of 2
010:   public static final int size=128;
011:     
012:   // allocate map storage
013:   public int[] map= new int[size*size];
014:     
015:   public void paint(Graphics g) {
016:   super.paint(g);
017:         
018:   // draw the map
019:     for (int y=0; y<size; y++) for (int x=0; x<size; x++) {
020:       g.setColor( (getCell(x,y)==0) ? Color.blue : Color.green );
021:       g.fillRect(x*2,y*2,2,2);
022:     }
023:   }
024: 
025:   public void setCell(int x, int y, int c) {
026:     map[x+size*y]=c;    
027:   }
028:   
029:   public int getCell(int x, int y) {
030:     return map[x+size*y];    
031:   }
032:   
033:   // this routine creates the fractal landscape
034:   public void createFractal() {
035:     // initial pattern
036:     setCell(0,0,0);
037:     setCell(size/2,0,0);
038:     setCell(0,size/2,0);
039:     setCell(size/2,size/2,1);
040:       
041:     // call the fractal process
042:       makeFractal(size/4);   
043:     
044:     // redraw the screen
045:     repaint();
046:   }
047:   
048:   // this routine implements the fractal building process
049:   public void makeFractal(int step) {
050:     for (int y=0; y<size; y+=step) {
051:       for (int x=0; x<size; x+=step) {
052:         // add random offsets
053:         int cx=x+ ((Math.random()<0.5) ? 0 : step);
054:         int cy=y+ ((Math.random()<0.5) ? 0 : step);
055:                
056:         // round to nearest multiple of step*2
057:         cx=(cx/(step*2))*step*2;
058:         cy=(cy/(step*2))*step*2;
059:               
060:         // wrap around if needed
061:         cx=cx%size; 
062:         cy=cy%size; 
063:               
064:         // copy from (cx,cy)
065:         setCell(x,y,getCell(cx,cy));
066:       }
067:     }
068:         
069:     // recursiveley calculate finer detail levels
070:     if (step>1) makeFractal(step/2);
071:   }
072:   
073:   // applet initialisation
074:   public void init() {
075:     super.init();
076:   
077:     // build the first fractal
078:     createFractal();
079:       
080:     // rebuild on mouse clicks
081:     addMouseListener(new MouseAdapter() {
082:       public void mouseClicked( MouseEvent e ) {
083:         createFractal();    
084:       }
085:     });    
086:   }
087:     
088:   // main function allows applet to run as an application
089:   public static void main(String[] args) {
090:     
091:     // create a frame
092:     Frame f = new Frame("Fractal Coastlines");
093:     f.addWindowListener(new WindowAdapter() {
094:       public void windowClosing(WindowEvent e) {
095:         System.exit(0);
096:       }
097:     });
098:     
099:     // set up frame and add applet
100:     f.setSize(300,300);
101:     f.setLayout(new BorderLayout());
102:     CoastApp q=new CoastApp();
103:     q.init();
104:     f.add(q);
105:     
106:     // go live....
107:     f.show();
108:   }
109: }
