001: // Extension of portal class which creates maps when they are entered
002: // quite useful since sub-maps only get generated if entered
003: // This improves speed and save game filesize.
004: //
005: // Each type of PlacePortal creates a specific kind of target map
006: //
007: // Once the target place map is created, it will not be deleted
008: // i.e. the same location can be entered again and again.
009: //
010: // Typically, new PlacePortals will be created either:
011: //  - At the generation of the map (or WorldMap) on which they are placed
012: //      e.g. the initial towns on the starting WorldMap
013: //  - As a result of being awarded a Quest
014: //  - As a result of completing part or all of a quest
015: 
016: package rl;
017: 
018: public class PlacePortal extends Portal {
019:   public static final int TOWN=1;
020:   public static final int RUIN=2;
021:   public static final int BANDITCAVES=3;
022:   public static final int DARKCAVES=4;
023:   public static final int BANDITCAVESDEEP=5;
024:   public static final int BANDITLAIR=6;
025:   public static final int GOBLINTRIBE=7;
026:   
027:   
028:   private int type;
029:   private int level;
030:   private Map map;
031:   
032:   public static final int[] images= {0,305,302,303,-1,3,3,304};  
033:   
034:   public PlacePortal(int t) {
035:     super(1);
036:     type=t;
037:   }
038:   
039:   public PlacePortal(int t, int l) {
040:     this(t);
041:     level=l;
042:   }
043:   
044:   public void travel(Thing t) {
045:     try {
046:       if (map==null) {
047:         switch (type) {
048:         
049:           case TOWN:
050:             map=new Town(51,51);
051:             makeLink(map);
052:             break;
053:         
054:           case RUIN:
055:             map=new DeepForest(71,71);
056:             makeLink(map);
057:             break;  
058:           
059:           
060:           
061:           case BANDITCAVES:
062:             map=new Caves(1,4);
063:             makeLink(map);
064:             map.addThing(new PlacePortal(BANDITCAVESDEEP,5));
065:             break;
066: 
067:           case DARKCAVES:
068:             map=new Caves(1,Game.hero.getStat(RPG.ST_LEVEL));
069:             map.entrance=new Portal(1);
070:             map.entrance.setDestination(getMap(),x,y);
071:             setDestination(map,64,32);
072:             break;
073:           
074:           case BANDITCAVESDEEP:
075:             map=new Caves(2,level);
076:             makeLink(map);
077:             if (RPG.d(2)==1) {
078:               map.addThing(new PlacePortal(BANDITCAVESDEEP,level+1));
079:             } else {
080:               map.addThing(new PlacePortal(BANDITLAIR,7));
081:             }
082:             break;
083:           
084:           case BANDITLAIR:
085:             map=new Caves(3,level);
086:             makeLink(map);
087:             map.addThing(new PlacePortal(BANDITCAVESDEEP,8));
088:             break;
089:         
090:           case GOBLINTRIBE: {
091:             map = new Outdoors(Tile.FORESTS);
092:             
093:             for (int i=0; i<10; i++) {
094:               map.addThing(GameScenery.create("tent")); 
095:             }
096:             
097:             map.entrance.remove();
098:             EdgePortal e=new EdgePortal(EdgePortal.ALL);
099:             map.entrance=e;
100:             map.addThing(e,0,0);
101:             
102:             
103:             // define the central area
104:             int p1=22; int q1=22;
105:             int p2=42; int q2=42;
106:             map.clearArea(p1,q1,p2,q2);
107:             
108:             
109:             for (int i=RPG.d(4,10); i>0; i--) {
110:               Being b=new Creature(Creature.GOBBO,2+RPG.po(3));               
111:               b.setPersonality(new Personality(Personality.CHATTER,Personality.CHATTER_GOBLIN));
112:               b.setAI(TownieAI.instance);
113:               b.setStat(RPG.ST_STATE,AI.STATE_INHABITANT);
114:               map.addThing(b);
115:             } 
116:             
117:             
118:             for (int i=RPG.d(4,10); i>0; i--) {
119:               Being b=new Creature(Creature.GOBLIN,4+RPG.po(2));               
120:               b.setPersonality(new Personality(Personality.CHATTER,Personality.CHATTER_GOBLIN));
121:               b.setAI(TownieAI.instance);
122:               b.setStat(RPG.ST_STATE,AI.STATE_INHABITANT);
123:               map.addThing(b);
124:             } 
125:             
126:             for (int i=RPG.d(3,4); i>0; i--) {
127:               Being b=new Creature(Creature.ORCWARRIOR,5+RPG.po(3));               
128:               b.setPersonality(new Personality(Personality.CHATTER,Personality.CHATTER_ORC));
129:               b.makeGuard(map,p1,q1,p2,q2);
130:               b.setStat(RPG.ST_STATE,AI.STATE_INHABITANT);
131:             } 
132:             
133:             for (int i=RPG.d(2,3); i>0; i--) {
134:               Being b=new Creature(Creature.SHAMAN,8+RPG.po(3));               
135:               b.setPersonality(new Personality(Personality.CHATTER,Personality.CHATTER_GOBLIN));
136:               b.makeGuard(map,p1,q1,p2,q2);
137:               b.setStat(RPG.ST_STATE,AI.STATE_INHABITANT);
138:             } 
139: 
140:             
141: 
142:             
143:             makeLink(map);
144:             break;
145:             
146:           } 
147:         }
148:       }
149:     } catch (Exception e) {
150:       Game.message("Failed to build target map in PlacePortal");
151:       return; 
152:     }
153:     super.travel(t);
154:   }
155:   
156:   public boolean isInvisible() {return false;}
157:   
158:   public int getImage() {
159:     return images[type]; 
160:   }
161: }
