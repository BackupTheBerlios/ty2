658:   // get location, initially place crosshairs at start
659:   public Point getTargetLocation(Thing start) {
660:     if (start==null) return getTargetLocation();
661:     mappanel.setCursor(start.x,start.y);
662:     mappanel.viewPosition(start.x,start.y);
663:     
664:     // repaint the status panel
665:     statuspanel.repaint();
666:     
667:     while (true) {
668:       KeyEvent e=Game.getInput(); 
669:       if (e==null) continue;
670:       
671:       char k = Character.toLowerCase(e.getKeyChar());
672:       
673:       int i=e.getKeyCode();
674:       
675:       // handle key conversions
676:       if (i==e.VK_UP) k='8';   
677:       if (i==e.VK_DOWN) k='2'; 
678:       if (i==e.VK_LEFT) k='4';   
679:       if (i==e.VK_RIGHT) k='6';         
680:       if (i==e.VK_HOME) k='7';   
681:       if (i==e.VK_END) k='1';   
682:       if (i==e.VK_PAGE_UP) k='9';   
683:       if (i==e.VK_PAGE_DOWN) k='3';   
684:       if (i==e.VK_ESCAPE) k ='q'; 
685: 
686:       int dx=0; int dy=0;
687:       if (k=='8') { dx=0; dy=-1; }
688:       if (k=='2') { dx=0; dy=1; }
689:       if (k=='4') { dx=-1; dy=0; }
690:       if (k=='6') { dx=1; dy=0; }
691:       if (k=='7') { dx=-1; dy=-1; }
692:       if (k=='9') { dx=1; dy=-1; }
693:       if (k=='1') { dx=-1; dy=1; }
694:       if (k=='3') { dx=1; dy=1; }
695:       
696:       if (k=='q') {
697:         mappanel.clearCursor();
698:         return null;
699:       }
700:       
701:       if (k==' ') {
702:         mappanel.clearCursor();
703:         return new Point(mappanel.curx,mappanel.cury);
704:       }
705:       
706:       mappanel.setCursor(mappanel.curx+dx,mappanel.cury+dy);
707:       mappanel.viewPosition(mappanel.curx,mappanel.cury);
708:     }
709:   }

726:   public void castSpell(final Spell s) {
727:     final Hero h=Game.hero;
728:     if (s==null) return;
729:     
730:     Game.message("You cast "+s.getName());
731:     
732:     switch (s.getTarget()) {
733:       case Spell.TARGET_SELF:
734:         s.castAtSelf(h);
735:         break;
736:       case Spell.TARGET_LOCATION:
737:         Thing f=(s.getUsage()==Spell.SPELL_OFFENCE)?map.findNearestFoe(h):null;
738:         if ((f!=null)&&(!map.isVisible(f.x,f.y))) f=null;
739:         Point p=getTargetLocation(f);
740:         if (p!=null) {
741:           // don't fire offensive spell at self by accident
742:           if ((p.x==h.x)&&(p.y==h.y)&&(s.getUsage()==Spell.SPELL_OFFENCE)) {
743:             Game.message("Are you sure you want to target yourself? (y/n)");
744:             char opt=Game.getOption("yn");
745:             if (opt=='n') break; 
746:           }
747:           
748:           if (map.isVisible(p.x,p.y)) {
749:             s.castAtLocation(h,map,p.x,p.y);
750:           } else {
751:             Game.message("You cannot see to focus your power");
752:           }
753:         }
754:         break;
755:        case Spell.TARGET_ITEM:
756:         Thing t = getInventoryItem(s.getName()+":");
757:         questapp.switchScreen(this);
758:         if (t!=null) {
759:            s.castAtObject(h,t);
760:         }
761:         break;
762:     }  
763:   }

765:   public void endTurn() {
766:     Hero h = Game.hero;
767:     
768:     if (h.getMap()!=map) {
769:       map=h.getMap();
770:       if (map==null) return;
771:       Game.enterMap(map,h.x,h.y); 
772:     }
773:     
774:     // Game.message(Integer.toString(h.aps));
775:     
776:     // test for adjacent foe
777:     Thing f=map.findNearestFoe(h);
778:     if ((f!=null)&&(f.x>=(h.x-1))&&(f.x<=(h.x+1))&&(f.y>=(h.y-1))&&(f.y<=(h.y+1))) {
779:       h.recharging=false;
780:     } else {
781:       h.recharging=true; 
782:     }
783:     
784:     Game.actor=null; // player's move is over
785:     
786:     int advance=-h.aps;
787:     while (advance>0) {
788:       int step=advance;
789:       if (step>500) step=500;
790:       
791:       // call map action
792:       // this includes increment of h.aps via Mobile.action(t)
793:       map.action(step);
794:       advance-=step;
795:       Game.time=Game.time+step;
796:     }
797:     Game.actor=Game.hero; // hero is now active
798:     
799:     // update screen info
800:     mappanel.viewPosition(h.x,h.y);  
801:     statuspanel.repaint();
802:     messagepanel.repaint();
803:   }
