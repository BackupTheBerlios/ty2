01: package rl;
02: 
03: // in transition to ItemType
04: 
05: public class Coin extends Stack {
06:   
07:   public static final Description DESC_COPPER = 
08:     new Describer("copper coin","A common copper coin");
09:   public static final Description DESC_SILVER=
10:     new Describer("silver coin","A shiny silver coin");
11:   public static final Description DESC_GOLD=
12:     new Describer("gold coin","A gleaming gold coin");
13:   public static final Description DESC_SOVEREIGN=
14:     new Describer("sovereign","An imperial gold sovereign");
15: 
16:   public static final int[] COINVALUES={1,10,100,1000};
17:   public static final int[] IMAGES={144,143,140,140};
18:   public static final Description[] DESCS={DESC_COPPER,DESC_SILVER,DESC_GOLD,DESC_SOVEREIGN};
19:     
20:   public static final int COPPER=    0;
21:   public static final int SILVER=    1;
22:   public static final int GOLD=      2;
23:   public static final int SOVEREIGN= 3;
24: 
25:   public Coin(int t,int q) {
26:     super(q);
27:     type=t;
28:   }
29:   
30:   public static String valueString(int v) {
31:     int t=0;
32:     while ((t<2)&&(v>0)&&((v%10)==0)) {
33:       t++;
34:       v/=10;
35:     }
36:     if (v<=0) return "nothing";
37:     switch (t) {
38:       case 0: return v+" copper";
39:       case 1: return v+" silver";
40:       case 2: return v+" gold";
41:       default: return "nothing";
42:     }
43:   }
44:   
45:   // stack of approximate value
46:   public static Coin createMoney(int v) {
47:     int type=0;
48:     int number=RPG.d(2,v);
49:     
50:     if ((number>=1000)&&(RPG.d(4)==1)) {
51:       type++;
52:     }
53:     if (number>=100) {
54:       type++;
55:     }
56:     if ((RPG.d(2)==1)&&(number>=10)) {
57:       type++;
58:     }
59:     for (int i=0; i<type; i++) number/=10;
60: 
61:     if (number<=0) number=1;
62:     
63:     if (type==1) return new Coin(SILVER,number);
64:     if (type==2) return new Coin(GOLD,number);
65:     if (type>=3) return new Coin(SOVEREIGN,number);
66:     return new Coin(COPPER,number);
67:   }
68:   
69:   public int getStat(int s) {
70:     switch(s) {
71:       case RPG.ST_ITEMVALUE: return COINVALUES[type];
72:       default: return super.getStat(s); 
73:     } 
74:   }
75:   
76:   public int getWeight() {
77:     return 10*number;
78:   }
79:   
80:   public int getUse() {return Thing.USE_MONEY;}
81:   
82:   public int getImage() {
83:     return IMAGES[type];
84:   }
85:   
86:   public Description getDescription() {
87:     return DESCS[type];
88:   }
89: }
