03: public class GuardianVault extends Vault {
04:   public Being guardian;
05:   
06:   public int visibleRange() {return 4;}
07:   
08:   public GuardianVault(Being guard, Thing treasure) {
09:     super(21,21);
10:   
11:     guardian=guard;
12:     entrance=new Portal("stairs up");
13:   
14:     switch(RPG.d(1)) {
15:       case 1: {
16:         fillArea(0,0,20,20,Tile.CAVEWALL);
17:         fillArea(6,5,14,15,Tile.CAVEFLOOR);
18:   
19:         for (int y=6; y<16; y=y+2) {
20:           setTile(7,y,Tile.CAVEWALL);
21:           setTile(13,y,Tile.CAVEWALL);
22:         }
23:   
24:         guardian.addThing(treasure); 
25:   
26:         addThing(guardian);
27:         addThing(entrance);
28:         break;
29:       }
30:       
31:       case 2: {
32:         makeRandomPath(0,0,20,20,0,0,20,20,Tile.CAVEFLOOR,false);
33:         for (int i=0; i<10; i++) {
34:           Point ss=findFreeSquare();
35:           Point fs=findFreeSquare();
36:           makeRandomPath(ss.x,ss.y,fs.x,fs.y,0,0,20,20,Tile.CAVEFLOOR,false);
37:         }
38:         replaceTiles(0,Tile.CAVEWALL);
39:         fillBorder(0,0,20,20,Tile.CAVEWALL);
40:         
41:         guardian.addThing(treasure); 
42:         Point es=findFreeSquare();
43:         addThing(entrance,es.x,es.y);
44:         Point gs=findFreeSquare();
45:         addThing(guardian,gs.x,gs.y);
46:       }  
47:     }
48:   }
49: 
50:   public String getEnterMessage() {
51:     return "This place seems to be a lair of some sort.";  
52:   }
53:   
54:   public String getLevelName() {
55:     return guardian.isVisible()?"Lair of "+guardian.getTheName():"A dark lair";  
56:   }
57: }
